32 /*!
33  * \brief Kerr black hole in Kerr-Schild coordinates
34  *
35  * \details
36  * The metric is \f$g_{\mu\nu} = \eta_{\mu\nu} + 2 H l_\mu l_\nu\f$,
37  * where \f$\eta_{\mu\nu}\f$ is the Minkowski metric, \f$H\f$ is a scalar
38  * function, and \f$l_\mu\f$ is the outgoing null vector.
39  * \f$H\f$ and \f$l_\mu\f$ are known functions of the coordinates
40  * and of the mass and spin vector.
41  *
42  * The following are input file options that can be specified:
43  * - Mass (default: 1.)
44  * - Center (default: {0,0,0})
45  * - Spin (default: {0,0,0})
46  *
47  * ## Kerr-Schild Coordinates
48  *
49  *
50  * A Kerr-Schild coordinate system is defined by
51  * \f{equation}{
52  * g_{\mu\nu} \equiv \eta_{\mu\nu} + 2 H l_\mu l_\nu,
53  * \f}
54  * where \f$H\f$ is a scalar function of the coordinates, \f$\eta_{\mu\nu}\f$
55  * is the Minkowski metric, and \f$l^\mu\f$ is a null vector. Note that the
56  * form of the metric along with the nullness of \f$l^\mu\f$ allows one to
57  * raise and lower indices of \f$l^\mu\f$ using \f$\eta_{\mu\nu}\f$, and
58  * that \f$l^t l^t = l_t l_t = l^i l_i\f$.
59  * Note also that
60  * \f{equation}{
61  * g^{\mu\nu} \equiv \eta^{\mu\nu} - 2 H l^\mu l^\nu,
62  * \f}
63  * and that \f$\sqrt{-g}=1\f$.
64  * Also, \f$l_\mu\f$ is a geodesic with respect to both the physical metric
65  * and the Minkowski metric:
66  * \f{equation}{
67  * l^\mu \partial_\mu l_\nu = l^\mu\nabla_\mu l_\nu = 0.
68  * \f}
69  *
70  *
71  * The corresponding 3+1 quantities are
72  * \f{eqnarray}{
73  * g_{i j} &=& \delta_{i j} + 2 H l_i l_j,\\
74  * g^{i j} &=& \delta^{i j} - {2 H l^i l^j \over 1+2H l^t l^t},\\
75  * {\rm det} g_{i j}&=& 1+2H l^t l^t,\\
76  * \beta^i &=& - {2 H l^t l^i \over 1+2H l^t l^t},\\
77  * N &=& \left(1+2 H l^t l^t\right)^{-1/2},\quad\hbox{(lapse)}\\
78  * \alpha &=& \left(1+2 H l^t l^t\right)^{-1},
79  * \quad\hbox{(densitized lapse)}\\
80  * K_{i j} &=& - \left(1+2 H l^t l^t\right)^{1/2}
81  * \left[l_i l_j \partial_{t} H + 2 H l_{(i} \partial_{t} l_{j)}\right]
82  * \nonumber \\
83  * &&-2\left(1+2 H l^t l^t\right)^{-1/2}
84  * \left[H l^t \partial_{(i}l_{j)} + H l_{(i}\partial_{j)}l^t
85  * + l^t l_{(i}\partial_{j)} H + 2H^2 l^t l_{(i} l^k\partial_{k}l_{j)}
86  * + H l^t l_i l_j l^k \partial_{k} H\right],\\
87  * \partial_{k}g_{i j}&=& 2 l_i l_j\partial_{k} H +
88  * 4 H l_{(i} \partial_{k}l_{j)},\\
89  * \partial_{k}N &=& -\left(1+2 H l^t l^t\right)^{-3/2}
90  * \left(l^tl^t\partial_{k}H+2Hl^t\partial_{k}l^t\right),\\
91  * \partial_{k}\beta^i &=& - 2\left(1+2H l^t l^t\right)^{-1}
92  * \left(l^tl^i\partial_{k}H+Hl^t\partial_{k}l^i+Hl^i\partial_{k}l^t\right)
93  * + 4 H l^t l^i \left(1+2H l^t l^t\right)^{-2}
94  * \left(l^tl^t\partial_{k}H+2Hl^t\partial_{k}l^t\right),\\
95  * \Gamma^k{}_{i j}&=& -\delta^{k m}\left(l_i l_j \partial_{m}H
96  * + 2l_{(i} \partial_{m}l_{j)} \right)
97  * + 2 H l_{(i}\partial_{j)} l^k
98  * \nonumber \\
99  * &&+\left(1+2 H l^t l^t\right)^{-1}
100  * \left[2 l^k l_{(i}\partial_{j)} H
101  * +2 H l_i l_j l^k l^m \partial_{m}H
102  * +2 H l^k \partial_{(i}l_{j)}
103  * +4 H^2 l^k l_{(i} (l^m \partial_{m}l_{j)}
104  * -\partial_{j)} l^t)
105  * \right].
106  * \f}
107  * Note that \f$l^i\f$ is **not** equal to \f$g^{i j} l_j\f$; it is equal
108  * to \f${}^{(4)}g^{i \mu} l_\mu\f$.
109  *
110  * ## Kerr Spacetime
111  *
112  * ### Spin in the z direction
113  *
114  * Assume Cartesian coordinates \f$(t,x,y,z)\f$. Then for stationary Kerr
115  * spacetime with mass \f$M\f$ and angular momentum \f$a M\f$
116  * in the \f$z\f$ direction,
117  * \f{eqnarray}{
118  * H &=& {M r^3 \over r^4 + a^2 z^2},\\
119  * l_\mu &=&
120  * \left(1,{rx+ay\over r^2+a^2},{ry-ax\over r^2+a^2},{z\over r}\right),
121  * \f}
122  * where \f$r\f$ is defined by
123  * \f{equation}{
124  * \label{eq:rdefinition1}
125  * {x^2+y^2\over a^2+r^2} + {z^2\over r^2} = 1,
126  * \f}
127  * or equivalently,
128  * \f{equation}{
129  * r^2 = {1\over 2}(x^2 + y^2 + z^2 - a^2)
130  * + \left({1\over 4}(x^2 + y^2 + z^2 - a^2)^2 + a^2 z^2\right)^{1/2}.
131  * \f}
132  *
133  * Possibly useful formula:
134  * \f{equation}{
135  * \partial_{i} r = {x_i + z \delta_{i z} \displaystyle {a^2\over r^2} \over
136  * 2 r\left(1 - \displaystyle {x^2 + y^2 + z^2 - a^2\over 2 r^2}\right)}.
137  * \f}
138  *
139  * ### Spin in an arbitrary direction
140  *
141  * For arbitrary spin direction, let \f$\vec{x}\equiv (x,y,z)\f$ and
142  * \f$\vec{a}\f$ be a flat-space three-vector with magnitude-squared
143  * (\f$\delta_{ij}\f$ norm) equal to \f$a^2\f$.
144  * Then the Kerr-Schild quantities for Kerr spacetime are:
145  * \f{eqnarray}{
146  * H &=& {M r^3 \over r^4 + (\vec{a}\cdot\vec{x})^2},\\
147  * \vec{l} &=& {r\vec{x}-\vec{a}\times\vec{x}+(\vec{a}\cdot\vec{x})\vec{a}/r
148  * \over r^2+a^2 },\\
149  * l_t &=& 1,\\
150  * \label{eq:rdefinition2}
151  * r^2 &=& {1\over 2}(\vec{x}\cdot\vec{x}-a^2)
152  * + \left({1\over 4}(\vec{x}\cdot\vec{x}-a^2)^2
153  * + (\vec{a}\cdot\vec{x})^2\right)^{1/2},
154  * \f}
155  * where \f$\vec{l}\equiv (l_x,l_y,l_z)\f$, and
156  * all dot and cross products are evaluated as flat-space 3-vector operations.
157  *
158  * Possibly useful formulae:
159  * \f{equation}{
160  * \partial_{i} r = {x_i + (\vec{a}\cdot\vec{x})a_i/r^2 \over
161  * 2 r\left(1 - \displaystyle {\vec{x}\cdot\vec{x}-a^2\over 2 r^2}\right)},
162  * \f}
163  * \f{equation}{
164  * {\partial_{i} H \over H} =
165  * {3\partial_{i}r\over r} - {4 r^3 \partial_{i}r +
166  * 2(\vec{a}\cdot\vec{x})\vec{a}
167  * \over r^4 + (\vec{a}\cdot\vec{x})^2},
168  * \f}
169  * \f{equation}{
170  * (r^2+a^2)\partial_{j} l_i =
171  * (x_i-2 r l_i-(\vec{a}\cdot\vec{x})a_i/r^2)\partial_{j}r
172  * + r\delta_{ij} + a_i a_j/r + \epsilon^{ijk} a_k.
173  * \f}
174  *
175  * ## Cartesian and Spherical Coordinates for Kerr
176  *
177  * The Kerr-Schild coordinates are defined in terms of the Cartesian
178  * coordinates \f$(x,y,z)\f$. If one wishes to express Kerr-Schild
179  * coordinates in terms of the spherical polar coordinates
180  * \f$(\tilde{r},\theta,\phi)\f$ then one can make the obvious and
181  * usual transformation
182  * \f{equation}{
183  * \label{eq:sphertocartsimple}
184  * x=\tilde{r}\sin\theta\cos\phi,\quad
185  * y=\tilde{r}\sin\theta\sin\phi,\quad
186  * z=\tilde{r}\cos\theta.
187  * \f}
188  *
189  * This is simple, and has the advantage that in this coordinate system
190  * for \f$M\to0\f$, Kerr spacetime becomes Minkowski space in spherical
191  * coordinates \f$(\tilde{r},\theta,\phi)\f$. However, the disadvantage is
192  * that the horizon of a Kerr hole is **not** located at constant
193  * \f$\tilde{r}\f$, but is located instead at constant \f$r\f$,
194  * where \f$r\f$ is the radial
195  * Boyer-Lindquist coordinate defined in (\f$\ref{eq:rdefinition2}\f$).
196  *
197  * For spin in the \f$z\f$ direction, one could use the transformation
198  * \f{equation}{
199  * x=\sqrt{r^2+a^2}\sin\theta\cos\phi,\quad
200  * y=\sqrt{r^2+a^2}\sin\theta\sin\phi,\quad
201  * z=r\cos\theta.
202  * \f}
203  * In this case, for \f$M\to0\f$, Kerr spacetime becomes Minkowski space in
204  * spheroidal coordinates, but now the horizon is on a constant-coordinate
205  * surface.
206  *
207  * Right now we use (\f$\ref{eq:sphertocartsimple}\f$), but we may
208  * wish to use the other transformation in the future.
209  */

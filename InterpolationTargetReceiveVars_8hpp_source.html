51 // apply_callback accomplishes the overload for the
52 // two signatures of callback functions.
53 // Uses SFINAE on return type.
54 template <typename T, typename DbTags, typename Metavariables>
55 auto apply_callback(
56  const gsl::not_null<db::DataBox<DbTags>*> box,
57  const gsl::not_null<Parallel::ConstGlobalCache<Metavariables>*> cache,
58  const typename Metavariables::temporal_id::type& temporal_id) noexcept
59  -> decltype(T::post_interpolation_callback::apply(box, cache, temporal_id),
60  bool()) {
61  return T::post_interpolation_callback::apply(box, cache, temporal_id);
62 }

64 template <typename T, typename DbTags, typename Metavariables>
65 auto apply_callback(
66  const gsl::not_null<db::DataBox<DbTags>*> box,
67  const gsl::not_null<Parallel::ConstGlobalCache<Metavariables>*> cache,
68  const typename Metavariables::temporal_id::type& temporal_id) noexcept
69  -> decltype(T::post_interpolation_callback::apply(*box, *cache,
70  temporal_id),
71  bool()) {
72  T::post_interpolation_callback::apply(*box, *cache, temporal_id);
73  // For the simpler callback function, we will always clean up volume data, so
74  // we return true here.
75  return true;
76 }

92 // Fills invalid points with some constant value.
93 template <typename InterpolationTargetTag, typename DbTags,
94  Requires<not has_fill_invalid_points_with_v<InterpolationTargetTag>> =
95  nullptr>
96 void fill_invalid_points(
97  const gsl::not_null<db::DataBox<DbTags>*> /*box*/) noexcept {}

99 template <
100  typename InterpolationTargetTag, typename DbTags,
101  Requires<has_fill_invalid_points_with_v<InterpolationTargetTag>> = nullptr>
102 void fill_invalid_points(
103  const gsl::not_null<db::DataBox<DbTags>*> box) noexcept {
104  if (not db::get<Tags::IndicesOfInvalidInterpPoints>(*box).empty()) {
105  db::mutate<
106  Tags::IndicesOfInvalidInterpPoints,
107  ::Tags::Variables<
108  typename InterpolationTargetTag::vars_to_interpolate_to_target>>(
109  box, [](const gsl::not_null<
110  db::item_type<Tags::IndicesOfInvalidInterpPoints>*>
111  indices_of_invalid_points,
112  const gsl::not_null<db::item_type<
113  ::Tags::Variables<typename InterpolationTargetTag::
114  vars_to_interpolate_to_target>>*>
115  vars_dest) noexcept {
116  const size_t npts_dest = vars_dest->number_of_grid_points();
117  const size_t nvars = vars_dest->number_of_independent_components;
118  for (auto index : *indices_of_invalid_points) {
119  for (size_t v = 0; v < nvars; ++v) {
120  // clang-tidy: no pointer arithmetic
121  vars_dest->data()[index + v * npts_dest] = // NOLINT
122  InterpolationTargetTag::post_interpolation_callback::
123  fill_invalid_points_with;
124  }
125  }
126  // Further functions may test if there are invalid points.
127  // Clear the invalid points now, since we have filled them.
128  indices_of_invalid_points->clear();
129  });
130  }
131 }

133 /// Calls the callback function, tells interpolators to clean up the current
134 /// temporal_id, and then if there are more temporal_ids to be interpolated,
135 /// starts the next one.
136 template <typename InterpolationTargetTag, typename DbTags,
137  typename Metavariables>
138 void callback_and_cleanup(
139  const gsl::not_null<db::DataBox<DbTags>*> box,
140  const gsl::not_null<Parallel::ConstGlobalCache<Metavariables>*>
141  cache) noexcept {
142  const auto temporal_id =
143  db::get<Tags::TemporalIds<Metavariables>>(*box).front();
144 
145  // Before doing anything else, deal with the possibility that some
146  // of the points might be outside of the Domain.
147  fill_invalid_points<InterpolationTargetTag>(box);
148 
149  // apply_callback should return true if we are done with this
150  // temporal_id. It should return false only if the callback
151  // calls another `intrp::Action` that needs the volume data at this
152  // same temporal_id. If it returns false, we exit here and do not
153  // clean up.
154  const bool done_with_temporal_id =
155  apply_callback<InterpolationTargetTag>(box, cache, temporal_id);
156 
157  if (not done_with_temporal_id) {
158  return;
159  }
160 
161  // We are now done with this temporal_id, so we can pop it and
162  // clean up volume data associated with it.
163  db::mutate<Tags::TemporalIds<Metavariables>,
164  Tags::CompletedTemporalIds<Metavariables>>(
165  box, [](const gsl::not_null<
166  db::item_type<Tags::TemporalIds<Metavariables>>*>
167  ids,
168  const gsl::not_null<
169  db::item_type<Tags::CompletedTemporalIds<Metavariables>>*>
170  completed_ids) noexcept {
171  completed_ids->push_back(ids->front());
172  ids->pop_front();
173  // We want to keep track of all completed temporal_ids to deal with
174  // the possibility of late calls to
175  // AddTemporalIdsToInterpolationTarget. We could keep all
176  // completed_ids forever, but we probably don't want it to get too
177  // large, so we limit its size. We assume that
178  // asynchronous calls to AddTemporalIdsToInterpolationTarget do not span
179  // more than 10 temporal_ids.
180  if (completed_ids->size() > 10) {
181  completed_ids->pop_front();
182  }
183  });
184 
185  // Tell interpolators to clean up at this temporal_id for this
186  // InterpolationTargetTag.
187  auto& interpolator_proxy =
188  Parallel::get_parallel_component<Interpolator<Metavariables>>(*cache);
189  Parallel::simple_action<Actions::CleanUpInterpolator<InterpolationTargetTag>>(
190  interpolator_proxy, temporal_id);
191 
192  // If there are further temporal_ids, begin interpolation for
193  // the next one.
194  const auto& temporal_ids = db::get<Tags::TemporalIds<Metavariables>>(*box);
195  if (not temporal_ids.empty()) {
196  auto& my_proxy = Parallel::get_parallel_component<
197  InterpolationTarget<Metavariables, InterpolationTargetTag>>(*cache);
198  Parallel::simple_action<
199  typename InterpolationTargetTag::compute_target_points>(
200  my_proxy, temporal_ids.front());
201  }
202 }

207 /// \ingroup ActionsGroup
208 /// \brief Receives interpolated variables from an `Interpolator` on a subset
209 /// of the target points.
210 ///
211 /// If interpolated variables for all target points have been received, then
212 /// - Calls `InterpolationTargetTag::post_interpolation_callback`
213 /// - Tells `Interpolator`s that the interpolation is complete
214 /// (by calling
215 /// `Actions::CleanUpInterpolator<InterpolationTargetTag>`)
216 /// - Removes the first `temporal_id` from `Tags::TemporalIds<Metavariables>`
217 /// - If there are more `temporal_id`s, begins interpolation at the next
218 /// `temporal_id` (by calling `InterpolationTargetTag::compute_target_points`)
219 ///
220 /// Uses:
221 /// - DataBox:
222 /// - `Tags::TemporalIds<Metavariables>`
223 /// - `Tags::IndicesOfFilledInterpPoints`
224 /// - `::Tags::Variables<typename
225 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
226 ///
227 /// DataBox changes:
228 /// - Adds: nothing
229 /// - Removes: nothing
230 /// - Modifies:
231 /// - `Tags::TemporalIds<Metavariables>`
232 /// - `Tags::CompletedTemporalIds<Metavariables>`
233 /// - `Tags::IndicesOfFilledInterpPoints`
234 /// - `::Tags::Variables<typename
235 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
236 ///
237 /// For requirements on InterpolationTargetTag, see InterpolationTarget
238 template <typename InterpolationTargetTag>
239 struct InterpolationTargetReceiveVars {
240  /// For requirements on Metavariables, see InterpolationTarget
241  template <typename ParallelComponent, typename DbTags, typename Metavariables,
242  typename ArrayIndex,
243  Requires<tmpl::list_contains_v<
244  DbTags, Tags::TemporalIds<Metavariables>>> = nullptr>
245  static void apply(
246  db::DataBox<DbTags>& box,
247  Parallel::ConstGlobalCache<Metavariables>& cache,
248  const ArrayIndex& /*array_index*/,
249  const std::vector<db::item_type<::Tags::Variables<
250  typename InterpolationTargetTag::vars_to_interpolate_to_target>>>&
251  vars_src,
252  const std::vector<std::vector<size_t>>& global_offsets) noexcept {
253  db::mutate<
254  Tags::IndicesOfFilledInterpPoints,
255  ::Tags::Variables<
256  typename InterpolationTargetTag::vars_to_interpolate_to_target>>(
257  make_not_null(&box),
258  [
259  &vars_src, &global_offsets
260  ](const gsl::not_null<db::item_type<Tags::IndicesOfFilledInterpPoints>*>
261  indices_of_filled,
262  const gsl::not_null<db::item_type<::Tags::Variables<
263  typename InterpolationTargetTag::vars_to_interpolate_to_target>>*>
264  vars_dest) noexcept {
265  const size_t npts_dest = vars_dest->number_of_grid_points();
266  const size_t nvars = vars_dest->number_of_independent_components;
267  for (size_t j = 0; j < global_offsets.size(); ++j) {
268  const size_t npts_src = global_offsets[j].size();
269  for (size_t i = 0; i < npts_src; ++i) {
270  // If a point is on the boundary of two (or more)
271  // elements, it is possible that we have received data
272  // for this point from more than one Interpolator.
273  // This will rarely occur, but it does occur, e.g. when
274  // a point is exactly on some symmetry
275  // boundary (such as the x-y plane) and this symmetry
276  // boundary is exactly the boundary between two
277  // elements. If this happens, we accept the first
278  // duplicated point, and we ignore subsequent
279  // duplicated points. The points are easy to keep track
280  // of because global_offsets uniquely identifies them.
281  if (indices_of_filled->insert(global_offsets[j][i]).second) {
282  for (size_t v = 0; v < nvars; ++v) {
283  // clang-tidy: no pointer arithmetic
284  vars_dest->data()[global_offsets[j][i] + // NOLINT
285  v * npts_dest] = // NOLINT
286  vars_src[j].data()[i + v * npts_src]; // NOLINT
287  }
288  }
289  }
290  }
291  });
292 
293  if (db::get<Tags::IndicesOfFilledInterpPoints>(box).size() +
294  db::get<Tags::IndicesOfInvalidInterpPoints>(box).size() ==
295  db::get<::Tags::Variables<
296  typename InterpolationTargetTag::vars_to_interpolate_to_target>>(
297  box)
298  .number_of_grid_points()) {
299  // All the valid points have been interpolated.
300  InterpolationTarget_detail::callback_and_cleanup<InterpolationTargetTag>(
301  make_not_null(&box), make_not_null(&cache));
302  }
303  }
304 };

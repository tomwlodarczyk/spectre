18 /*!
19  * \ingroup TensorExpressionsGroup
20  * \brief Represents the indices in a TensorExpression
21  *
22  * \details
23  * Used to denote a tensor index in a tensor slot. This allows the following
24  * type of expressions to work:
25  * \code{.cpp}
26  * auto T = evaluate<ti_a_t, ti_b_t>(F(ti_a, ti_b) + S(ti_b, ti_a));
27  * \endcode
28  * where `using ti_a_t = TensorIndex<0>;` and `TensorIndex<0> ti_a;`, that is,
29  * `ti_a` and `ti_b` are place holders for objects of type `TensorIndex<0>` and
30  * `TensorIndex<1>` respectively.
31  */
32 template <std::size_t I>
33 struct TensorIndex {
34  using value_type = std::size_t;
35  using type = TensorIndex<I>;
36  static constexpr value_type value = I;
37 };

293 template <typename Derived, typename DataType, typename Symm,
294  typename IndexList, template <typename...> class ArgsList,
295  typename... Args>
296 struct TensorExpression<Derived, DataType, Symm, IndexList, ArgsList<Args...>> {
297  static_assert(sizeof...(Args) == 0 or
298  sizeof...(Args) == tmpl::size<IndexList>::value,
299  "the number of Tensor indices must match the number of "
300  "components specified in an expression.");
301  using type = DataType;
302  using symmetry = Symm;
303  using index_list = IndexList;
304  static constexpr auto num_tensor_indices =
305  tmpl::size<index_list>::value == 0 ? 1 : tmpl::size<index_list>::value;
306  /// Typelist of the tensor indices, e.g. `_a_t` and `_b_t` in `F(_a, _b)`
307  using args_list = ArgsList<Args...>;
308 
309  // @{
310  /// Cast down to the derived class. This is enabled by the
311  /// [CRTP](https://en.wikipedia.org/wiki/Curiously_recurring_template_pattern)
312  template <typename V = Derived,
313  Requires<not tt::is_a<Tensor, V>::value> = nullptr>
314  SPECTRE_ALWAYS_INLINE const Derived& operator~() const {
315  return static_cast<const Derived&>(*this);
316  }
317  // @}
318  // @{
319  /// If the Derived class is a Tensor return a const reference to a
320  /// TensorExpression.
321  ///
322  /// Since Tensor is not derived from TensorExpression (because of
323  /// complications arising from the indices being part of the expression,
324  /// specifically Tensor may need to derive off of hundreds or thousands of
325  /// base classes, which is not feasible), return a reference to a
326  /// TensorExpression, which has a sufficient interface to evaluate the
327  /// expression.
328  /// \returns const TensorExpression<Derived, DataType, Symm, IndexList,
329  /// ArgsList<Args...>>&
330  template <typename V = Derived,
331  Requires<tt::is_a<Tensor, V>::value> = nullptr>
332  SPECTRE_ALWAYS_INLINE auto operator~() const {
333  return static_cast<const TensorExpression<Derived, DataType, Symm,
334  IndexList, ArgsList<Args...>>&>(
335  *this);

341  /// \ingroup TensorExpressionsGroup
342  /// Helper struct to compute the correct tensor index array from a
343  /// typelist of std::integral_constant's indicating the ordering. This is
344  /// needed for dealing with expressions such as \f$T_{ab} = F_{ba}\f$ and gets

350  template <template <typename...> class RedArgsList, typename... RedArgs>
351  struct ComputeCorrectTensorIndex<RedArgsList<RedArgs...>> {
352  template <typename U, std::size_t Size>
353  SPECTRE_ALWAYS_INLINE static constexpr std::array<U, Size> apply(
354  const std::array<U, Size>& tensor_index) {
355  return std::array<U, Size>{{tensor_index[RedArgs::value]...}};
356  }

361  /// \brief return the value of type DataType with tensor index `tensor_index`
362  ///
363  /// \details
364  /// _Note:_ This version is selected if Derived is a Tensor
365  ///
366  /// One big challenge with a general Tensor Expression implementation is the
367  /// ordering of the Indices on the RHS and LHS of the expression. This
368  /// algorithm implemented in ::rhs_elements_in_lhs and
369  /// ::generate_transformation handles the index sorting.
370  ///
371  /// Here are some examples of what the algorithm does:
372  ///
373  /// LhsIndices is the desired ordering.
374  ///
375  /// LHS:
376  /// \code
377  /// <0, 1>
378  /// \endcode
379  /// RHS:
380  /// \code
381  /// <1, 2, 3, 0> -Transform> <3, 1, 2, 0>
382  /// \endcode
383  ///
384  /// LHS:
385  /// \code
386  /// <0, 1, 2> <a, b, c>
387  /// \endcode
388  /// RHS:
389  /// \code
390  /// <2, 0, 1> -Transform> <2 , 1, 0>
391  /// \endcode
392  ///
393  /// Below is pseudo-code of the algorithm written in a non-functional way
394  /// \verbatim
395  /// for Element in RHS:
396  /// if (Element in LHS):
397  /// index_in_LHS = index_of<LHS, Element>
398  /// tensor_index_to_find = at<RHS_with_only_LHS, index_in_LHS>
399  /// index_to_replace_with = index_of<RHS, tensor_index_to_find>
400  /// T_RHS = push_back<T_RHS, index_to_replace_with>
401  /// else:
402  /// T_RHS = push_back<T_RHS, iteration>
403  /// endif
404  /// end for
405  /// \endverbatim
406  ///
407  /// \tparam LhsIndices the tensor indices on the LHS on the expression
408  /// \tparam V used for SFINAE
409  /// \param tensor_index the tensor component to retrieve
410  /// \return the value of the DataType of component `tensor_index`
411  template <typename... LhsIndices, typename ArrayValueType,
412  typename V = Derived,
413  Requires<tt::is_a<Tensor, V>::value> = nullptr>
414  SPECTRE_ALWAYS_INLINE type
415  get(const std::array<ArrayValueType, num_tensor_indices>& tensor_index)
416  const {
417  ASSERT(t_ != nullptr,
418  "A TensorExpression that should be holding a pointer to a Tensor "
419  "is holding a nullptr.");
420  using rhs = args_list;
421  // To deal with Tensor products we need the ordering of only the subset of
422  // tensor indices present in this term
423  using lhs = rhs_elements_in_lhs<rhs, tmpl::list<LhsIndices...>>;
424  using rhs_only_with_lhs = rhs_elements_in_lhs<lhs, rhs>;
425  using transformation = generate_transformation<rhs, lhs, rhs_only_with_lhs>;
426  return t_->get(
427  ComputeCorrectTensorIndex<transformation>::apply(tensor_index));
428  }
429 
430  /// \brief return the value of type DataType with tensor index `tensor_index`
431  ///
432  /// \details
433  /// _Note:_ This version is selected if Derived is a TensorExpression
434  ///
435  /// Forward the tensor_index onwards to the next TensorExpression
436  ///
437  /// \tparam LhsIndices the tensor indices on the LHS on the expression
438  /// \tparam V used for SFINAE
439  /// \param tensor_index the tensor component to retrieve
440  /// \return the value of the DataType of component `tensor_index`
441  template <typename... LhsIndices, typename ArrayValueType,

488  /// v16.3 takes around 8 minutes. These tests were done on a Haswell Core i5.
489  const Derived* t_ = nullptr;

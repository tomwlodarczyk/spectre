27 /*!
28  * \brief Analytic initial data for the relativistic Orszag-Tang vortex.
29  *
30  * The relativistic version of the Orszag-Tang vortex is a
31  * 2-dimensional test case for relativistic MHD systems (see, e.g.,
32  * \cite Beckwith2011iy). It describes the flow of an ideal fluid with
33  * adiabatic index \f$5/3\f$. The initial conditions (and hence the
34  * states at later times) are periodic in both \f$x\f$ and \f$y\f$
35  * with period 1. The initial conditions are:
36  * \f{align*}
37  * \rho &= \frac{25}{36 \pi} \\
38  * p &= \frac{5}{12 \pi} \\
39  * v_x &= -\frac{1}{2} \sin(2 \pi y) \\
40  * v_y &= \frac{1}{2} \sin(2 \pi x) \\
41  * B_x &= -\frac{1}{\sqrt{4 \pi}} \sin(2 \pi y) \\
42  * B_y &= \frac{1}{\sqrt{4 \pi}} \sin(4 \pi x)
43  * \f}
44  * with \f$\rho\f$ the rest mass density, \f$p\f$ the pressure,
45  * \f$v_i\f$ the spatial velocity, and \f$B_i\f$ the magnetic field.
46  *
47  * \parblock
48  * \note We do not currently support 2-dimensional RMHD, so this class
49  * provides 3-dimensional data with no \f$z\f$-dependence.
50  * \endparblock
51  *
52  * \parblock
53  * \note There are multiple errors in the description of this test
54  * problem in the original SpECTRE paper \cite Kidder2016hev and there
55  * is a sign error in the velocity in \cite Beckwith2011iy. Despite these
56  * errors, the actual tests performed for those papers matched the standard
57  * problem as presented here.
58  * \endparblock
59  */

124  /// Retrieve a collection of hydro variables at `x`
125  template <typename DataType, typename... Tags>

127  const tnsr::I<DataType, 3, Frame::Inertial>& x,
128  tmpl::list<Tags...> /*meta*/) const noexcept {

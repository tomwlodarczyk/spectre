26 /// \ingroup ActionsGroup
27 /// \ingroup TimeGroup
28 /// \brief Adjust the step size for local time stepping
29 ///
30 /// Uses:
31 /// - ConstGlobalCache:
32 /// - Tags::StepChoosers<StepChooserRegistrars>
33 /// - Tags::StepController
34 /// - Tags::TimeStepperBase
35 /// - DataBox:
36 /// - Tags::HistoryEvolvedVariables
37 /// - Tags::TimeStepId
38 /// - Tags::TimeStep
39 ///
40 /// DataBox changes:
41 /// - Adds: nothing
42 /// - Removes: nothing
43 /// - Modifies: Tags::Next<Tags::TimeStepId>, Tags::TimeStep
44 template <typename StepChooserRegistrars>
45 struct ChangeStepSize {
46  using step_choosers_tag = Tags::StepChoosers<StepChooserRegistrars>;
47  using const_global_cache_tags =
48  tmpl::list<step_choosers_tag, Tags::StepController>;
49 
50  template <typename DbTags, typename... InboxTags, typename Metavariables,
51  typename ArrayIndex, typename ActionList,
52  typename ParallelComponent>
53  static std::tuple<db::DataBox<DbTags>&&> apply(
54  db::DataBox<DbTags>& box, tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
55  const Parallel::ConstGlobalCache<Metavariables>& cache,
56  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
57  const ParallelComponent* const /*meta*/) noexcept {
58  static_assert(Metavariables::local_time_stepping,
59  "ChangeStepSize can only be used with local time-stepping.");
60 
61  const LtsTimeStepper& time_stepper =
62  Parallel::get<Tags::TimeStepperBase>(cache);
63  const auto& step_choosers = Parallel::get<step_choosers_tag>(cache);
64  const auto& step_controller = Parallel::get<Tags::StepController>(cache);
65 
66  const auto& time_id = db::get<Tags::TimeStepId>(box);
67 
68  if (not time_stepper.can_change_step_size(
69  time_id, db::get<Tags::HistoryEvolvedVariables<>>(box))) {
70  return std::forward_as_tuple(std::move(box));
71  }
72 
73  const auto& current_step = db::get<Tags::TimeStep>(box);
74 
75  // The step choosers return the magnitude of the desired step, so
76  // we always want the minimum requirement, but we have to negate
77  // the final answer if time is running backwards.
78  double desired_step = std::numeric_limits<double>::infinity();
79  for (const auto& step_chooser : step_choosers) {
80  desired_step =
81  std::min(desired_step, step_chooser->desired_step(box, cache));
82  }
83  if (not current_step.is_positive()) {
84  desired_step = -desired_step;
85  }
86 
87  const auto new_step =
88  step_controller.choose_step(time_id.step_time(), desired_step);
89  if (new_step != current_step) {
90  const auto new_next_time_id =
91  time_stepper.next_time_id(time_id, new_step);
92 
93  db::mutate<Tags::Next<Tags::TimeStepId>, Tags::TimeStep>(
94  make_not_null(&box),
95  [&new_next_time_id, &new_step](
96  const gsl::not_null<TimeStepId*> next_time_id,
97  const gsl::not_null<TimeDelta*> time_step) noexcept {
98  *next_time_id = new_next_time_id;
99  *time_step = new_step;
100  });
101  }
102 
103  return std::forward_as_tuple(std::move(box));
104  }
105 };
106 } // namespace Actions

77  template <class T>
78  static constexpr bool can_bind_reference() noexcept {
79  using rem_ref_value_type = typename std::remove_reference<value_type>::type;
80  using rem_ref_T = typename std::remove_reference<T>::type;
81  using is_lvalue_type = std::integral_constant<
82  bool, cpp17::is_lvalue_reference_v<T> or
83  cpp17::is_same_v<std::reference_wrapper<rem_ref_value_type>,
84  rem_ref_T> or
85  cpp17::is_same_v<std::reference_wrapper<
86  std::remove_const_t<rem_ref_value_type>>,
87  rem_ref_T>>;
88  return not cpp17::is_reference_v<value_type> or
89  (cpp17::is_lvalue_reference_v<value_type> and
90  is_lvalue_type::value) or
91  (cpp17::is_rvalue_reference_v<value_type> and
92  not cpp17::is_lvalue_reference_v<T>);
93  }

154 constexpr int select_expand_subitems_impl(const size_t pack_size) noexcept {
155  // selects the appropriate fast track based on the pack size. Fast tracks for
156  // 2 and 4 limit the DataBox to about 800 items. This could be increased by
157  // adding fast tracks for say 8 and 64.
158  return pack_size >= 4 ? 3 : pack_size >= 2 ? 2 : static_cast<int>(pack_size);

218  * Expands the `ComputeTagsList` into a parameter pack, and also makes the
219  * decision about expanding the simple tags. The `ComputeTagsList` is expanded
220  * into a parameter pack so that we can use type aliases to do the computation,

1069 /*!
1070  * \ingroup DataBoxGroup
1071  * \brief Allows changing the state of one or more non-computed elements in
1072  * the DataBox
1073  *
1074  * `mutate()`'s first argument is the DataBox from which to retrieve the tags
1075  * `MutateTags`. The objects corresponding to the `MutateTags` are then passed
1076  * to `invokable`, which is a lambda or a function object taking as many
1077  * arguments as there are `MutateTags` and with the arguments being of types
1078  * `gsl::not_null<db::item_type<MutateTags>*>...`. Inside the `invokable` no
1079  * items can be retrieved from the DataBox `box`. This is to avoid confusing
1080  * subtleties with order of evaluation of compute items, as well as dangling
1081  * references. If an `invokable` needs read access to items in `box` they should
1082  * be passed as additional arguments to `mutate`. Capturing them by reference in
1083  * a lambda does not work because of a bug in GCC 6.3 and earlier. For a
1084  * function object the read-only items can also be stored as const references
1085  * inside the object by passing `db::get<TAG>(t)` to the constructor.
1086  *
1087  * \example
1088  * \snippet Test_DataBox.cpp databox_mutate_example
1089  */
1090 template <typename... MutateTags, typename TagList, typename Invokable,
1091  typename... Args>
1092 void mutate(const gsl::not_null<DataBox<TagList>*> box, Invokable&& invokable,
1093  Args&&... args) noexcept {
1094  static_assert(
1095  tmpl2::flat_all_v<
1096  DataBox_detail::has_unique_matching_tag_v<TagList, MutateTags>...>,
1097  "One of the tags being mutated could not be found in the DataBox or "

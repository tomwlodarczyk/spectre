116 /// \ingroup ActionsGroup
117 /// \brief Sends points on a Kerr horizon to an `Interpolator`.
118 ///
119 /// The points are such that they conform to the horizon of a Kerr
120 /// black hole (in Kerr-Schild coordinates) with given center, mass,
121 /// and dimensionless spin, as specified in the options.
122 ///
123 /// \note The returned points are not actually on the horizon;
124 /// instead, they are collocation points of a Strahlkorper whose
125 /// spectral representation matches the horizon shape up to order
126 /// Lmax, where Lmax is the spherical-harmonic order specified in the
127 /// options. As Lmax increases, the returned points converge to the
128 /// horizon.
129 ///
130 /// Uses:
131 /// - DataBox:
132 /// - `::Tags::Domain<3, Frame>`
133 /// - `::Tags::Variables<typename
134 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
135 ///
136 /// DataBox changes:
137 /// - Adds: nothing
138 /// - Removes: nothing
139 /// - Modifies:
140 /// - `Tags::IndicesOfFilledInterpPoints`
141 /// - `::Tags::Variables<typename
142 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
143 ///
144 /// For requirements on InterpolationTargetTag, see InterpolationTarget
145 template <typename InterpolationTargetTag, typename Frame>
146 struct KerrHorizon {
147  using const_global_cache_tags =
148  tmpl::list<Tags::KerrHorizon<InterpolationTargetTag>>;
149  using initialization_tags =
150  tmpl::append<StrahlkorperTags::items_tags<Frame>,
151  StrahlkorperTags::compute_items_tags<Frame>>;
152  template <typename DbTags, typename Metavariables>
153  static auto initialize(
154  db::DataBox<DbTags>&& box,
155  const Parallel::ConstGlobalCache<Metavariables>& cache) noexcept {
156  const auto& options =
157  Parallel::get<Tags::KerrHorizon<InterpolationTargetTag>>(cache);
158 
159  // Make a Strahlkorper with the correct shape.
160  ::Strahlkorper<Frame> strahlkorper(
161  options.l_max, options.l_max,
162  get(gr::Solutions::kerr_horizon_radius(
163  ::YlmSpherepack(options.l_max, options.l_max).theta_phi_points(),
164  options.mass, options.dimensionless_spin)),
165  options.center);
166 
167  // Put Strahlkorper and its ComputeItems into a new DataBox.
168  return db::create_from<
169  db::RemoveTags<>,
170  db::AddSimpleTags<StrahlkorperTags::items_tags<Frame>>,
171  db::AddComputeTags<StrahlkorperTags::compute_items_tags<Frame>>>(
172  std::move(box), std::move(strahlkorper));
173  }
174  template <typename ParallelComponent, typename DbTags, typename Metavariables,
175  typename ArrayIndex,
176  Requires<tmpl::list_contains_v<
177  DbTags, Tags::TemporalIds<Metavariables>>> = nullptr>
178  static void apply(
179  db::DataBox<DbTags>& box,
180  Parallel::ConstGlobalCache<Metavariables>& cache,
181  const ArrayIndex& /*array_index*/,
182  const typename Metavariables::temporal_id::type& temporal_id) noexcept {
183  // In the future, when we add support for multiple Frames,
184  // the code that transforms coordinates from the Strahlkorper Frame
185  // to Frame::Inertial will go here. That transformation
186  // may depend on `temporal_id`.
187  send_points_to_interpolator<InterpolationTargetTag>(
188  box, cache,
189  db::get<StrahlkorperTags::CartesianCoords<::Frame::Inertial>>(box),
190  temporal_id);
191  }
192 };

192  static Phase determine_next_phase(
193  const Phase& current_phase,
194  const Parallel::CProxy_ConstGlobalCache<
195  EvolutionMetavars>& /*cache_proxy*/) noexcept {
196  switch (current_phase) {
197  case Phase::Initialization:
198  return Phase::InitializeTimeStepperHistory;
199  case Phase::InitializeTimeStepperHistory:
200  return Phase::RegisterWithObserver;
201  case Phase::RegisterWithObserver:
202  return Phase::Evolve;
203  case Phase::Evolve:
204  return Phase::Exit;
205  case Phase::Exit:
206  ERROR(
207  "Should never call determine_next_phase with the current phase "
208  "being 'Exit'");
209  default:
210  ERROR(
211  "Unknown type of phase. Did you static_cast<Phase> an integral "
212  "value?");
213  }
214  }

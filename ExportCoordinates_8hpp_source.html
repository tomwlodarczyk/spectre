37 template <size_t Dim>
38 struct ExportCoordinates {
39  // Compile-time interface for observers
40  struct ObservationType {};
41  template <typename ParallelComponent, typename DbTagsList,
42  typename ArrayIndex>
43  static std::pair<observers::TypeOfObservation, observers::ObservationId>
44  register_info(const db::DataBox<DbTagsList>& /*box*/,
45  const ArrayIndex& /*array_index*/) noexcept {
46  return {observers::TypeOfObservation::ReductionAndVolume,
47  observers::ObservationId(0., ObservationType{})};
48  }
49 
50  template <
51  typename DbTagsList, typename... InboxTags, typename Metavariables,
52  typename ArrayIndex, typename ActionList, typename ParallelComponent,
53  Requires<tmpl::list_contains_v<DbTagsList, Tags::Mesh<Dim>>> = nullptr>
54  static std::tuple<db::DataBox<DbTagsList>&&, bool> apply(
55  db::DataBox<DbTagsList>& box,
56  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
57  const Parallel::ConstGlobalCache<Metavariables>& cache,
58  const ArrayIndex& array_index, const ActionList /*meta*/,
59  const ParallelComponent* const /*meta*/) {
60  const auto& mesh = get<Tags::Mesh<Dim>>(box);
61  const auto& inertial_coordinates =
62  db::get<::Tags::Coordinates<Dim, Frame::Inertial>>(box);
63  const std::string element_name = MakeString{} << ElementId<Dim>(array_index)
64  << '/';
65  // Collect volume data
66  // Remove tensor types, only storing individual components
67  std::vector<TensorComponent> components;
68  components.reserve(Dim);
69  for (size_t d = 0; d < Dim; d++) {
70  components.emplace_back(element_name + "InertialCoordinates_" +
71  inertial_coordinates.component_name(
72  inertial_coordinates.get_tensor_index(d)),
73  inertial_coordinates.get(d));
74  }
75  // Send data to volume observer
76  auto& local_observer =
77  *Parallel::get_parallel_component<observers::Observer<Metavariables>>(
78  cache)
79  .ckLocalBranch();
80  Parallel::simple_action<observers::Actions::ContributeVolumeData>(
81  local_observer, observers::ObservationId(0., ObservationType{}),
82  std::string{"/element_data"},
83  observers::ArrayComponentId(
84  std::add_pointer_t<ParallelComponent>{nullptr},
85  Parallel::ArrayIndex<ElementIndex<Dim>>(array_index)),
86  std::move(components), mesh.extents());
87  return {std::move(box), true};
88  }
89 };
90 } // namespace Actions

129  const Parallel::CProxy_ConstGlobalCache<

137  return Phase::Exit;

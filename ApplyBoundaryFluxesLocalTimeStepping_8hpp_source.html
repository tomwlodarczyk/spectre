35 /// \ingroup ActionsGroup
36 /// \ingroup DiscontinuousGalerkinGroup
37 /// \brief Perform the boundary part of the update of the variables
38 /// for local time stepping.
39 ///
40 /// Uses:
41 /// - ConstGlobalCache:
42 /// - Tags::TimeStepperBase
43 /// - Metavariables::normal_dot_numerical_flux
44 /// - DataBox:
45 /// - Tags::Mesh<volume_dim>
46 /// - Tags::Mortars<Tags::Mesh<volume_dim - 1>, volume_dim>
47 /// - Tags::Mortars<Tags::MortarSize<volume_dim - 1>, volume_dim>
48 /// - Tags::TimeStep
49 ///
50 /// DataBox changes:
51 /// - Adds: nothing
52 /// - Removes: nothing
53 /// - Modifies:
54 /// - variables_tag
55 /// - FluxCommunicationTypes<Metavariables>::mortar_data_tag
56 struct ApplyBoundaryFluxesLocalTimeStepping {
57  template <typename DbTags, typename... InboxTags, typename Metavariables,
58  typename ArrayIndex, typename ActionList,
59  typename ParallelComponent>
60  static std::tuple<db::DataBox<DbTags>&&> apply(
61  db::DataBox<DbTags>& box,
62  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
63  const Parallel::ConstGlobalCache<Metavariables>& cache,
64  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
65  const ParallelComponent* const /*meta*/) noexcept { // NOLINT const
66  static_assert(Metavariables::local_time_stepping,
67  "ApplyBoundaryFluxesLocalTimeStepping can only be used with "
68  "local time-stepping.");
69 
70  using system = typename Metavariables::system;
71  constexpr size_t volume_dim = system::volume_dim;
72  using variables_tag = typename system::variables_tag;
73 
74  using flux_comm_types = FluxCommunicationTypes<Metavariables>;
75 
76  using mortar_data_tag =
77  typename flux_comm_types::local_time_stepping_mortar_data_tag;
78  db::mutate<variables_tag, mortar_data_tag>(
79  make_not_null(&box),
80  [&cache](
81  const gsl::not_null<db::item_type<variables_tag>*> vars,
82  const gsl::not_null<db::item_type<mortar_data_tag>*> mortar_data,
83  const db::const_item_type<Tags::Mesh<volume_dim>>& mesh,
84  const db::const_item_type<Tags::Mortars<Tags::Mesh<volume_dim - 1>,
85  volume_dim>>& mortar_meshes,
86  const db::const_item_type<Tags::Mortars<
87  Tags::MortarSize<volume_dim - 1>, volume_dim>>& mortar_sizes,
88  const db::const_item_type<Tags::TimeStep>& time_step) noexcept {
89  // Having the lambda just wrap another lambda works around a
90  // gcc 6.4.0 segfault.
91  [
92  &cache, &vars, &mortar_data, &mesh, &mortar_meshes, &mortar_sizes,
93  &time_step
94  ]() noexcept {
95  const auto& normal_dot_numerical_flux_computer =
96  get<typename Metavariables::normal_dot_numerical_flux>(cache);
97  const LtsTimeStepper& time_stepper =
98  get<Tags::TimeStepperBase>(cache);
99 
100  for (auto& mortar_id_and_data : *mortar_data) {
101  const auto& mortar_id = mortar_id_and_data.first;
102  auto& data = mortar_id_and_data.second;
103  const auto& direction = mortar_id.first;
104  const size_t dimension = direction.dimension();
105 
106  const auto face_mesh = mesh.slice_away(dimension);
107  const auto perpendicular_extent = mesh.extents(dimension);
108  const auto& mortar_mesh = mortar_meshes.at(mortar_id);
109  const auto& mortar_size = mortar_sizes.at(mortar_id);
110 
111  // This lambda must only capture quantities that are
112  // independent of the simulation state.
113  const auto coupling =
114  [
115  &face_mesh, &mortar_mesh, &mortar_size,
116  &normal_dot_numerical_flux_computer, &perpendicular_extent
117  ](const typename flux_comm_types::LocalData& local_data,
118  const typename flux_comm_types::PackagedData&
119  remote_data) noexcept {
120  return compute_boundary_flux_contribution<flux_comm_types>(
121  normal_dot_numerical_flux_computer, local_data, remote_data,
122  face_mesh, mortar_mesh, perpendicular_extent, mortar_size);
123  };
124 
125  const auto lifted_data = time_stepper.compute_boundary_delta(
126  coupling, make_not_null(&data), time_step);
127 
128  add_slice_to_data(vars, lifted_data, mesh.extents(), dimension,
129  index_to_slice_at(mesh.extents(), direction));
130  }
131  }();
132  },
133  db::get<Tags::Mesh<volume_dim>>(box),
134  db::get<Tags::Mortars<Tags::Mesh<volume_dim - 1>, volume_dim>>(box),
135  db::get<Tags::Mortars<Tags::MortarSize<volume_dim - 1>, volume_dim>>(
136  box),
137  db::get<Tags::TimeStep>(box));
138 
139  return std::forward_as_tuple(std::move(box));
140  }
141 };
142 } // namespace Actions
143 } // namespace dg

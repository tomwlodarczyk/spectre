35 /*!
36  * \brief A static spherically symmetric star
37  *
38  * An analytic solution for a static, spherically-symmetric star found by
39  * solving the Tolman-Oppenheimer-Volkoff (TOV) equations. The equation of
40  * state is assumed to be that of a polytropic fluid.
41  *
42  * \tparam RadialSolution selects not only how the TOV equations are solved, but
43  * also the radial coordinate that the solution is given in (e.g. areal or
44  * isotropic). See the documentation of the specific `RadialSolution`s for
45  * more details.
46  */
47 template <typename RadialSolution>
48 class TovStar : public MarkAsAnalyticSolution {
49  public:
50  /*!
51  * \brief The radial variables needed to compute the full `TOVStar` solution
52  *
53  * RadialSolution must provide a method that fills the radial variables
54  * given the equation of state and the Cartesian coordinates \f$x^i\f$
55  *
56  * If the spherically symmetric metric is written as
57  *
58  * \f[
59  * ds^2 = - e^{2 \Phi_t} dt^2 + e^{2 \Phi_r} dr^2 + e^{2 \Phi_\Omega} r^2
60  * d\Omega^2
61  * \f]
62  *
63  * where \f$r = \delta_{mn} x^m x^n\f$ is the `radial_coordinate` and
64  * \f$\Phi_t\f$, \f$\Phi_r\f$, and \f$\Phi_\Omega\f$ are respectvely the
65  * `metric_time_potential`, `metric_radial_potential`, and
66  * `metric_angular_potential`, then the lapse, shift, and spatial metric in
67  * the Cartesian coordinates are
68  *
69  * \f{align*}
70  * \alpha &= e^{\Phi_t} \\
71  * \beta^i &= 0 \\
72  * \gamma_{ij} &= \delta_{ij} e^{2 \Phi_r} + \delta_{im} \delta_{jn}
73  * \frac{x^m x^n}{r^2} \left( e^{2 \Phi_r} - e^{2 \Phi_\Omega} \right)
74  * \f}
75  *
76  */
77  template <typename DataType>
78  struct RadialVariables {
79  explicit RadialVariables(DataType radial_coordinate_in)
80  : radial_coordinate(std::move(radial_coordinate_in)),
81  rest_mass_density(
82  make_with_value<Scalar<DataType>>(radial_coordinate, 0.0)),
83  pressure(make_with_value<Scalar<DataType>>(radial_coordinate, 0.0)),
84  specific_internal_energy(
85  make_with_value<Scalar<DataType>>(radial_coordinate, 0.0)),
86  specific_enthalpy(
87  make_with_value<Scalar<DataType>>(radial_coordinate, 0.0)),
88  metric_time_potential(
89  make_with_value<DataType>(radial_coordinate, 0.0)),
90  dr_metric_time_potential(
91  make_with_value<DataType>(radial_coordinate, 0.0)),
92  metric_radial_potential(
93  make_with_value<DataType>(radial_coordinate, 0.0)),
94  dr_metric_radial_potential(
95  make_with_value<DataType>(radial_coordinate, 0.0)),
96  metric_angular_potential(
97  make_with_value<DataType>(radial_coordinate, 0.0)),
98  dr_metric_angular_potential(
99  make_with_value<DataType>(radial_coordinate, 0.0)) {}
100  DataType radial_coordinate{};
101  Scalar<DataType> rest_mass_density{};
102  Scalar<DataType> pressure{};
103  Scalar<DataType> specific_internal_energy{};
104  Scalar<DataType> specific_enthalpy{};
105  DataType metric_time_potential{};
106  DataType dr_metric_time_potential{};
107  DataType metric_radial_potential{};
108  DataType dr_metric_radial_potential{};
109  DataType metric_angular_potential{};
110  DataType dr_metric_angular_potential{};
111  };

76 /*!
77  * \ingroup DiscontinuousGalerkinGroup
78  * \brief %Observe volume tensor fields.
79  *
80  * Writes volume quantities:
81  * - `InertialCoordinates`
82  * - Tensors listed in `Tensors` template parameter
83  * - `Error(*)` = errors in `AnalyticSolutionTensors` =
84  * \f$\text{value} - \text{analytic solution}\f$
85  *
86  * \warning Currently, only one volume observation event can be
87  * triggered at a given time. Causing multiple events to run at once
88  * will produce unpredictable results.
89  */
90 template <size_t VolumeDim, typename ObservationValueTag, typename... Tensors,
91  typename... AnalyticSolutionTensors, typename EventRegistrars,
92  typename... NonSolutionTensors>
93 class ObserveFields<VolumeDim, ObservationValueTag, tmpl::list<Tensors...>,
94  tmpl::list<AnalyticSolutionTensors...>, EventRegistrars,
95  tmpl::list<NonSolutionTensors...>>
96  : public Event<EventRegistrars> {
97  private:
98  static_assert(
99  cpp17::is_same_v<
100  tmpl::list_difference<tmpl::list<AnalyticSolutionTensors...>,
101  tmpl::list<Tensors...>>,
102  tmpl::list<>>,
103  "All AnalyticSolutionTensors must be listed in Tensors.");
104  using coordinates_tag = ::Tags::Coordinates<VolumeDim, Frame::Inertial>;
105 
106  template <typename T>
107  static std::string component_suffix(const T& tensor,
108  size_t component_index) noexcept {
109  return tensor.rank() == 0
110  ? ""
111  : "_" + tensor.component_name(
112  tensor.get_tensor_index(component_index));
113  }
114 
115  public:
116  /// \cond
117  explicit ObserveFields(CkMigrateMessage* /*unused*/) noexcept {}
118  using PUP::able::register_constructor;
119  WRAPPED_PUPable_decl_template(ObserveFields); // NOLINT
120  /// \endcond

122  struct VariablesToObserve {
123  static constexpr OptionString help = "Subset of variables to observe";
124  using type = std::vector<std::string>;
125  static type default_value() noexcept { return {Tensors::name()...}; }
126  static size_t lower_bound_on_size() noexcept { return 1; }
127  };

129  using options = tmpl::list<VariablesToObserve>;
130  static constexpr OptionString help =
131  "Observe volume tensor fields.\n"
132  "\n"
133  "Writes volume quantities:\n"
134  " * InertialCoordinates\n"
135  " * Tensors listed in Tensors template parameter\n"
136  " * Error(*) = errors in AnalyticSolutionTensors\n"
137  " = value - analytic solution\n"
138  "\n"
139  "Warning: Currently, only one volume observation event can be\n"
140  "triggered at a given time. Causing multiple events to run at once\n"
141  "will produce unpredictable results.";

143  explicit ObserveFields(const std::vector<std::string>& variables_to_observe =
144  VariablesToObserve::default_value(),
145  const OptionContext& context = {})
146  : variables_to_observe_(variables_to_observe.begin(),
147  variables_to_observe.end()) {
148  const std::unordered_set<std::string> valid_tensors{Tensors::name()...};
149  for (const auto& name : variables_to_observe_) {
150  if (valid_tensors.count(name) != 1) {
151  PARSE_ERROR(
152  context,
153  name << " is not an available variable. Available variables:\n"
154  << (std::vector<std::string>{Tensors::name()...}));
155  }
156  if (alg::count(variables_to_observe, name) != 1) {
157  PARSE_ERROR(context, name << " specified multiple times");
158  }
159  }
160  variables_to_observe_.insert(coordinates_tag::name());
161  }

163  using argument_tags = tmpl::list<ObservationValueTag, ::Tags::Mesh<VolumeDim>,
164  coordinates_tag, AnalyticSolutionTensors...,
165  ::Tags::Analytic<AnalyticSolutionTensors>...,
166  NonSolutionTensors...>;

168  template <typename Metavariables, typename ParallelComponent>
169  void operator()(
170  const db::const_item_type<ObservationValueTag>& observation_value,
171  const Mesh<VolumeDim>& mesh,
172  const tnsr::I<DataVector, VolumeDim, Frame::Inertial>&
173  inertial_coordinates,
174  const db::const_item_type<
175  AnalyticSolutionTensors>&... analytic_solution_tensors,
176  const db::const_item_type<
177  ::Tags::Analytic<AnalyticSolutionTensors>>&... analytic_solutions,
178  const db::const_item_type<NonSolutionTensors>&... non_solution_tensors,
179  Parallel::ConstGlobalCache<Metavariables>& cache,
180  const ElementIndex<VolumeDim>& array_index,
181  const ParallelComponent* const /*meta*/) const noexcept {
182  const std::string element_name =
183  MakeString{} << ElementId<VolumeDim>(array_index) << '/';
184 
185  // Remove tensor types, only storing individual components.
186  std::vector<TensorComponent> components;
187  // This is larger than we need if we are only observing some
188  // tensors, but that's not a big deal and calculating the correct
189  // size is nontrivial.
190  components.reserve(alg::accumulate(
191  std::initializer_list<size_t>{
192  inertial_coordinates.size(),
193  2 * db::const_item_type<AnalyticSolutionTensors>::size()...,
194  db::const_item_type<NonSolutionTensors>::size()...},
195  0_st));
196 
197  const auto record_tensor_components = [ this, &components, &element_name ](
198  const auto tensor_tag_v, const auto& tensor) noexcept {
199  using tensor_tag = tmpl::type_from<decltype(tensor_tag_v)>;
200  if (variables_to_observe_.count(tensor_tag::name()) == 1) {
201  for (size_t i = 0; i < tensor.size(); ++i) {
202  components.emplace_back(
203  element_name + tensor_tag::name() + component_suffix(tensor, i),
204  tensor[i]);
205  }
206  }
207  };
208  record_tensor_components(tmpl::type_<coordinates_tag>{},
209  inertial_coordinates);
210  EXPAND_PACK_LEFT_TO_RIGHT(record_tensor_components(
211  tmpl::type_<AnalyticSolutionTensors>{}, analytic_solution_tensors));
212  EXPAND_PACK_LEFT_TO_RIGHT(record_tensor_components(
213  tmpl::type_<NonSolutionTensors>{}, non_solution_tensors));
214 
215  const auto record_errors = [ this, &components, &element_name ](
216  const auto tensor_tag_v, const auto& tensor,
217  const auto& analytic_tensor) noexcept {
218  using tensor_tag = tmpl::type_from<decltype(tensor_tag_v)>;
219  if (variables_to_observe_.count(tensor_tag::name()) == 1) {
220  for (size_t i = 0; i < tensor.size(); ++i) {
221  DataVector error = tensor[i] - analytic_tensor[i];
222  components.emplace_back(element_name + "Error(" + tensor_tag::name() +
223  ")" + component_suffix(tensor, i),
224  std::move(error));
225  }
226  }
227  };
228  EXPAND_PACK_LEFT_TO_RIGHT(
229  record_errors(tmpl::type_<AnalyticSolutionTensors>{},
230  analytic_solution_tensors, analytic_solutions));
231 
232  (void)(record_errors); // Silence GCC warning about unused variable
233 
234  // Send data to volume observer
235  auto& local_observer =
236  *Parallel::get_parallel_component<observers::Observer<Metavariables>>(
237  cache)
238  .ckLocalBranch();
239  Parallel::simple_action<observers::Actions::ContributeVolumeData>(
240  local_observer,
241  observers::ObservationId(
242  observation_value,
243  typename Metavariables::element_observation_type{}),
244  std::string{"/element_data"},
245  observers::ArrayComponentId(
246  std::add_pointer_t<ParallelComponent>{nullptr},
247  Parallel::ArrayIndex<ElementIndex<VolumeDim>>(array_index)),
248  std::move(components), mesh.extents());
249  }

251  // NOLINTNEXTLINE(google-runtime-references)
252  void pup(PUP::er& p) noexcept override {
253  Event<EventRegistrars>::pup(p);
254  p | variables_to_observe_;
255  }
256 
257  private:
258  std::unordered_set<std::string> variables_to_observe_{};
259 };

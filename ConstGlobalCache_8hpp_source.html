87 /// In order to be able to use a mock action testing framework we need to be
88 /// able to get the correct parallel component from the global cache even when
89 /// the correct component is a mock. We do this by having the mocked components
90 /// have a member type alias `component_being_mocked`, and having
91 /// `Parallel::get_component` check if the component to be retrieved is in the
92 /// `metavariables::component_list`. If it is not in the `component_list` then
93 /// we search for a mock component that is mocking the component we are trying
94 /// to retrieve.
95 template <typename ComponentList, typename ParallelComponent>
96 using get_component_if_mocked = tmpl::front<tmpl::type_from<tmpl::conditional_t<
97  tmpl::list_contains_v<ComponentList, ParallelComponent>,
98  tmpl::type_<tmpl::list<ParallelComponent>>,
99  tmpl::lazy::find<ComponentList,
100  tmpl::type_<get_component_if_mocked_helper<
101  tmpl::pin<ParallelComponent>, tmpl::_1>>>>>>;
102 } // namespace ConstGlobalCache_detail

104 /// \ingroup ParallelGroup
105 /// A Charm++ chare that caches constant data once per Charm++ node.
106 ///
107 /// `Metavariables` must define the following metavariables:
108 /// - `component_list` typelist of ParallelComponents
109 /// - `const_global_cache_tags` (possibly empty) typelist of tags of
110 /// constant data
111 ///
112 /// The tag list for the items added to the ConstGlobalCache is created by
113 /// combining the following tag lists:
114 /// - `Metavariables::const_global_cache_tags` which should contain only those
115 /// tags that cannot be added from the other tag lists below.
116 /// - `Component::const_global_cache_tags` for each `Component` in
117 /// `Metavariables::component_list` which should contain the tags needed by
118 /// any simple actions called on the Component, as well as tags need by the
119 /// `allocate_array` function of an array component. The type alias may be
120 /// omitted for an empty list.
121 /// - `Action::const_global_cache_tags` for each `Action` in the
122 /// `phase_dependent_action_list` of each `Component` of
123 /// `Metavariables::component_list` which should contain the tags needed by
124 /// that Action. The type alias may be omitted for an empty list.
125 ///
126 /// The tags in the `const_global_cache_tags` type lists are db::SimpleTag%s
127 /// that have a `using option_tags` type alias and a static function
128 /// `create_from_options` that are used to create the constant data from input
129 /// file options.
130 ///
131 /// References to items in the ConstGlobalCache are also added to the
132 /// db::DataBox of each `Component` in the `Metavariables::component_list` with
133 /// the same tag with which they were inserted into the ConstGlobalCache.
134 template <typename Metavariables>
135 class ConstGlobalCache : public CBase_ConstGlobalCache<Metavariables> {

140  tmpl::bind<Parallel::proxy_from_parallel_component, tmpl::_1>>>;
141 
142  public:
143  /// Access to the Metavariables template parameter
144  using metavariables = Metavariables;
145  /// Typelist of the ParallelComponents stored in the ConstGlobalCache
146  using component_list = typename Metavariables::component_list;

246 /// \ingroup ParallelGroup
247 /// \brief Access data in the cache
248 ///
249 /// \requires ConstGlobalCacheTag is a tag in tag_list
250 ///
251 /// \returns a constant reference to an object in the cache
252 template <typename ConstGlobalCacheTag, typename Metavariables>
253 auto get(const ConstGlobalCache<Metavariables>& cache) noexcept -> const
254  ConstGlobalCache_detail::type_for_get<ConstGlobalCacheTag, Metavariables>& {
255  // We check if the tag is to be retrieved directly or via a base class
256  using tag = tmpl::front<ConstGlobalCache_detail::get_list_of_matching_tags<
257  ConstGlobalCacheTag, Metavariables>>;
258  static_assert(tmpl::size<ConstGlobalCache_detail::get_list_of_matching_tags<
259  ConstGlobalCacheTag, Metavariables>>::value == 1,
260  "Found more than one tag matching the ConstGlobalCacheTag "
261  "requesting to be retrieved.");
262  return make_overloader(
263  [](std::true_type /*is_unique_ptr*/, auto&& local_cache)
264  -> decltype(
265  *(tuples::get<tag>(local_cache.const_global_cache_).get())) {
266  return *(
267  tuples::get<tag>(local_cache.const_global_cache_)
268  .get());
269  },
270  [](std::false_type /*is_unique_ptr*/, auto&& local_cache)
271  -> decltype(tuples::get<tag>(local_cache.const_global_cache_)) {
272  return tuples::get<tag>(
273  local_cache.const_global_cache_);
274  })(typename tt::is_a<std::unique_ptr, typename tag::type>::type{}, cache);
275 }

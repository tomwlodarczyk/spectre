37 struct InitializeGrTags {
38  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
39  typename ArrayIndex, typename ActionList,
40  typename ParallelComponent>
41  static auto apply(db::DataBox<DbTagsList>& box,
42  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
43  const Parallel::ConstGlobalCache<Metavariables>& cache,
44  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
45  const ParallelComponent* const /*meta*/) noexcept {
46  using system = typename Metavariables::system;
47  static constexpr size_t dim = system::volume_dim;
48  using gr_tag = typename system::spacetime_variables_tag;
49  using simple_tags = db::AddSimpleTags<gr_tag>;
50  using compute_tags = db::AddComputeTags<>;
51 
52  const double initial_time = db::get<Initialization::Tags::InitialTime>(box);
53  using GrVars = typename gr_tag::type;
54 
55  const size_t num_grid_points =
56  db::get<::Tags::Mesh<dim>>(box).number_of_grid_points();
57  const auto& inertial_coords =
58  db::get<::Tags::Coordinates<dim, Frame::Inertial>>(box);
59 
60  // Set initial data from analytic solution
61  GrVars gr_vars{num_grid_points};
62  make_overloader(
63  [ initial_time, &inertial_coords ](
64  std::true_type /*is_analytic_solution*/,
65  const gsl::not_null<GrVars*> local_gr_vars,
66  const auto& local_cache) noexcept {
67  using solution_tag = ::Tags::AnalyticSolutionBase;
68  local_gr_vars->assign_subset(
69  Parallel::get<solution_tag>(local_cache)
70  .variables(inertial_coords, initial_time,
71  typename GrVars::tags_list{}));
72  },
73  [&inertial_coords](std::false_type /*is_analytic_solution*/,
74  const gsl::not_null<GrVars*> local_gr_vars,
75  const auto& local_cache) noexcept {
76  using analytic_data_tag = ::Tags::AnalyticDataBase;
77  local_gr_vars->assign_subset(
78  Parallel::get<analytic_data_tag>(local_cache)
79  .variables(inertial_coords, typename GrVars::tags_list{}));
80  })(
81  evolution::is_analytic_solution<typename Metavariables::initial_data>{},
82  make_not_null(&gr_vars), cache);
83 
84  return std::make_tuple(
85  Initialization::merge_into_databox<InitializeGrTags, simple_tags,
86  compute_tags>(std::move(box),
87  std::move(gr_vars)));
88  }
89 };

91 struct InitializeM1Tags {
92  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
93  typename ArrayIndex, typename ActionList,
94  typename ParallelComponent>
95  static auto apply(db::DataBox<DbTagsList>& box,
96  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
97  const Parallel::ConstGlobalCache<Metavariables>& cache,
98  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
99  const ParallelComponent* const /*meta*/) noexcept {
100  using system = typename Metavariables::system;
101  using evolved_variables_tag = typename system::variables_tag;
102  using hydro_variables_tag = typename system::hydro_variables_tag;
103  using m1_variables_tag = typename system::primitive_variables_tag;
104  // List of variables to be created... does NOT include
105  // evolved_variables_tag because the evolved variables
106  // are created by the ConservativeSystem initialization.
107  using simple_tags =
108  db::AddSimpleTags<hydro_variables_tag, m1_variables_tag>;
109  using compute_tags = db::AddComputeTags<>;
110 
111  using EvolvedVars = typename evolved_variables_tag::type;
112  using HydroVars = typename hydro_variables_tag::type;
113  using M1Vars = typename m1_variables_tag::type;
114 
115  static constexpr size_t dim = system::volume_dim;
116  const double initial_time = db::get<Initialization::Tags::InitialTime>(box);
117  const size_t num_grid_points =
118  db::get<::Tags::Mesh<dim>>(box).number_of_grid_points();
119  const auto& inertial_coords =
120  db::get<::Tags::Coordinates<dim, Frame::Inertial>>(box);
121 
122  db::mutate<evolved_variables_tag>(
123  make_not_null(&box),
124  make_overloader(
125  [ initial_time, &inertial_coords ](
126  const gsl::not_null<EvolvedVars*> evolved_vars,
127  std::true_type /*is_analytic_solution*/,
128  const auto& local_cache) noexcept {
129  using solution_tag = ::Tags::AnalyticSolutionBase;
130  evolved_vars->assign_subset(
131  Parallel::get<solution_tag>(local_cache)
132  .variables(inertial_coords, initial_time,
133  typename evolved_variables_tag::tags_list{}));
134  },
135  [&inertial_coords](const gsl::not_null<EvolvedVars*> evolved_vars,
136  std::false_type /*is_analytic_solution*/,
137  const auto& local_cache) noexcept {
138  using analytic_data_tag = ::Tags::AnalyticDataBase;
139  evolved_vars->assign_subset(
140  Parallel::get<analytic_data_tag>(local_cache)
141  .variables(inertial_coords,
142  typename evolved_variables_tag::tags_list{}));
143  }),
144  evolution::is_analytic_solution<typename Metavariables::initial_data>{},
145  cache);
146 
147  // Get hydro variables
148  HydroVars hydro_variables{num_grid_points};
149  make_overloader(
150  [ initial_time, &inertial_coords ](
151  std::true_type /*is_analytic_solution*/,
152  const gsl::not_null<HydroVars*> hydro_vars,
153  const auto& local_cache) noexcept {
154  using solution_tag = ::Tags::AnalyticSolutionBase;
155  hydro_vars->assign_subset(
156  Parallel::get<solution_tag>(local_cache)
157  .variables(inertial_coords, initial_time,
158  typename hydro_variables_tag::tags_list{}));
159  },
160  [&inertial_coords](std::false_type /*is_analytic_solution*/,
161  const gsl::not_null<HydroVars*> hydro_vars,
162  const auto& local_cache) noexcept {
163  using analytic_data_tag = ::Tags::AnalyticDataBase;
164  hydro_vars->assign_subset(
165  Parallel::get<analytic_data_tag>(local_cache)
166  .variables(inertial_coords,
167  typename hydro_variables_tag::tags_list{}));
168  })(
169  evolution::is_analytic_solution<typename Metavariables::initial_data>{},
170  make_not_null(&hydro_variables), cache);
171 
172  M1Vars m1_variables{num_grid_points, -1.};
173 
174  return std::make_tuple(
175  db::create_from<db::RemoveTags<>, simple_tags, compute_tags>(
176  std::move(box), std::move(hydro_variables),
177  std::move(m1_variables)));
178  }
179 };

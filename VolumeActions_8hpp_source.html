42 /*!
43  * \ingroup ObserverGroup
44  * \brief Send volume tensor data to the observer.
45  *
46  * The caller of this Action (which is to be invoked on the Observer parallel
47  * component) must pass in an `observation_id` used to uniquely identify the
48  * observation in time, the name of the `h5::VolumeData` subfile in the HDF5
49  * file (e.g. `/element_data`, where the slash is important), the contributing
50  * parallel component element's component id, a vector of the `TensorComponent`s
51  * to be written to disk, and an `Index<Dim>` of the extents of the volume.
52  *
53  * \warning Currently this action can only be called once per `observation_id`
54  * per `array_component_id`. Calling it more often is undefined behavior and may
55  * result in incomplete data being written and/or memory leaks. The reason is
56  * that we do not register the number of times a component with an ID at a
57  * specific observation id will call the observer. However, this will be a
58  * feature implemented in the future.
59  */
60 struct ContributeVolumeData {
61  template <
62  typename ParallelComponent, typename DbTagsList, typename Metavariables,
63  typename ArrayIndex, size_t Dim,
64  Requires<tmpl::list_contains_v<DbTagsList, Tags::TensorData>> = nullptr>
65  static void apply(db::DataBox<DbTagsList>& box,
66  Parallel::ConstGlobalCache<Metavariables>& cache,
67  const ArrayIndex& /*array_index*/,
68  const observers::ObservationId& observation_id,
69  const std::string& subfile_name,
70  const observers::ArrayComponentId& array_component_id,
71  std::vector<TensorComponent>&& in_received_tensor_data,
72  const Index<Dim>& received_extents) noexcept {
73  db::mutate<Tags::TensorData>(
74  make_not_null(&box),
75  [
76  &cache, &observation_id, &array_component_id, &received_extents,
77  received_tensor_data = std::move(in_received_tensor_data),
78  &subfile_name
79  ](const gsl::not_null<db::item_type<Tags::TensorData>*> volume_data,
80  const std::unordered_set<ArrayComponentId>&
81  volume_component_ids) mutable noexcept {
82  if (volume_data->count(observation_id) == 0 or
83  volume_data->at(observation_id).count(array_component_id) == 0) {
84  std::vector<size_t> extents(received_extents.begin(),
85  received_extents.end());
86  volume_data->operator[](observation_id)
87  .emplace(
88  array_component_id,
89  ExtentsAndTensorVolumeData(
90  std::move(extents), std::move(received_tensor_data)));
91  } else {
92  auto& current_data =
93  volume_data->at(observation_id).at(array_component_id);
94  ASSERT(
95  alg::equal(current_data.extents, received_extents),
96  "The extents from the same volume component at a specific "
97  "observation should always be the same. For example, the "
98  "extents of a dG element should be the same for all calls to "
99  "ContributeVolumeData that occur at the same time.");
100  current_data.tensor_components.insert(
101  current_data.tensor_components.end(),
102  std::make_move_iterator(received_tensor_data.begin()),
103  std::make_move_iterator(received_tensor_data.end()));
104  }
105 
106  // Check if we have received all "volume" data from the registered
107  // elements. If so we copy it to the nodegroup volume writer.
108  if (volume_data->at(observation_id).size() ==
109  volume_component_ids.size()) {
110  auto& local_writer = *Parallel::get_parallel_component<
111  ObserverWriter<Metavariables>>(cache)
112  .ckLocalBranch();
113  Parallel::simple_action<ContributeVolumeDataToWriter>(
114  local_writer, observation_id, subfile_name,
115  std::move((*volume_data)[observation_id]));
116  volume_data->erase(observation_id);
117  }
118  },
119  db::get<Tags::VolumeArrayComponentIds>(box));
120  }
121 };

123 /*!
124  * \ingroup ObserverGroup
125  * \brief Move data to the observer writer for writing to disk.
126  */
127 struct ContributeVolumeDataToWriter {
128  template <typename ParallelComponent, typename DbTagsList,
129  typename Metavariables, typename ArrayIndex,
130  Requires<tmpl::list_contains_v<DbTagsList, Tags::TensorData> and
131  tmpl::list_contains_v<DbTagsList,
132  Tags::VolumeObserversContributed>> =
133  nullptr>
134  static void apply(db::DataBox<DbTagsList>& box,
135  Parallel::ConstGlobalCache<Metavariables>& cache,
136  const ArrayIndex& /*array_index*/,
137  const observers::ObservationId& observation_id,
138  const std::string& subfile_name,
139  std::unordered_map<observers::ArrayComponentId,
140  ExtentsAndTensorVolumeData>&&
141  in_volume_data) noexcept {
142  // This is the number of callers that have registered (that are associated
143  // with the observation type of this observation_id).
144  // We expect that this Action will be called once by each of them.
145  const auto expected_number_of_calls = [&box, &observation_id ]() noexcept {
146  const auto hash = observation_id.observation_type_hash();
147  const auto& registered = db::get<Tags::VolumeObserversRegistered>(box);
148  return (registered.count(hash) == 1) ? registered.at(hash).size() : 0;
149  }
150  ();
151  db::mutate<Tags::TensorData, Tags::VolumeObserversContributed>(
152  make_not_null(&box),
153  [
154  &cache, &expected_number_of_calls, &observation_id,
155  in_volume_data = std::move(in_volume_data), &subfile_name
156  ](const gsl::not_null<std::unordered_map<
157  observers::ObservationId,
158  std::unordered_map<observers::ArrayComponentId,
159  ExtentsAndTensorVolumeData>>*>
160  volume_data,
161  const gsl::not_null<
162  std::unordered_map<observers::ObservationId, size_t>*>
163  volume_observers_contributed) mutable noexcept {
164  if (volume_data->count(observation_id) == 0) {
165  // We haven't been called before on this processing element.
166  volume_data->operator[](observation_id) = std::move(in_volume_data);
167  (*volume_observers_contributed)[observation_id] = 1;
168  } else {
169  auto& current_data = volume_data->at(observation_id);
170  current_data.insert(std::make_move_iterator(in_volume_data.begin()),
171  std::make_move_iterator(in_volume_data.end()));
172  (*volume_observers_contributed)[observation_id]++;
173  }
174  // Check if we have received all "volume" data from the Observer
175  // group. If so we write to disk.
176  if (volume_observers_contributed->at(observation_id) ==
177  expected_number_of_calls) {
178  Parallel::threaded_action<ThreadedActions::WriteVolumeData>(
179  Parallel::get_parallel_component<ObserverWriter<Metavariables>>(
180  cache)[static_cast<size_t>(Parallel::my_node())],
181  observation_id, subfile_name);
182  volume_observers_contributed->erase(observation_id);
183  }
184  });
185  }
186 };
187 } // namespace Actions

190 /*!
191  * \ingroup ObserverGroup
192  * \brief Writes volume data at the `observation_id` to disk.
193  */
194 struct WriteVolumeData {
195  template <
196  typename ParallelComponent, typename DbTagsList, typename Metavariables,
197  typename ArrayIndex,
198  Requires<tmpl::list_contains_v<DbTagsList, Tags::H5FileLock> and
199  tmpl::list_contains_v<DbTagsList, Tags::TensorData>> = nullptr>
200  static void apply(db::DataBox<DbTagsList>& box,
201  const Parallel::ConstGlobalCache<Metavariables>& cache,
202  const ArrayIndex& /*array_index*/,
203  const gsl::not_null<CmiNodeLock*> node_lock,
204  const observers::ObservationId& observation_id,
205  const std::string& subfile_name) noexcept {
206  // Get data from the DataBox in a thread-safe manner
207  Parallel::lock(node_lock);
208  std::unordered_map<observers::ArrayComponentId, ExtentsAndTensorVolumeData>
209  volume_data{};
210  CmiNodeLock file_lock;
211  db::mutate<Tags::H5FileLock, Tags::TensorData>(
212  make_not_null(&box),
213  [&observation_id, &file_lock, &volume_data ](
214  const gsl::not_null<CmiNodeLock*> in_file_lock,
215  const gsl::not_null<db::item_type<Tags::TensorData>*>
216  in_volume_data) noexcept {
217  volume_data = std::move((*in_volume_data)[observation_id]);
218  in_volume_data->erase(observation_id);
219  file_lock = *in_file_lock;
220  });
221  Parallel::unlock(node_lock);
222 
223  // Write to file. We use a separate node lock because writing can be very
224  // time consuming (it's network dependent, depends on how full the disks
225  // are, what other users are doing, etc.) and we want to be able to continue
226  // to work on the nodegroup while we are writing data to disk.
227  Parallel::lock(&file_lock);
228  {
229  // Scoping is for closing HDF5 file before we release the lock.
230  const auto& file_prefix = Parallel::get<Tags::VolumeFileName>(cache);
231  h5::H5File<h5::AccessType::ReadWrite> h5file(
232  file_prefix + std::to_string(Parallel::my_node()) + ".h5", true);
233  constexpr size_t version_number = 0;
234  auto& volume_file =
235  h5file.try_insert<h5::VolumeData>(subfile_name, version_number);
236  std::vector<ExtentsAndTensorVolumeData> dg_elements;
237  dg_elements.reserve(volume_data.size());
238  for (auto id_and_element : volume_data) {
239  dg_elements.push_back(id_and_element.second);
240  }
241  // Write the data to the file
242  volume_file.write_volume_data(observation_id.hash(),
243  observation_id.value(), dg_elements);
244  }
245  Parallel::unlock(&file_lock);
246  }
247 };

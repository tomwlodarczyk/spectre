14 /*!
15  * \ingroup TensorGroup
16  * \brief Compute the Euclidean magnitude of a rank-1 tensor
17  *
18  * \details
19  * Computes the square root of the sum of the squares of the components of
20  * the rank-1 tensor.
21  */
22 template <typename DataType, typename Index>
23 Scalar<DataType> magnitude(
24  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector) noexcept {
25  return Scalar<DataType>{sqrt(get(dot_product(vector, vector)))};
26 }

28 /*!
29  * \ingroup TensorGroup
30  * \brief Compute the magnitude of a rank-1 tensor
31  *
32  * \details
33  * Returns the square root of the input tensor contracted twice with the given
34  * metric.
35  */
36 template <typename DataType, typename Index>
37 Scalar<DataType> magnitude(
38  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector,
39  const Tensor<DataType, Symmetry<1, 1>,
40  index_list<change_index_up_lo<Index>,
41  change_index_up_lo<Index>>>& metric) noexcept {
42  return Scalar<DataType>{sqrt(get(dot_product(vector, vector, metric)))};
43 }

45 /// \ingroup TensorGroup
46 /// \brief Compute square root of the Euclidean magnitude of a rank-0 tensor
47 ///
48 /// \details
49 /// Computes the square root of the absolute value of the scalar.
50 template <typename DataType>
51 Scalar<DataType> sqrt_magnitude(
52  const Scalar<DataType>& input) noexcept {
53  return Scalar<DataType>{sqrt(abs(get(input)))};

112 /// \ingroup DataBoxTagsGroup
113 /// \ingroup DataStructuresGroup
114 /// Normalizes the (co)vector represented by Tag
115 ///
116 /// This tag inherits from `Tags::Normalized<Tag>`
117 template <typename Tag>
118 struct NormalizedCompute : Normalized<Tag>, db::ComputeTag {
119  using base = Normalized<Tag>;
120  static constexpr auto function(
121  const db::const_item_type<Tag>&
122  vector_in, // Compute items need to take const references
123  const db::const_item_type<Magnitude<Tag>>& magnitude) noexcept {
124  auto vector = vector_in;
125  for (size_t d = 0; d < vector.index_dim(0); ++d) {
126  vector.get(d) /= get(magnitude);
127  }
128  return vector;
129  }
130  using argument_tags = tmpl::list<Tag, Magnitude<Tag>>;
131 };

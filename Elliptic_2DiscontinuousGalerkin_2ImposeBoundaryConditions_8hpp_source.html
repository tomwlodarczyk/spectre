35 /*!
36  * \brief Packages data on external boundaries so that they represent
37  * homogeneous (zero) Dirichlet boundary conditions.
38  *
39  * This action imposes homogeneous boundary conditions on all fields in
40  * `system::primal_fields`. The fields are wrapped in
41  * `LinearSolver::Tags::Operand`. The result should be a subset of the
42  * `system::variables`. Because we are working with the linear solver operand,
43  * we cannot impose non-zero boundary conditions here. Instead, non-zero
44  * boundary conditions are handled as contributions to the linear solver source
45  * during initialization.
46  *
47  * \warning This actions works only for scalar fields right now. It should be
48  * considered a temporary solution and will have to be reworked for more
49  * involved boundary conditions.
50  *
51  * With:
52  * - `interior<Tag> =
53  * Tags::Interface<Tags::BoundaryDirectionsInterior<volume_dim>, Tag>`
54  * - `exterior<Tag> =
55  * Tags::Interface<Tags::BoundaryDirectionsExterior<volume_dim>, Tag>`
56  *
57  * Uses:
58  * - Metavariables:
59  * - `normal_dot_numerical_flux`
60  * - `temporal_id`
61  * - System:
62  * - `volume_dim`
63  * - `variables_tag`
64  * - `primal_fields`
65  * - ConstGlobalCache:
66  * - `normal_dot_numerical_flux`
67  * - DataBox:
68  * - `Tags::Element<volume_dim>`
69  * - `temporal_id`
70  * - `Tags::BoundaryDirectionsInterior<volume_dim>`
71  * - `Tags::BoundaryDirectionsExterior<volume_dim>`
72  * - `interior<variables_tag>`
73  * - `exterior<variables_tag>`
74  * - `interior<normal_dot_numerical_flux::type::argument_tags>`
75  * - `exterior<normal_dot_numerical_flux::type::argument_tags>`
76  *
77  * DataBox changes:
78  * - Modifies:
79  * - `exterior<variables_tag>`
80  * - `Tags::VariablesBoundaryData`
81  */
82 template <typename Metavariables>
83 struct ImposeHomogeneousDirichletBoundaryConditions {
84  using const_global_cache_tags =
85  tmpl::list<typename Metavariables::normal_dot_numerical_flux>;
86 
87  template <typename DbTags, typename... InboxTags, typename ArrayIndex,
88  typename ActionList, typename ParallelComponent>
89  static auto apply(db::DataBox<DbTags>& box,
90  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
91  const Parallel::ConstGlobalCache<Metavariables>& cache,
92  const ArrayIndex& /*array_index*/,
93  const ActionList /*meta*/,
94  const ParallelComponent* const /*meta*/) noexcept {
95  using system = typename Metavariables::system;
96  using dirichlet_tags = typename system::primal_fields;
97  constexpr size_t volume_dim = system::volume_dim;
98 
99  // Set the data on exterior (ghost) faces to impose the boundary conditions
100  db::mutate<::Tags::Interface<::Tags::BoundaryDirectionsExterior<volume_dim>,
101  typename system::variables_tag>>(
102  make_not_null(&box),
103  // Need to use system::volume_dim below instead of just
104  // volume_dim to avoid an ICE on gcc 7.
105  [](const gsl::not_null<db::item_type<::Tags::Interface<
106  ::Tags::BoundaryDirectionsExterior<system::volume_dim>,
107  typename system::variables_tag>>*>
108  exterior_boundary_vars,
109  const db::const_item_type<::Tags::Interface<
110  ::Tags::BoundaryDirectionsInterior<system::volume_dim>,
111  typename system::variables_tag>>& interior_vars) noexcept {
112  for (auto& exterior_direction_and_vars : *exterior_boundary_vars) {
113  auto& direction = exterior_direction_and_vars.first;
114  auto& exterior_vars = exterior_direction_and_vars.second;
115 
116  // By default, use the variables on the external boundary for the
117  // exterior
118  exterior_vars = interior_vars.at(direction);
119 
120  // For those variables where we have boundary conditions, impose
121  // zero Dirichlet b.c. here. The non-zero boundary conditions are
122  // handled as contributions to the source in InitializeElement.
123  // Imposing them here would not work because we are working with the
124  // linear solver operand.
125  tmpl::for_each<dirichlet_tags>([
126  &interior_vars, &exterior_vars, &direction
127  ](auto dirichlet_tag_val) noexcept {
128  using dirichlet_tag =
129  tmpl::type_from<decltype(dirichlet_tag_val)>;
130  using dirichlet_operand_tag =
131  LinearSolver::Tags::Operand<dirichlet_tag>;
132  // Use mirror principle. This only works for scalars right now.
133  get(get<dirichlet_operand_tag>(exterior_vars)) =
134  -1. *
135  get<dirichlet_operand_tag>(interior_vars.at(direction)).get();
136  });
137  }
138  },
139  get<::Tags::Interface<::Tags::BoundaryDirectionsInterior<volume_dim>,
140  typename system::variables_tag>>(box));
141 
142  const auto& element = db::get<::Tags::Element<volume_dim>>(box);
143  const auto& temporal_id = db::get<typename Metavariables::temporal_id>(box);
144 
145  const auto& normal_dot_numerical_flux_computer =
146  get<typename Metavariables::normal_dot_numerical_flux>(cache);
147 
148  auto interior_data = DgActions_detail::compute_local_mortar_data(
149  box, normal_dot_numerical_flux_computer,
150  ::Tags::BoundaryDirectionsInterior<volume_dim>{}, Metavariables{});
151 
152  auto exterior_data = DgActions_detail::compute_packaged_data(
153  box, normal_dot_numerical_flux_computer,
154  ::Tags::BoundaryDirectionsExterior<volume_dim>{}, Metavariables{});
155 
156  // Store local and packaged data on the mortars
157  for (const auto& direction : element.external_boundaries()) {
158  const auto mortar_id = std::make_pair(
159  direction, ElementId<volume_dim>::external_boundary_id());
160 
161  db::mutate<::Tags::VariablesBoundaryData>(
162  make_not_null(&box),
163  [
164  &mortar_id, &temporal_id, &direction, &interior_data, &exterior_data
165  ](const gsl::not_null<
166  db::item_type<::Tags::VariablesBoundaryData, DbTags>*>
167  mortar_data) noexcept {
168  mortar_data->at(mortar_id).local_insert(
169  temporal_id, std::move(interior_data.at(direction)));
170  mortar_data->at(mortar_id).remote_insert(
171  temporal_id, std::move(exterior_data.at(direction)));
172  });
173  }
174  return std::forward_as_tuple(std::move(box));
175  }
176 };

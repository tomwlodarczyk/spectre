75  using initial_residual_magnitude_tag =

77  using orthogonalization_iteration_id_tag =
78  db::add_tag_prefix<LinearSolver::Tags::Orthogonalization,
79  LinearSolver::Tags::IterationId>;
80  using orthogonalization_history_tag =
81  db::add_tag_prefix<LinearSolver::Tags::OrthogonalizationHistory,
82  fields_tag>;
83 
84  public:
85  template <typename DbTagsList, typename... InboxTags, typename ArrayIndex,
86  typename Metavariables, typename ActionList,
87  typename ParallelComponent>
88  static auto apply(db::DataBox<DbTagsList>& box,
89  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
90  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
91  const ArrayIndex& /*array_index*/,
92  const ActionList /*meta*/,
93  const ParallelComponent* const /*meta*/) noexcept {
94  using compute_tags =
95  db::AddComputeTags<LinearSolver::Tags::HasConvergedCompute<fields_tag>>;
96  return std::make_tuple(
97  ::Initialization::merge_into_databox<
98  InitializeResidualMonitor,
99  db::AddSimpleTags<residual_magnitude_tag,
100  initial_residual_magnitude_tag,
101  LinearSolver::Tags::IterationId,
102  orthogonalization_iteration_id_tag,
103  orthogonalization_history_tag>,
104  compute_tags>(std::move(box),
105  std::numeric_limits<double>::signaling_NaN(),
106  std::numeric_limits<double>::signaling_NaN(),
107  db::item_type<LinearSolver::Tags::IterationId>{0},
108  db::item_type<orthogonalization_iteration_id_tag>{0},
109  DenseMatrix<double>{2, 1, 0.}),
110  true);
111  }
112 };

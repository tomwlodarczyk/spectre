77  Tags::NumericalFlux<Poisson::FirstOrderInternalPenaltyFlux<Dim>>;
78 
79  // Collect all items to store in the cache.
80  using const_global_cache_tags =
81  tmpl::list<analytic_solution_tag,
82  elliptic::Tags::FluxesComputer<typename system::fluxes>>;

102  elliptic::dg::Actions::InitializeFluxes<Metavariables>,
103  // Initialization is done. Avoid introducing an extra phase by
104  // advancing the linear solver to the first step here.
105  typename linear_solver::prepare_step,
106  Initialization::Actions::RemoveOptionsAndTerminatePhase>;
107 
108  // Specify all parallel components that will execute actions at some point.
109  using component_list = tmpl::append<
110  tmpl::list<elliptic::DgElementArray<
111  Metavariables,
112  tmpl::list<

135  typename linear_solver::perform_step,
136  typename linear_solver::prepare_step>>>>>,
137  typename linear_solver::component_list,
138  tmpl::list<observers::Observer<Metavariables>,
139  observers::ObserverWriter<Metavariables>>>;
140 
141  // Specify the transitions between phases.
142  static Phase determine_next_phase(
143  const Phase& current_phase,
144  const Parallel::CProxy_ConstGlobalCache<
145  Metavariables>& /*cache_proxy*/) noexcept {
146  switch (current_phase) {
147  case Phase::Initialization:
148  return Phase::RegisterWithObserver;
149  case Phase::RegisterWithObserver:
150  return Phase::Solve;
151  case Phase::Solve:
152  return Phase::Exit;
153  case Phase::Exit:
154  ERROR(
155  "Should never call determine_next_phase with the current phase "
156  "being 'Exit'");
157  default:
158  ERROR(
159  "Unknown type of phase. Did you static_cast<Phase> an integral "
160  "value?");
161  }
162  }
163 };

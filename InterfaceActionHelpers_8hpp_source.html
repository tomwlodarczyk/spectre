22 template <typename Metavariables, typename DataBoxType, typename DirectionsTag,
23  typename NumericalFlux, size_t VolumeDim = DirectionsTag::volume_dim,
24  typename FluxCommTypes = dg::FluxCommunicationTypes<Metavariables>,
25  typename PackagedData = typename FluxCommTypes::PackagedData>
26 DirectionMap<VolumeDim, PackagedData> compute_packaged_data(

28  const NumericalFlux& normal_dot_numerical_flux_computer,
29  const DirectionsTag /*meta*/, const Metavariables /*meta*/) noexcept {
30  return interface_apply<
31  DirectionsTag,
32  tmpl::flatten<tmpl::list<::Tags::Mesh<VolumeDim - 1>,
33  typename NumericalFlux::argument_tags>>,

39  normal_dot_numerical_flux_computer.package_data(make_not_null(&result),
40  args...);
41  return result;

43  box);

46 template <typename Metavariables, typename DataBoxType, typename DirectionsTag,
47  typename NumericalFlux, size_t VolumeDim = DirectionsTag::volume_dim,
48  typename FluxCommTypes = dg::FluxCommunicationTypes<Metavariables>,
49  typename LocalData = typename FluxCommTypes::LocalData>
50 DirectionMap<VolumeDim, LocalData> compute_local_mortar_data(
51  const DataBoxType& box,
52  const NumericalFlux& normal_dot_numerical_flux_computer,
53  const DirectionsTag /*meta*/, const Metavariables /*meta*/) noexcept {
54  using normal_dot_fluxes_tag =
55  Tags::Interface<DirectionsTag,
56  typename FluxCommTypes::normal_dot_fluxes_tag>;
57 
58  const auto& face_meshes =
59  db::get<Tags::Interface<DirectionsTag, Tags::Mesh<VolumeDim - 1>>>(box);
60  const auto& magnitude_of_face_normals = db::get<Tags::Interface<
61  DirectionsTag, Tags::Magnitude<Tags::UnnormalizedFaceNormal<VolumeDim>>>>(
62  box);
63  const auto& normal_dot_fluxes = db::get<normal_dot_fluxes_tag>(box);
64 
65  const auto packaged_data =
66  compute_packaged_data(box, normal_dot_numerical_flux_computer,
67  DirectionsTag{}, Metavariables{});
68 
69  DirectionMap<VolumeDim, LocalData> all_interface_data{};
70  for (const auto& direction : get<DirectionsTag>(box)) {
71  LocalData interface_data{};
72  interface_data.magnitude_of_face_normal =
73  magnitude_of_face_normals.at(direction);
74  interface_data.mortar_data.initialize(
75  face_meshes.at(direction).number_of_grid_points());
76  interface_data.mortar_data.assign_subset(packaged_data.at(direction));
77 
78  if (tmpl::size<typename FluxCommTypes::LocalMortarData::tags_list>::value !=
79  tmpl::size<typename FluxCommTypes::PackagedData::tags_list>::value) {
80  // The local fluxes were not (all) included in the packaged
81  // data, so we need to add them to the mortar data
82  // explicitly.
83  interface_data.mortar_data.assign_subset(normal_dot_fluxes.at(direction));
84  }
85  all_interface_data.insert({direction, std::move(interface_data)});
86  }
87  return all_interface_data;
88 }

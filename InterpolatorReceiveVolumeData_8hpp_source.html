25 /// \ingroup ActionsGroup
26 /// \brief Adds volume data from an `Element`.
27 ///
28 /// Attempts to interpolate if it already has received target points from
29 /// any InterpolationTargets.
30 ///
31 /// Uses:
32 /// - DataBox:
33 /// - `Tags::NumberOfElements`
34 ///
35 /// DataBox changes:
36 /// - Adds: nothing
37 /// - Removes: nothing
38 /// - Modifies:
39 /// - `Tags::VolumeVarsInfo<Metavariables>`
40 /// - `Tags::InterpolatedVarsHolders<Metavariables>`
41 struct InterpolatorReceiveVolumeData {
42  template <
43  typename ParallelComponent, typename DbTags, typename Metavariables,
44  typename ArrayIndex, size_t VolumeDim,
45  Requires<tmpl::list_contains_v<DbTags, Tags::NumberOfElements>> = nullptr>
46  static void apply(
47  db::DataBox<DbTags>& box,
48  Parallel::ConstGlobalCache<Metavariables>& cache,
49  const ArrayIndex& /*array_index*/,
50  const typename Metavariables::temporal_id::type& temporal_id,
51  const ElementId<VolumeDim>& element_id, const ::Mesh<VolumeDim>& mesh,
52  Variables<typename Metavariables::interpolator_source_vars>&&
53  vars) noexcept {
54  bool add_temporal_ids_to_targets = false;
55  db::mutate<Tags::VolumeVarsInfo<Metavariables>>(
56  make_not_null(&box),
57  [
58  &add_temporal_ids_to_targets, &temporal_id, &element_id, &mesh, &vars
59  ](const gsl::not_null<
60  db::item_type<Tags::VolumeVarsInfo<Metavariables>>*>
61  container) noexcept {
62  if (container->find(temporal_id) == container->end()) {
63  add_temporal_ids_to_targets = true;
64  container->emplace(
65  temporal_id,
66  std::unordered_map<
67  ElementId<VolumeDim>,
68  typename Tags::VolumeVarsInfo<Metavariables>::Info>{});
69  }
70  container->at(temporal_id)
71  .emplace(std::make_pair(
72  element_id,
73  typename Tags::VolumeVarsInfo<Metavariables>::Info{
74  mesh, std::move(vars)}));
75  });
76 
77  // Try to interpolate data for all InterpolationTargets.
78  tmpl::for_each<typename Metavariables::interpolation_target_tags>([
79  &add_temporal_ids_to_targets, &box, &cache, &temporal_id
80  ](auto tag_v) noexcept {
81  using tag = typename decltype(tag_v)::type;
82 
83  // The first time (and only the first time) that this interpolator
84  // is called at this temporal_id, tell all the
85  // InterpolationTargets that they will interpolate at this
86  // temporal_id.
87  if (add_temporal_ids_to_targets) {
88  auto& target = Parallel::get_parallel_component<
89  InterpolationTarget<Metavariables, tag>>(cache);
90  Parallel::simple_action<AddTemporalIdsToInterpolationTarget<tag>>(
91  target, std::vector<typename Metavariables::temporal_id::type>{
92  temporal_id});
93  }
94 
95  try_to_interpolate<tag>(make_not_null(&box), make_not_null(&cache),
96  temporal_id);
97  });
98  }
99 };

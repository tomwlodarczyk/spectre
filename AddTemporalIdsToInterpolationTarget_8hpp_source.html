20 /// \ingroup ActionsGroup
21 /// \brief Adds `temporal_id`s on which this InterpolationTarget
22 /// should be triggered.
23 ///
24 /// Uses:
25 /// - DataBox:
26 /// - `Tags::TemporalIds<Metavariables>`
27 /// - `Tags::CompletedTemporalIds<Metavariables>`
28 ///
29 /// DataBox changes:
30 /// - Adds: nothing
31 /// - Removes: nothing
32 /// - Modifies:
33 /// - `Tags::TemporalIds<Metavariables>`
34 template <typename InterpolationTargetTag>
35 struct AddTemporalIdsToInterpolationTarget {
36  template <typename ParallelComponent, typename DbTags, typename Metavariables,
37  typename ArrayIndex,
38  Requires<tmpl::list_contains_v<
39  DbTags, typename Tags::TemporalIds<Metavariables>>> = nullptr>
40  static void apply(db::DataBox<DbTags>& box,
41  Parallel::ConstGlobalCache<Metavariables>& cache,
42  const ArrayIndex& /*array_index*/,
43  std::vector<typename Metavariables::temporal_id::type>&&
44  temporal_ids) noexcept {
45  const bool begin_interpolation =
46  db::get<Tags::TemporalIds<Metavariables>>(box).empty();
47 
48  db::mutate_apply<tmpl::list<Tags::TemporalIds<Metavariables>>,
49  tmpl::list<Tags::CompletedTemporalIds<Metavariables>>>(
50  [&temporal_ids](const gsl::not_null<
51  db::item_type<Tags::TemporalIds<Metavariables>>*>
52  ids,
53  const db::const_item_type<Tags::CompletedTemporalIds<
54  Metavariables>>& completed_ids) noexcept {
55  // We allow this Action to be called multiple times with the
56  // same temporal_ids (e.g. from each node of a NodeGroup
57  // ParallelComponent such as Interpolator). If multiple calls
58  // occur, we care only about the first one, and ignore the others.
59  // The first call will often begin interpolation.
60  // So if multiple calls occur, it is possible that some of them
61  // may arrive late, even after interpolation
62  // has been completed on one or more of the temporal_ids (and after
63  // that id has already been removed from `ids`). If this happens,
64  // we don't want to add the temporal_ids again. For that
65  // reason we keep track of the temporal_ids that we have already
66  // completed interpolation on. So here we do not add any temporal_ids
67  // that are already present in `ids` or `completed_ids`.
68  for (auto& id : temporal_ids) {
69  if (std::find(completed_ids.begin(), completed_ids.end(), id) ==
70  completed_ids.end() and
71  std::find(ids->begin(), ids->end(), id) == ids->end()) {
72  ids->push_back(id);
73  }
74  }
75  },
76  make_not_null(&box));
77 
78  // Begin interpolation if it is not already in progress
79  // (i.e. waiting for data), and if there are temporal_ids to
80  // interpolate. If there's an interpolation in progress, then a
81  // later interpolation will be started as soon as the earlier one
82  // finishes.
83  const auto& ids = db::get<Tags::TemporalIds<Metavariables>>(box);
84  if (begin_interpolation and not ids.empty()) {
85  auto& my_proxy =
86  Parallel::get_parallel_component<ParallelComponent>(cache);
87  Parallel::simple_action<
88  typename InterpolationTargetTag::compute_target_points>(my_proxy,
89  ids.front());
90  }
91  }
92 };
93 } // namespace Actions
94 } // namespace intrp

177 /// \ingroup ParallelGroup
178 /// \brief Given a list of tags and a tagged tuple containing items
179 /// created from input options, return a tagged tuple of items constructed
180 /// by calls to create_from_options for each tag in the list.
181 template <typename... Tags, typename... OptionTags>
182 tuples::TaggedTuple<Tags...> create_from_options(
183  const tuples::TaggedTuple<OptionTags...>& options,
184  tmpl::list<Tags...> /*meta*/) noexcept {
185  return {detail::create_initialization_item_from_options<Tags>(
186  options, typename Tags::option_tags{})...};

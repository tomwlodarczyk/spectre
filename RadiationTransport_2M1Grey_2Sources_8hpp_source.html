45 /*!
46  * \brief Compute the curvature source terms for the flux-balanced
47  * grey M1 radiation transport.
48  *
49  *
50  * A flux-balanced system has the generic form:
51  * \f[
52  * \partial_t U_i + \partial_m F^m(U_i) = S(U_i)
53  * \f]
54  *
55  * where \f$F^a()\f$ denotes the flux of a conserved variable \f$U_i\f$ and
56  * \f$S()\f$ denotes the source term for the conserved variable.
57  *
58  * For the grey M1 formalism (neglecting coupling to the fluid):
59  * \f{align*}
60  * S({\tilde E}) &= \alpha \tilde P^{ij} K_{ij} - \tilde S^i \partial_i
61  * \alpha,\\ S({\tilde S_i}) &= -\tilde E \partial_i \alpha + \tilde S_k
62  * \partial_i \beta^k
63  * + \frac{1}{2} \alpha \tilde P^{jk} \partial_i \gamma_{jk},
64  * \f}
65  *
66  * where \f${\tilde E}\f$, \f${\tilde S_i}\f$, \f${\tilde P}^{ij}\f$ are the
67  * densitized energy, momentum, and pressure tensor of the neutrinos/photons,
68  * \f$K_{ij}\f$ is the extrinsic curvature, and \f$\alpha\f$, \f$\beta^i\f$,
69  * \f$\gamma_{ij}\f$ are the lapse, shift and 3-metric.
70  *
71  * In the main function, we loop over all neutrino species, and then call
72  * the actual implementation of the curvature source terms.
73  */
74 template <typename... NeutrinoSpecies>
75 struct ComputeSources {
76  using return_tags = tmpl::list<
77  ::Tags::Source<Tags::TildeE<Frame::Inertial, NeutrinoSpecies>>...,
78  ::Tags::Source<Tags::TildeS<Frame::Inertial, NeutrinoSpecies>>...>;
79 
80  using argument_tags = tmpl::list<
81  Tags::TildeE<Frame::Inertial, NeutrinoSpecies>...,
82  Tags::TildeS<Frame::Inertial, NeutrinoSpecies>...,
83  Tags::TildeP<Frame::Inertial, NeutrinoSpecies>..., gr::Tags::Lapse<>,
84  ::Tags::deriv<gr::Tags::Lapse<DataVector>, tmpl::size_t<3>,
85  Frame::Inertial>,
86  ::Tags::deriv<gr::Tags::Shift<3, Frame::Inertial, DataVector>,
87  tmpl::size_t<3>, Frame::Inertial>,
88  ::Tags::deriv<gr::Tags::SpatialMetric<3, Frame::Inertial, DataVector>,
89  tmpl::size_t<3>, Frame::Inertial>,
90  gr::Tags::InverseSpatialMetric<3>,
91  gr::Tags::ExtrinsicCurvature<3, Frame::Inertial, DataVector>>;

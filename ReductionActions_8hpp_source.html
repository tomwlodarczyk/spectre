52 /*!
53  * \ingroup ObserverGroup
54  * \brief Send reduction data to the observer group.
55  *
56  * Once everything at a specific `ObservationId` has been contributed to the
57  * reduction, the groups reduce to their local nodegroup.
58  *
59  * The caller of this Action (which is to be invoked on the Observer parallel
60  * component) must pass in an `observation_id` used to uniquely identify the
61  * observation in time, the name of the `h5::Dat` subfile in the HDF5 file (e.g.
62  * `/element_data`, where the slash is important), a `std::vector<std::string>`
63  * of names of the quantities being reduced (e.g. `{"Time", "L1ErrorDensity",
64  * "L2ErrorDensity"}`), and the `Parallel::ReductionData` that holds the
65  * `ReductionDatums` containing info on how to do the reduction.
66  *
67  * The observer components need to know all expected reduction data types by
68  * compile-time, so they rely on the
69  * `Metavariables::observed_reduction_data_tags` alias to collect them in one
70  * place. To this end, each Action that contributes reduction data must expose
71  * the type alias as:
72  *
73  * \snippet ObserverHelpers.hpp make_reduction_data_tags
74  *
75  * Then, in the `Metavariables` collect them from all observing Actions like
76  * this:
77  *
78  * \snippet tests/Unit/Elliptic/Systems/Poisson/Actions/Test_Observe.cpp collect_reduction_data_tags
79  */
80 struct ContributeReductionData {
81  template <
82  typename ParallelComponent, typename DbTagsList, typename Metavariables,
83  typename ArrayIndex, typename... Ts,
84  Requires<
85  tmpl::list_contains_v<DbTagsList, Tags::ReductionData<Ts...>> and
86  tmpl::list_contains_v<DbTagsList, Tags::ReductionDataNames<Ts...>> and
87  tmpl::list_contains_v<DbTagsList,
88  Tags::ReductionObserversContributed>> = nullptr>
89  static auto apply(db::DataBox<DbTagsList>& box,
90  Parallel::ConstGlobalCache<Metavariables>& cache,
91  const ArrayIndex& /*array_index*/,
92  const observers::ObservationId& observation_id,
93  const std::string& subfile_name,
94  const std::vector<std::string>& reduction_names,
95  Parallel::ReductionData<Ts...>&& reduction_data) noexcept {
96  db::mutate<Tags::ReductionData<Ts...>, Tags::ReductionDataNames<Ts...>,
97  Tags::ReductionObserversContributed>(
98  make_not_null(&box),
99  [
100  &observation_id, reduction_data = std::move(reduction_data),
101  &reduction_names, &cache, &subfile_name
102  ](const gsl::not_null<std::unordered_map<
103  ObservationId, Parallel::ReductionData<Ts...>>*>
104  reduction_data_map,
105  const gsl::not_null<
106  std::unordered_map<ObservationId, std::vector<std::string>>*>
107  reduction_names_map,
108  const gsl::not_null<
109  std::unordered_map<observers::ObservationId, size_t>*>
110  reduction_observers_contributed,
111  const std::unordered_set<ArrayComponentId>&
112  reduction_component_ids) mutable noexcept {
113  auto& contribute_count =
114  (*reduction_observers_contributed)[observation_id];
115  if (reduction_data_map->count(observation_id) == 0) {
116  reduction_data_map->emplace(observation_id,
117  std::move(reduction_data));
118  reduction_names_map->emplace(observation_id, reduction_names);
119  contribute_count = 1;
120  } else {
121  ASSERT(
122  reduction_names_map->at(observation_id) == reduction_names,
123  "Reduction names differ at ObservationId "
124  << observation_id
125  << " with the expected names being "
126  // Use MakeString to get around ADL for STL stream operators
127  // (MakeString is in global namespace).
128  << (MakeString{} << reduction_names_map->at(observation_id)
129  << " and the received names being "
130  << reduction_names));
131  reduction_data_map->operator[](observation_id)
132  .combine(std::move(reduction_data));
133  contribute_count++;
134  }
135 
136  // Check if we have received all reduction data from the registered
137  // elements. If so, we reduce to the local ObserverWriter nodegroup.
138  if (contribute_count == reduction_component_ids.size()) {
139  const auto node_id = Parallel::my_node();
140  auto& local_writer = *Parallel::get_parallel_component<
141  ObserverWriter<Metavariables>>(cache)
142  .ckLocalBranch();
143  Parallel::threaded_action<ThreadedActions::WriteReductionData>(
144  local_writer, observation_id, subfile_name,
145  node_id == 0 ? std::move((*reduction_names_map)[observation_id])
146  : std::vector<std::string>{},
147  std::move((*reduction_data_map)[observation_id]));
148  reduction_data_map->erase(observation_id);
149  reduction_names_map->erase(observation_id);
150  reduction_observers_contributed->erase(observation_id);
151  }
152  },
153  db::get<Tags::ReductionArrayComponentIds>(box));
154  }
155 };
156 } // namespace Actions

159 /*!
160  * \ingroup ObserverGroup
161  * \brief Collect the reduction data from the Observer group on the
162  * ObserverWriter nodegroup before sending to node 0 for writing to disk.
163  *
164  * \note This action is also used for writing on node 0.
165  */
166 struct WriteReductionData {
167  private:
168  template <typename... Ts, size_t... Is>
169  static void write_data(const std::string& subfile_name,
170  std::vector<std::string>&& legend,
171  std::tuple<Ts...>&& data,
172  const std::string& file_prefix,
173  std::index_sequence<Is...> /*meta*/) noexcept {
174  static_assert(sizeof...(Ts) > 0,
175  "Must be reducing at least one piece of data");
176  std::vector<double> data_to_append{
177  static_cast<double>(std::get<Is>(data))...};
178 
179  h5::H5File<h5::AccessType::ReadWrite> h5file(file_prefix + ".h5", true);
180  constexpr size_t version_number = 0;
181  auto& time_series_file = h5file.try_insert<h5::Dat>(
182  subfile_name, std::move(legend), version_number);
183  time_series_file.append(data_to_append);
184  }
185 
186  public:
187  template <
188  typename ParallelComponent, typename DbTagsList, typename Metavariables,
189  typename ArrayIndex, typename... ReductionDatums,
190  Requires<tmpl::list_contains_v<
191  DbTagsList, Tags::ReductionData<ReductionDatums...>> and
192  tmpl::list_contains_v<
193  DbTagsList, Tags::ReductionDataNames<ReductionDatums...>> and
194  tmpl::list_contains_v<DbTagsList,
195  Tags::ReductionObserversContributed> and
196  tmpl::list_contains_v<DbTagsList, Tags::H5FileLock>> = nullptr>
197  static void apply(db::DataBox<DbTagsList>& box,
198  Parallel::ConstGlobalCache<Metavariables>& cache,
199  const ArrayIndex& /*array_index*/,
200  const gsl::not_null<CmiNodeLock*> node_lock,
201  const observers::ObservationId& observation_id,
202  const std::string& subfile_name,
203  std::vector<std::string>&& reduction_names,
204  Parallel::ReductionData<ReductionDatums...>&&
205  in_reduction_data) noexcept {
206  CmiNodeLock file_lock;
207  bool write_to_disk = false;
208  std::vector<std::string> legend{};
209  Parallel::lock(node_lock);
210  const auto expected_calls_from_this_node =
211  [&box, &observation_id ]() noexcept {
212  const auto hash = observation_id.observation_type_hash();
213  const auto& registered_reduction_observers =
214  db::get<Tags::ReductionObserversRegistered>(box);
215  return (registered_reduction_observers.count(hash) == 1)
216  ? registered_reduction_observers.at(hash).size()
217  : 0;
218  }
219  ();
220  const auto expected_calls_from_other_nodes =
221  [&box, &observation_id ]() noexcept {
222  const auto hash = observation_id.observation_type_hash();
223  const auto& registered_reduction_observers =
224  db::get<Tags::ReductionObserversRegisteredNodes>(box);
225  return (registered_reduction_observers.count(hash) == 1)
226  ? registered_reduction_observers.at(hash).size()
227  : 0;
228  }
229  ();
230  db::mutate<Tags::ReductionData<ReductionDatums...>,
231  Tags::ReductionDataNames<ReductionDatums...>,
232  Tags::ReductionObserversContributed, Tags::H5FileLock>(
233  make_not_null(&box),
234  [
235  &cache, &expected_calls_from_this_node,
236  &expected_calls_from_other_nodes, &file_lock, &in_reduction_data,
237  &legend, &observation_id, &reduction_names, &subfile_name,
238  &write_to_disk
239  ](const gsl::not_null<
240  db::item_type<Tags::ReductionData<ReductionDatums...>>*>
241  reduction_data,
242  const gsl::not_null<
243  std::unordered_map<ObservationId, std::vector<std::string>>*>
244  reduction_names_map,
245  const gsl::not_null<
246  std::unordered_map<observers::ObservationId, size_t>*>
247  reduction_observers_contributed,
248  const gsl::not_null<CmiNodeLock*> reduction_file_lock) noexcept {
249  auto& contribute_count =
250  (*reduction_observers_contributed)[observation_id];
251  const auto node_id = Parallel::my_node();
252 
253  if (node_id == 0 and not reduction_names.empty()) {
254  reduction_names_map->emplace(observation_id,
255  std::move(reduction_names));
256  }
257 
258  if (UNLIKELY(node_id == 0 and expected_calls_from_other_nodes == 0 and
259  expected_calls_from_this_node == 1)) {
260  // Here this Action will be called only once, so we take
261  // a shortcut and just write to disk.
262  write_to_disk = true;
263  file_lock = *reduction_file_lock;
264  legend = std::move(reduction_names_map->operator[](observation_id));
265  reduction_names_map->erase(observation_id);
266  } else if (reduction_data->count(observation_id) == 0) {
267  // This Action has been called for the first time,
268  // so all we need to do is move the input data to the
269  // reduction_data in the DataBox.
270  reduction_data->operator[](observation_id) =
271  std::move(in_reduction_data);
272  contribute_count = 1;
273  } else if (node_id == 0 and
274  contribute_count == expected_calls_from_this_node +
275  expected_calls_from_other_nodes -
276  1) {
277  // This is the final time this Action is called on node 0.
278  // The `-1` in the above `if` statement is because on the final
279  // step, contribute_count (which was incremented at the end of each
280  // previous call) should be one less than the expected number of
281  // calls.
282  in_reduction_data.combine(
283  std::move(reduction_data->operator[](observation_id)));
284  reduction_data->erase(observation_id);
285  reduction_observers_contributed->erase(observation_id);
286  write_to_disk = true;
287  file_lock = *reduction_file_lock;
288  legend = std::move(reduction_names_map->operator[](observation_id));
289  reduction_names_map->erase(observation_id);
290  } else {
291  // This Action is being called at least the second time
292  // (but not the final time if on node 0).
293  reduction_data->at(observation_id)
294  .combine(std::move(in_reduction_data));
295  contribute_count++;
296  }
297 
298  // Check if we have received all reduction data from the Observer
299  // group. If so we reduce to node 0 for writing to disk.
300  if (node_id != 0 and
301  reduction_observers_contributed->at(observation_id) ==
302  expected_calls_from_this_node) {
303  Parallel::threaded_action<WriteReductionData>(
304  Parallel::get_parallel_component<ObserverWriter<Metavariables>>(
305  cache)[0],
306  observation_id, subfile_name, std::vector<std::string>{},
307  std::move(reduction_data->operator[](observation_id)));
308  reduction_observers_contributed->erase(observation_id);
309  reduction_data->erase(observation_id);
310  }
311  });
312  Parallel::unlock(node_lock);
313 
314  if (write_to_disk) {
315  Parallel::lock(&file_lock);
316  in_reduction_data.finalize();
317  WriteReductionData::write_data(
318  subfile_name, std::move(legend), std::move(in_reduction_data.data()),
319  Parallel::get<Tags::ReductionFileName>(cache),
320  std::make_index_sequence<sizeof...(ReductionDatums)>{});
321  Parallel::unlock(&file_lock);
322  }
323  }
324 };

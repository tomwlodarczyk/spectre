20 /*!
21  * \brief Conjugate momentum to the spacetime metric.
22  *
23  * \details If \f$ \psi_{ab} \f$ is the spacetime metric, and \f$ N \f$ and
24  * \f$ N^i \f$ are the lapse and shift respectively, then we define
25  * \f$ \Pi_{ab} = -\frac{1}{N} ( \partial_t \psi_{ab} + N^{i} \phi_{iab} ) \f$
26  * where \f$\phi_{iab}\f$ is the variable defined by the tag Phi.
27  */

30  using type = tnsr::aa<DataVector, Dim, Frame>;
31  static std::string name() noexcept { return "Pi"; }
32 };

95 /*!
96  * \brief Gauge control parameter to specify the spatial weighting function
97  * that enters damped harmonic gauge source function.
98  *
99  * \details The evolution gauge source function is multiplied by a spatial
100  * weight function which controls where and how much it sources the damped
101  * harmonic gauge equation. The weight function is:
102  * \f$ W(x^i) = \exp[-(r/\sigma_r)^2] \f$.
103  * This weight function can be written with an extra factor inside the exponent
104  * in literature, e.g. \cite Deppe2018uye. We will absorb it here in
105  * \f$\sigma_r\f$. The parameter this tag tags is \f$ \sigma_r \f$.
106  */
107 template <typename Frame>
108 struct GaugeHSpatialWeightDecayWidth : db::SimpleTag {
109  using type = double;
110  static std::string name() noexcept { return "GaugeHSpatialWeightDecayWidth"; }
111  using option_tags = tmpl::list<OptionTags::GaugeHSpatialDecayWidth<Frame>>;
112  static double create_from_options(
113  const double gauge_spatial_decay_width) noexcept {
114  return gauge_spatial_decay_width;
115  }
116 };

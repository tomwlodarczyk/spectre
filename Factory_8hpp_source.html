31 template <typename BaseClass, typename CreateList, typename Metavariables,
32  Requires<(tmpl::size<CreateList>::value != 0)> = nullptr>
33 std::unique_ptr<BaseClass> create_derived(const std::string& id,
34  const Option& opts) {
35  using derived = tmpl::front<CreateList>;
36 
37  if (pretty_type::short_name<derived>() != id) {
38  return create_derived<BaseClass, tmpl::pop_front<CreateList>,
39  Metavariables>(id, opts);
40  }
41 
42  ASSERT((not create_derived<BaseClass, tmpl::pop_front<CreateList>,
43  Metavariables>(id, opts)),
44  "Duplicate factory id: " << id);
45 
46  return std::make_unique<derived>(opts.parse_as<derived, Metavariables>());
47 }

49 struct print_derived {
50  // Not a stream because brigand requires the functor to be copyable.
51  std::string value;
52  template <typename T>
53  void operator()(tmpl::type_<T> /*meta*/) noexcept {
54  // These are zero-based
55  const size_t name_col = 2;
56  const size_t help_col = 22;
57  const size_t end_col = 80;
58 
59  std::ostringstream ss;
60  ss << std::left
61  << std::setw(name_col) << ""
62  << std::setw(help_col - name_col - 1) << pretty_type::short_name<T>();
63  if (ss.str().size() >= help_col) {
64  ss << "\n" << std::setw(help_col - 1) << "";
65  }
66 
67  std::string help_snippet(T::help);
68  if (help_snippet.size() > end_col - help_col) {
69  help_snippet.resize(end_col - help_col - 3);
70  help_snippet += "...";
71  }
72  std::replace(help_snippet.begin(), help_snippet.end(), '\n', ' ');
73  ss << " " << help_snippet << "\n";
74 
75  value += ss.str();
76  }

87 template <typename BaseClass, typename Metavariables>
88 std::unique_ptr<BaseClass> create(const Option& options) {
89  const auto& node = options.node();
90  Option derived_opts(options.context());
91  derived_opts.append_context("While operating factory for " +
92  pretty_type::short_name<BaseClass>());
93  std::string id;
94  if (node.IsScalar()) {
95  id = node.as<std::string>();
96  } else if (node.IsMap()) {
97  if (node.size() != 1) {
98  PARSE_ERROR(derived_opts.context(),
99  "Expected a single class to create, got "
100  << node.size() << ":\n" << node);
101  }
102  id = node.begin()->first.as<std::string>();
103  derived_opts.set_node(node.begin()->second);
104  } else if (node.IsNull()) {
105  PARSE_ERROR(derived_opts.context(),
106  "Expected a class to create:\n" << help_derived<BaseClass>());
107  } else {
108  PARSE_ERROR(derived_opts.context(),
109  "Expected a class or a class with options, got:\n"
110  << node);
111  }
112  auto derived =
113  create_derived<BaseClass, typename BaseClass::creatable_classes,
114  Metavariables>(id, derived_opts);
115  if (derived != nullptr) {
116  return derived;
117  }
118  PARSE_ERROR(derived_opts.context(),
119  "Unknown Id '" << id << "'\n" << help_derived<BaseClass>());
120 }

26 auto evaluate(const T& te) {
27  static_assert(

29  "Must have the same number of indices on the LHS and RHS of a tensor "

31  using rhs = tmpl::transform<tmpl::remove_duplicates<typename T::args_list>,

37  "as well.");
38  return Tensor<typename T::type, typename T::symmetry, typename T::index_list>(
39  te, tmpl::list<LhsIndices...>{});

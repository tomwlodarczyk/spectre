36 /// \ingroup InitializationGroup
37 /// \brief Allocate and set variables needed for evolution of conservative
38 /// systems
39 ///
40 /// Uses:
41 /// - DataBox:
42 /// * `Tags::Mesh<Dim>`
43 ///
44 /// DataBox changes:
45 /// - Adds:
46 /// * System::variables_tag
47 /// * db::add_tag_prefix<Tags::Flux, System::variables_tag>
48 /// * db::add_tag_prefix<Tags::Source, System::variables_tag>
49 ///
50 /// - Removes: nothing
51 /// - Modifies: nothing
52 struct ConservativeSystem {
53  using initialization_tags = tmpl::list<Initialization::Tags::InitialTime>;
54 
55  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
56  typename ArrayIndex, typename ActionList,
57  typename ParallelComponent,
58  Requires<tmpl::list_contains_v<
59  typename db::DataBox<DbTagsList>::simple_item_tags,
60  Initialization::Tags::InitialTime>> = nullptr>
61  static auto apply(db::DataBox<DbTagsList>& box,
62  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
63  const Parallel::ConstGlobalCache<Metavariables>& cache,
64  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
65  const ParallelComponent* const /*meta*/) noexcept {
66  using system = typename Metavariables::system;
67  static_assert(system::is_in_flux_conservative_form,
68  "System is not in flux conservative form");
69  static constexpr size_t dim = system::volume_dim;
70  using variables_tag = typename system::variables_tag;
71  using fluxes_tag = db::add_tag_prefix<::Tags::Flux, variables_tag,
72  tmpl::size_t<dim>, Frame::Inertial>;
73  using sources_tag = db::add_tag_prefix<::Tags::Source, variables_tag>;
74  using simple_tags =
75  db::AddSimpleTags<variables_tag, fluxes_tag, sources_tag>;
76  using compute_tags = db::AddComputeTags<>;
77 
78  const size_t num_grid_points =
79  db::get<::Tags::Mesh<dim>>(box).number_of_grid_points();
80  typename variables_tag::type vars(num_grid_points);
81  typename fluxes_tag::type fluxes(num_grid_points);
82  typename sources_tag::type sources(num_grid_points);
83 
84  return std::make_tuple(initialize_vars(
85  merge_into_databox<ConservativeSystem, simple_tags, compute_tags>(
86  std::move(box), std::move(vars), std::move(fluxes),
87  std::move(sources)),
88  cache));
89  }
90 
91  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
92  typename ArrayIndex, typename ActionList,
93  typename ParallelComponent,
94  Requires<not tmpl::list_contains_v<
95  typename db::DataBox<DbTagsList>::simple_item_tags,
96  Initialization::Tags::InitialTime>> = nullptr>
97  static std::tuple<db::DataBox<DbTagsList>&&> apply(
98  db::DataBox<DbTagsList>& /*box*/,
99  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
100  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
101  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
102  const ParallelComponent* const /*meta*/) noexcept {
103  ERROR(
104  "Could not find dependency 'Initialization::Tags::InitialTime' in "
105  "DataBox.");
106  }
107 
108  private:
109  template <
110  typename DbTagsList, typename Metavariables,
111  Requires<Metavariables::system::has_primitive_and_conservative_vars> =
112  nullptr>
113  static auto initialize_vars(
114  db::DataBox<DbTagsList>&& box,
115  const Parallel::ConstGlobalCache<Metavariables>& cache) noexcept {
116  using system = typename Metavariables::system;
117  static constexpr size_t dim = system::volume_dim;
118  using primitives_tag = typename system::primitive_variables_tag;
119  using simple_tags =
120  db::AddSimpleTags<primitives_tag,
121  typename Metavariables::equation_of_state_tag>;
122  using compute_tags = db::AddComputeTags<>;
123 
124  const double initial_time = db::get<Initialization::Tags::InitialTime>(box);
125  using PrimitiveVars = typename primitives_tag::type;
126 
127  const size_t num_grid_points =
128  db::get<::Tags::Mesh<dim>>(box).number_of_grid_points();
129 
130  const auto& inertial_coords =
131  db::get<::Tags::Coordinates<dim, Frame::Inertial>>(box);
132 
133  // Set initial data from analytic solution
134  PrimitiveVars primitive_vars{num_grid_points};
135  auto equation_of_state = make_overloader(
136  [ initial_time, &
137  inertial_coords ](std::true_type /*is_analytic_solution*/,
138  const gsl::not_null<PrimitiveVars*> prim_vars,
139  const auto& local_cache) noexcept {
140  using solution_tag = ::Tags::AnalyticSolutionBase;
141  prim_vars->assign_subset(
142  Parallel::get<solution_tag>(local_cache)
143  .variables(
144  inertial_coords, initial_time,
145  typename Metavariables::analytic_variables_tags{}));
146  return Parallel::get<solution_tag>(local_cache).equation_of_state();
147  },
148  [&inertial_coords](std::false_type /*is_analytic_solution*/,
149  const gsl::not_null<PrimitiveVars*> prim_vars,
150  const auto& local_cache) noexcept {
151  using analytic_data_tag = ::Tags::AnalyticDataBase;
152  prim_vars->assign_subset(
153  Parallel::get<analytic_data_tag>(local_cache)
154  .variables(
155  inertial_coords,
156  typename Metavariables::analytic_variables_tags{}));
157  return Parallel::get<analytic_data_tag>(local_cache)
158  .equation_of_state();
159  })(
160  evolution::is_analytic_solution<typename Metavariables::initial_data>{},
161  make_not_null(&primitive_vars), cache);
162 
163  return Initialization::merge_into_databox<ConservativeSystem, simple_tags,
164  compute_tags>(
165  std::move(box), std::move(primitive_vars),
166  std::move(equation_of_state));
167  }
168 
169  template <typename DbTagsList, typename Metavariables,
170  Requires<not Metavariables::system::
171  has_primitive_and_conservative_vars> = nullptr>
172  static auto initialize_vars(
173  db::DataBox<DbTagsList>&& box,
174  const Parallel::ConstGlobalCache<Metavariables>& cache) noexcept {
175  using system = typename Metavariables::system;
176  static constexpr size_t dim = system::volume_dim;
177  using variables_tag = typename system::variables_tag;
178 
179  const double initial_time =
180  db::get<Initialization::Tags::InitialTime>(box);
181  const auto& inertial_coords =
182  db::get<::Tags::Coordinates<dim, Frame::Inertial>>(box);
183 
184  // Set initial data from analytic solution
185  using Vars = typename variables_tag::type;
186  using solution_tag = ::Tags::AnalyticSolutionBase;
187  db::mutate<variables_tag>(
188  make_not_null(&box), [&cache, &inertial_coords, initial_time ](
189  const gsl::not_null<Vars*> vars) noexcept {
190  vars->assign_subset(Parallel::get<solution_tag>(cache).variables(
191  inertial_coords, initial_time, typename Vars::tags_list{}));
192  });
193 
194  return std::move(box);
195  }
196 };

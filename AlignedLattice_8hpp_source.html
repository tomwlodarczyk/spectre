21 /// \brief Create a Domain consisting of multiple aligned Blocks arrayed in a
22 /// lattice.
23 ///
24 /// This is useful for setting up problems with piecewise smooth initial data,
25 /// problems that specify different boundary conditions on distinct parts of
26 /// the boundary, or problems that need different length scales initially.
27 ///
28 /// \note Adaptive mesh refinement can never join Block%s, so use the fewest
29 /// number of Block%s that your problem needs. More initial Element%s can be
30 /// created by specifying a larger `InitialRefinement`.
31 template <size_t VolumeDim, typename TargetFrame>
32 class AlignedLattice : public DomainCreator<VolumeDim, TargetFrame> {
33  public:
34  struct BlockBounds {
35  using type = std::array<std::vector<double>, VolumeDim>;
36  static constexpr OptionString help = {
37  "Coordinates of block boundaries in each dimension."};
38  };
39 
40  struct IsPeriodicIn {
41  using type = std::array<bool, VolumeDim>;
42  static constexpr OptionString help = {
43  "Whether the domain is periodic in each dimension."};
44  static type default_value() noexcept {
45  return make_array<VolumeDim>(false);
46  }
47  };
48 
49  struct InitialRefinement {
50  using type = std::array<size_t, VolumeDim>;
51  static constexpr OptionString help = {
52  "Initial refinement level in each dimension."};
53  };
54 
55  struct InitialGridPoints {
56  using type = std::array<size_t, VolumeDim>;
57  static constexpr OptionString help = {
58  "Initial number of grid points in each dimension."};
59  };
60 
61  struct BlocksToExclude {
62  using type = std::vector<std::array<size_t, VolumeDim>>;
63  static constexpr OptionString help = {
64  "List of Block indices to exclude, if any."};
65  static type default_value() noexcept {
66  return std::vector<std::array<size_t, VolumeDim>>{};
67  }
68  };
69 
70  using options = tmpl::list<BlockBounds, IsPeriodicIn, InitialRefinement,
71  InitialGridPoints, BlocksToExclude>;
72 
73  static constexpr OptionString help = {
74  "AlignedLattice creates a regular lattice of blocks whose corners are\n"
75  "given by tensor products of the specified BlockBounds. Each Block in\n"
76  "the lattice is identified by a VolumeDim-tuple of zero-based indices\n"
77  "Supplying a list of these tuples to BlocksToExclude will result in\n"
78  "the domain having the corresponding Blocks excluded. See the Domain\n"
79  "Creation tutorial in the documentation for more information on Block\n"
80  "numberings in rectilinear domains. Note that if any Blocks are\n"
81  "excluded, setting the option IsPeriodicIn to `true` in any dimension\n"
82  "will trigger an error, as periodic boundary\n"
83  "conditions for this domain with holes is not supported."};
84 
85  AlignedLattice(typename BlockBounds::type block_bounds,
86  typename IsPeriodicIn::type is_periodic_in,
87  typename InitialRefinement::type initial_refinement_levels,
88  typename InitialGridPoints::type initial_number_of_grid_points,
89  typename BlocksToExclude::type blocks_to_exclude) noexcept;
90 
91  AlignedLattice() = default;
92  AlignedLattice(const AlignedLattice&) = delete;
93  AlignedLattice(AlignedLattice&&) noexcept = default;
94  AlignedLattice& operator=(const AlignedLattice&) = delete;
95  AlignedLattice& operator=(AlignedLattice&&) noexcept = default;
96  ~AlignedLattice() override = default;
97 
98  Domain<VolumeDim, TargetFrame> create_domain() const noexcept override;
99 
100  std::vector<std::array<size_t, VolumeDim>> initial_extents() const
101  noexcept override;
102 
103  std::vector<std::array<size_t, VolumeDim>> initial_refinement_levels() const
104  noexcept override;
105 
106  private:
107  typename BlockBounds::type block_bounds_{
108  make_array<VolumeDim, std::vector<double>>({})};
109  typename IsPeriodicIn::type is_periodic_in_{make_array<VolumeDim>(false)};
110  typename InitialRefinement::type initial_refinement_levels_{
111  make_array<VolumeDim>(std::numeric_limits<size_t>::max())};
112  typename InitialGridPoints::type initial_number_of_grid_points_{
113  make_array<VolumeDim>(std::numeric_limits<size_t>::max())};
114  typename BlocksToExclude::type blocks_to_exclude_{};
115  Index<VolumeDim> number_of_blocks_by_dim_{};
116 };
117 } // namespace creators
118 } // namespace domain

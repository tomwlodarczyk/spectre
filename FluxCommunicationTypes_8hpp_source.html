44  typename Metavariables::normal_dot_numerical_flux::type::package_tags>;

56  /// The type of the data needed for the local part of the flux
57  /// numerical flux computations. Contains the PackagedData, the
58  /// normal fluxes, and MagnitudeOfFaceNormal.
59  struct LocalData {
60  /// Data on the mortar mesh

67  p | mortar_data;

95  /// The inbox tag for flux communication.

104  boost::hash<std::pair<Direction<volume_dim>,
105  ElementId<volume_dim>>>>>;
106  };
107 };

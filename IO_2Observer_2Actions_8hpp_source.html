29 /// \brief Register a class that will call
30 /// `observers::ThreadedActions::ContributeVolumeData`.
31 ///
32 /// Should be invoked on ObserverWriter.
33 struct RegisterVolumeContributorWithObserverWriter {
34  public:
35  template <typename ParallelComponent, typename DbTagsList,
36  typename Metavariables, typename ArrayIndex,
37  typename... ReductionDatums,
38  Requires<tmpl::list_contains_v<
39  DbTagsList, Tags::VolumeObserversRegistered>> = nullptr>
40  static void apply(db::DataBox<DbTagsList>& box,
41  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
42  const ArrayIndex& /*array_index*/,
43  const observers::ObservationId& observation_id,
44  const size_t processing_element) noexcept {
45  db::mutate<Tags::VolumeObserversRegistered>(
46  make_not_null(&box),
47  [&observation_id, &processing_element ](
48  const gsl::not_null<std::unordered_map<size_t, std::set<size_t>>*>
49  volume_observers_registered) noexcept {
50  // Currently the only part of the observation_id that is used is the
51  // `observation_type_hash()`. But in the future with load balancing
52  // we will use the full observation_id, and elements will need
53  // to register and unregister themselves at specific times.
54  const size_t hash = observation_id.observation_type_hash();
55 
56  if (volume_observers_registered->count(hash) == 0) {
57  (*volume_observers_registered)[hash] = std::set<size_t>{};
58  }
59  // We don't care if we insert the same processing element
60  // more than once. We care only about which processing
61  // elements have registered.
62  volume_observers_registered->at(hash).insert(processing_element);
63  });
64  }
65 };

67 /// \brief Register a class that will call
68 /// `observers::ThreadedActions::WriteReductionData` or
69 /// `observers::ThreadedActions::ContributeReductionData`.
70 ///
71 /// Should be invoked on ObserverWriter.
72 struct RegisterReductionContributorWithObserverWriter {
73  public:
74  template <
75  typename ParallelComponent, typename DbTagsList, typename Metavariables,
76  typename ArrayIndex, typename... ReductionDatums,
77  Requires<tmpl::list_contains_v<DbTagsList,
78  Tags::ReductionObserversRegistered> and
79  tmpl::list_contains_v<DbTagsList,
80  Tags::ReductionObserversRegisteredNodes>> =
81  nullptr>
82  static void apply(db::DataBox<DbTagsList>& box,
83  Parallel::ConstGlobalCache<Metavariables>& cache,
84  const ArrayIndex& /*array_index*/,
85  const observers::ObservationId& observation_id,
86  const size_t processing_element_or_node,
87  const bool called_from_other_node = false) noexcept {
88  const auto node_id = static_cast<size_t>(Parallel::my_node());
89  if (not called_from_other_node) {
90  // processing_element_or_node is the processing element of the caller.
91  db::mutate<Tags::ReductionObserversRegistered>(
92  make_not_null(&box),
93  [&cache, &node_id, &observation_id, &processing_element_or_node ](
94  const gsl::not_null<std::unordered_map<size_t, std::set<size_t>>*>
95  reduction_observers_registered) noexcept {
96  // Currently the only part of the observation_id that is used is the
97  // `observation_type_hash()`. But in the future with load balancing
98  // we will use the full observation_id, and elements will need
99  // to register and unregister themselves at specific times.
100  const size_t hash = observation_id.observation_type_hash();
101 
102  if (reduction_observers_registered->count(hash) == 0) {
103  (*reduction_observers_registered)[hash] = std::set<size_t>{};
104 
105  // If this is not node 0, call this Action on node 0 to register
106  // the nodes that we expect reductions to be called on.
107  if (node_id != 0) {
108  Parallel::simple_action<
109  Actions::RegisterReductionContributorWithObserverWriter>(
110  Parallel::get_parallel_component<
111  ObserverWriter<Metavariables>>(cache)[0],
112  observation_id, node_id, true);
113  }
114  }
115 
116  // We don't care if we insert the same processing element
117  // more than once. We care only about which processing
118  // elements have registered.
119  reduction_observers_registered->at(hash).insert(
120  processing_element_or_node);
121  });
122  } else if (called_from_other_node) {
123  // processing_element_or_node is the node_id of the caller.
124 
125  ASSERT(node_id == 0, "Only node zero, not node "
126  << node_id
127  << ", should be called from another node");
128 
129  db::mutate<Tags::ReductionObserversRegisteredNodes>(
130  make_not_null(&box),
131  [&processing_element_or_node, &observation_id ](
132  const gsl::not_null<std::unordered_map<size_t, std::set<size_t>>*>
133  reduction_observers_registered_nodes) noexcept {
134  const size_t hash = observation_id.observation_type_hash();
135 
136  if (reduction_observers_registered_nodes->count(hash) == 0) {
137  (*reduction_observers_registered_nodes)[hash] =
138  std::set<size_t>{};
139  }
140  reduction_observers_registered_nodes->at(hash).insert(
141  processing_element_or_node);
142  });
143  }
144  }
145 };

147 /*!
148  * \brief Action that is called on the Observer parallel component to register
149  * the parallel component that will send the data to the observer
150  */
151 struct RegisterSenderWithSelf {
152  template <
153  typename ParallelComponent, typename DbTagList, typename Metavariables,
154  typename ArrayIndex,
155  Requires<tmpl::list_contains_v<
156  DbTagList, observers::Tags::ReductionArrayComponentIds> and
157  tmpl::list_contains_v<
158  DbTagList, observers::Tags::VolumeArrayComponentIds>> =
159  nullptr>
160  static void apply(db::DataBox<DbTagList>& box,
161  Parallel::ConstGlobalCache<Metavariables>& cache,
162  const ArrayIndex& /*array_index*/,
163  const observers::ObservationId& observation_id,
164  const observers::ArrayComponentId& component_id,
165  const TypeOfObservation& type_of_observation) noexcept {
166  const auto register_reduction =
167  [&box, &cache, &component_id, &observation_id ]() noexcept {
168  db::mutate<observers::Tags::ReductionArrayComponentIds>(
169  make_not_null(&box), [&component_id](
170  const auto array_component_ids) noexcept {
171  ASSERT(array_component_ids->count(component_id) == 0,
172  "Trying to insert a component_id more than once for "
173  "reduction. This means an element is registering itself "
174  "with the observers more than once.");
175  array_component_ids->insert(component_id);
176  });
177  const auto my_proc = static_cast<size_t>(Parallel::my_proc());
178  auto& observer_writer =
179  *Parallel::get_parallel_component<
180  observers::ObserverWriter<Metavariables>>(cache)
181  .ckLocalBranch();
182  Parallel::simple_action<
183  Actions::RegisterReductionContributorWithObserverWriter>(
184  observer_writer, observation_id, my_proc);
185  };
186  const auto register_volume =
187  [&box, &cache, &component_id, &observation_id ]() noexcept {
188  db::mutate<observers::Tags::VolumeArrayComponentIds>(
189  make_not_null(&box), [&component_id](
190  const auto array_component_ids) noexcept {
191  ASSERT(array_component_ids->count(component_id) == 0,
192  "Trying to insert a component_id more than once for "
193  "volume observation. This means an element is registering "
194  "itself with the observers more than once.");
195  array_component_ids->insert(component_id);
196  });
197  const auto my_proc = static_cast<size_t>(Parallel::my_proc());
198  auto& observer_writer =
199  *Parallel::get_parallel_component<
200  observers::ObserverWriter<Metavariables>>(cache)
201  .ckLocalBranch();
202  Parallel::simple_action<
203  Actions::RegisterVolumeContributorWithObserverWriter>(
204  observer_writer, observation_id, my_proc);
205  };
206 
207  switch (type_of_observation) {
208  case TypeOfObservation::ReductionAndVolume:
209  register_reduction();
210  register_volume();
211  return;
212  case TypeOfObservation::Reduction:
213  register_reduction();
214  return;
215  case TypeOfObservation::Volume:
216  register_volume();
217  return;
218  default:
219  ERROR(
220  "Registering an unknown TypeOfObservation. Should be one of "
221  "'Reduction', 'Volume', or 'ReductionAndVolume'");
222  };
223  }
224 };

226 /*!
227  * \brief Registers itself with the local observer parallel component so the
228  * observer knows to expect data from this component, and also whether to expect
229  * Reduction, Volume, or both Reduction and Volume data observation calls.
230  */
231 template <typename RegisterHelper>
232 struct RegisterWithObservers {
233  template <typename DbTagList, typename... InboxTags, typename Metavariables,
234  typename ArrayIndex, typename ActionList,
235  typename ParallelComponent>
236  static std::tuple<db::DataBox<DbTagList>&&> apply(
237  db::DataBox<DbTagList>& box,
238  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
239  Parallel::ConstGlobalCache<Metavariables>& cache,
240  const ArrayIndex& array_index, const ActionList /*meta*/,
241  const ParallelComponent* const /*meta*/) noexcept {
242  auto& observer =
243  *Parallel::get_parallel_component<observers::Observer<Metavariables>>(
244  cache)
245  .ckLocalBranch();
246  std::pair<observers::TypeOfObservation, observers::ObservationId>
247  type_of_observation_and_observation_id_pair =
248  RegisterHelper::template register_info<ParallelComponent>(
249  box, array_index);
250 
251  // TODO(): We should really loop through all the events, check if there are
252  // any reduction observers, and any volume observers, if so, then we
253  // register us as having one of:
254  // - TypeOfObservation::Reduction
255  // - TypeOfObservation::Volume
256  // - TypeOfObservation::ReductionAndVolume
257  //
258  // At that point we won't need the template parameter on the class anymore
259  // and everything can be read from an input file.
260  Parallel::simple_action<RegisterSenderWithSelf>(
261  observer,
262  std::move( // NOLINT
263  type_of_observation_and_observation_id_pair.second), // NOLINT
264  observers::ArrayComponentId(
265  std::add_pointer_t<ParallelComponent>{nullptr},
266  Parallel::ArrayIndex<std::decay_t<ArrayIndex>>{array_index}),
267  std::move( // NOLINT
268  type_of_observation_and_observation_id_pair.first)); // NOLINT
269  return {std::move(box)};
270  }
271 };

273 /*!
274  * \brief Registers a singleton with the ObserverWriter.
275  *
276  * The singleton that observes is expected to call WriteReductionData
277  * on node 0. An example of how to do this is
278  *```
279  * template <typename DbTags, typename Metavariables>
280  * static void apply(
281  * const db::DataBox<DbTags>& box,
282  * Parallel::ConstGlobalCache<Metavariables>& cache,
283  * const typename Metavariables::temporal_id::type& temporal_id) noexcept {
284  * auto& proxy = Parallel::get_parallel_component<
285  * observers::ObserverWriter<Metavariables>>(cache);
286  *
287  * // We call this on proxy[0] because the 0th element of a NodeGroup is
288  * // always guaranteed to be present.
289  * Parallel::threaded_action<observers::ThreadedActions::WriteReductionData>(
290  * proxy[0],
291  * observers::ObservationId(temporal_id.time(), ObservationType{}),
292  * std::string{"/" + pretty_type::short_name<InterpolationTargetTag>()},
293  * detail::make_legend(TagsToObserve{}),
294  * detail::make_reduction_data(box, temporal_id.time().value(),
295  * TagsToObserve{}));
296  * }
297  *
298  *```
299  *
300  */
301 template <typename RegisterHelper>
302 struct RegisterSingletonWithObserverWriter {
303  template <typename DbTagList, typename... InboxTags, typename Metavariables,
304  typename ArrayIndex, typename ActionList,
305  typename ParallelComponent>
306  static std::tuple<db::DataBox<DbTagList>&&, bool> apply(
307  db::DataBox<DbTagList>& box,
308  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
309  Parallel::ConstGlobalCache<Metavariables>& cache,
310  const ArrayIndex& array_index, const ActionList /*meta*/,
311  const ParallelComponent* const /*meta*/) noexcept {
312  std::pair<observers::TypeOfObservation, observers::ObservationId>
313  type_of_observation_and_observation_id_pair =
314  RegisterHelper::template register_info<ParallelComponent>(
315  box, array_index);
316 
317  switch (type_of_observation_and_observation_id_pair.first) {
318  case TypeOfObservation::Reduction:
319  break;
320  case TypeOfObservation::ReductionAndVolume:
321  case TypeOfObservation::Volume:
322  ERROR(
323  "Registering volume observations is not supported for singletons. "
324  "The TypeOfObservation should be 'Reduction'.");
325  default:
326  ERROR(
327  "Registering an unknown TypeOfObservation. It should be "
328  "'Reduction' for singleton.");
329  };
330 
331  // The actual value of the processing element doesn't matter
332  // here; it is used only to give an ObserverWriter a count of how many
333  // times it will be called.
334  constexpr size_t fake_processing_element = 0;
335 
336  // We call only on node 0; the observation call will occur only
337  // on node 0.
338  Parallel::simple_action<
339  Actions::RegisterReductionContributorWithObserverWriter>(
340  Parallel::get_parallel_component<
341  observers::ObserverWriter<Metavariables>>(cache)[0],
342  std::move( // NOLINT
343  type_of_observation_and_observation_id_pair.second), // NOLINT
344  fake_processing_element);
345  return {std::move(box), true};
346  }
347 };

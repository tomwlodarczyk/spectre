26 /// \brief A propagating shock between two constant states
27 ///
28 /// The shock propagates left-to-right, with profile
29 /// \f$U(x, t) = U_R + (U_L - U_R) H(-(x - x_0 - v t))\f$.
30 /// Here \f$U_L\f$ and \f$U_R\f$ are the left and right constant states,
31 /// satisfying \f$U_L > U_R\f$; \f$H\f$ is the Heaviside function; \f$x_0\f$ is
32 /// the initial (i.e., \f$t=0\f$) position of the shock; and
33 /// \f$v = 0.5 (U_L + U_R)\f$ is the speed of the shock.
34 ///
35 /// \note At the shock, where \f$x = x_0 + vt\f$, we have \f$U(x, t) = U_L\f$.
36 /// (This is inherited from the Heaviside implementation `step_function`.)
37 /// Additionally, the time derivative \f$\partial_t u0\f$ is zero, rather than
38 /// the correct delta function.
39 class Step : public MarkAsAnalyticSolution {
40  public:
41  struct LeftValue {
42  using type = double;
43  static type lower_bound() noexcept { return 0.0; }
44  static type default_value() noexcept { return 2.0; }
45  static constexpr OptionString help{"The value of U, left of the shock"};
46  };
47  struct RightValue {
48  using type = double;
49  static type lower_bound() noexcept { return 0.0; }
50  static type default_value() noexcept { return 1.0; }
51  static constexpr OptionString help{"The value of U, right of the shock"};
52  };
53  struct InitialPosition {
54  using type = double;
55  static constexpr OptionString help{"The shock's position at t==0"};
56  };
57 
58  using options = tmpl::list<LeftValue, RightValue, InitialPosition>;
59  static constexpr OptionString help{"A propagating shock solution"};

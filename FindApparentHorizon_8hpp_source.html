49 /// \brief post interpolation callback (see InterpolationTarget)
50 /// that does a FastFlow iteration and triggers another one until convergence.
51 ///
52 /// Assumes that InterpolationTargetTag contains an additional
53 /// struct called `post_horizon_find_callback`, which has a function
54 ///```
55 /// static void apply(const DataBox<DbTags>&,
56 /// const intrp::ConstGlobalCache<Metavariables>&,
57 /// const Metavariables::temporal_id&) noexcept;
58 ///```
59 /// that is called if the FastFlow iteration has converged.
60 ///
61 /// Uses:
62 /// - Metavariables:
63 /// - `temporal_id`
64 /// - DataBox:
65 /// - `::Tags::Verbosity`
66 /// - `::gr::Tags::InverseSpatialMetric<3,Frame>`
67 /// - `::gr::Tags::ExtrinsicCurvature<3,Frame>`
68 /// - `::gr::Tags::SpatialChristoffelSecondKind<3,Frame>`
69 /// - `::ah::Tags::FastFlow`
70 /// - `StrahlkorperTags::Strahlkorper<Frame>`
71 ///
72 /// Modifies:
73 /// - DataBox:
74 /// - `::ah::Tags::FastFlow`
75 /// - `StrahlkorperTags::Strahlkorper<Frame>`
76 ///
77 /// This is an InterpolationTargetTag::post_interpolation_callback;
78 /// see InterpolationTarget for a description of InterpolationTargetTag.
79 template <typename InterpolationTargetTag>
80 struct FindApparentHorizon {
81  using observation_types = typename InterpolationTargetTag::
82  post_horizon_find_callback::observation_types;
83  template <typename DbTags, typename Metavariables>
84  static bool apply(
85  const gsl::not_null<db::DataBox<DbTags>*> box,
86  const gsl::not_null<Parallel::ConstGlobalCache<Metavariables>*> cache,
87  const typename Metavariables::temporal_id::type& temporal_id) noexcept {
88 
89  // Before doing anything else, deal with the possibility that some
90  // of the points might be outside of the Domain.
91  const auto num_invalid_pts =
92  db::get<Tags::IndicesOfInvalidInterpPoints>(*box).size();
93  if(num_invalid_pts > 0) {
94  ERROR("FindApparentHorizon: Found points that are not in any block");
95  }
96 
97  const auto& verbosity = db::get<::Tags::Verbosity>(*box);
98  const auto& inv_g =
99  db::get<::gr::Tags::InverseSpatialMetric<3, Frame::Inertial>>(*box);
100  const auto& ex_curv =
101  db::get<::gr::Tags::ExtrinsicCurvature<3, Frame::Inertial>>(*box);
102  const auto& christoffel =
103  db::get<::gr::Tags::SpatialChristoffelSecondKind<3, Frame::Inertial>>(
104  *box);
105 
106  std::pair<FastFlow::Status, FastFlow::IterInfo> status_and_info;
107 
108  // Do a FastFlow iteration.
109  db::mutate<::ah::Tags::FastFlow,
110  StrahlkorperTags::Strahlkorper<Frame::Inertial>>(
111  box, [&inv_g, &ex_curv, &christoffel, &status_and_info ](
112  const gsl::not_null<::FastFlow*> fast_flow,
113  const gsl::not_null<::Strahlkorper<Frame::Inertial>*>
114  strahlkorper) noexcept {
115  status_and_info =
116  fast_flow->template iterate_horizon_finder<Frame::Inertial>(
117  strahlkorper, inv_g, ex_curv, christoffel);
118  });
119 
120  // Determine whether we have converged, whether we need another step,
121  // or whether we have encountered an error.
122 
123  const auto& status = status_and_info.first;
124  const auto& info = status_and_info.second;
125  const auto has_converged = converged(status);
126 
127  if (verbosity > ::Verbosity::Quiet or
128  (verbosity > ::Verbosity::Silent and has_converged)) {
129  // The things printed out here are:
130  // its = current iteration number.
131  // R = min and max of residual over all prolonged grid points.
132  // |R| = L2 norm of residual, counting only L modes solved for.
133  // |R_mesh| = L2 norm of residual over prolonged grid points.
134  // r = min and max radius of trial horizon surface.
135  //
136  // Difference between |R| and |R_mesh|:
137  // The horizon is represented in a Y_lm expansion up to l=l_surface;
138  // the residual |R| represents the failure of that surface to satisfy
139  // the apparent horizon equation.
140  //
141  // However, at each iteration we also interpolate the horizon surface
142  // to a higher resolution ("prolongation"). The prolonged surface
143  // includes Ylm coefficents up to l=l_mesh, where l_mesh > l_surface.
144  // The residual computed on this higher-resolution surface is |R_mesh|.
145  //
146  // As iterations proceed, |R| should decrease until it reaches numerical
147  // roundoff error, because we are varying all the Y_lm coefficients up to
148  // l=l_surface to minimize the residual. However, |R_mesh| should
149  // eventually stop decreasing as iterations proceed, hitting a
150  // floor that represents the numerical truncation error of the solution.
151  //
152  // The convergence criterion looks at both |R| and |R_mesh|: Once
153  // |R| is small enough and |R_mesh| has stabilized, it is pointless
154  // to keep iterating (even though one could iterate until |R| reaches
155  // roundoff).
156  //
157  // Problems with convergence of the apparent horizon finder can often
158  // be diagnosed by looking at the behavior of |R| and |R_mesh| as a
159  // function of iteration.
160  Parallel::printf(
161  "%s: its=%d: %.1e<R<%.0e, |R|=%.1g, "
162  "|R_grid|=%.1g, %.4g<r<%.4g\n",
163  pretty_type::short_name<InterpolationTargetTag>(), info.iteration,
164  info.min_residual, info.max_residual, info.residual_ylm,
165  info.residual_mesh, info.r_min, info.r_max);
166  }
167 
168  if (status == FastFlow::Status::SuccessfulIteration) {
169  // Do another iteration of the same horizon search.
170  const auto& temporal_ids =
171  db::get<intrp::Tags::TemporalIds<Metavariables>>(*box);
172  auto& interpolation_target = Parallel::get_parallel_component<
173  intrp::InterpolationTarget<Metavariables, InterpolationTargetTag>>(
174  *cache);
175  Parallel::simple_action<
176  typename InterpolationTargetTag::compute_target_points>(
177  interpolation_target, temporal_ids.front());
178  // We return false because we don't want this iteration to clean
179  // up the volume data, since we are using it for the next iteration
180  // (i.e. the simple_action that we just called).
181  return false;
182  } else if (not has_converged) {
183  ERROR("Apparent horizon finder "
184  << pretty_type::short_name<InterpolationTargetTag>()
185  << " failed, reason = " << status);
186  }
187  // If we get here, the horizon finder has converged.
188 
189  // The interpolated variables
190  // ::Tags::Variables<InterpolationTargetTag::vars_to_interpolate_to_target>
191  // have been interpolated from the volume to the points on the
192  // prolonged_strahlkorper, not to the points on the actual
193  // strahlkorper. So here we do a restriction of these quantities onto
194  // the actual strahlkorper.
195 
196  // Type alias to make code more understandable.
197  using vars_tags =
198  typename InterpolationTargetTag::vars_to_interpolate_to_target;
199  db::mutate_apply<tmpl::list<::Tags::Variables<vars_tags>>,
200  tmpl::list<StrahlkorperTags::Strahlkorper<Frame::Inertial>,
201  ::ah::Tags::FastFlow>>(
202  [
203  ](const gsl::not_null<db::item_type<::Tags::Variables<vars_tags>>*>
204  vars,
205  const db::const_item_type<
206  StrahlkorperTags::Strahlkorper<Frame::Inertial>>& strahlkorper,
207  const db::const_item_type<::ah::Tags::FastFlow>& fast_flow) noexcept {
208  const size_t L_mesh = fast_flow.current_l_mesh(strahlkorper);
209  const auto prolonged_strahlkorper =
210  Strahlkorper<Frame::Inertial>(L_mesh, L_mesh, strahlkorper);
211  auto new_vars = db::item_type<::Tags::Variables<vars_tags>>(
212  strahlkorper.ylm_spherepack().physical_size());
213 
214  tmpl::for_each<vars_tags>([
215  &strahlkorper, &prolonged_strahlkorper, &vars, &new_vars
216  ](auto tag_v) noexcept {
217  using tag = typename decltype(tag_v)::type;
218  const auto& old_var = get<tag>(*vars);
219  auto& new_var = get<tag>(new_vars);
220  auto old_iter = old_var.begin();
221  auto new_iter = new_var.begin();
222  for (; old_iter != old_var.end() and new_iter != new_var.end();
223  ++old_iter, ++new_iter) {
224  *new_iter = strahlkorper.ylm_spherepack().spec_to_phys(
225  prolonged_strahlkorper.ylm_spherepack().prolong_or_restrict(
226  prolonged_strahlkorper.ylm_spherepack().phys_to_spec(
227  *old_iter),
228  strahlkorper.ylm_spherepack()));
229  }
230  });
231  *vars = std::move(new_vars);
232  },
233  box);
234 
235  InterpolationTargetTag::post_horizon_find_callback::apply(*box, *cache,
236  temporal_id);
237 
238  // Prepare for finding horizon at a new time.
239  // For now, the initial guess for the new
240  // horizon is the result of the old one, so we don't need
241  // to modify the strahlkorper.
242  // Eventually we will hold more than one previous guess and
243  // will do time-extrapolation to set the next guess.
244  db::mutate<::ah::Tags::FastFlow>(
245  box, [](const gsl::not_null<::FastFlow*> fast_flow) noexcept {
246  fast_flow->reset_for_next_find();
247  });
248  // We return true because we are now done with all the volume data
249  // at this temporal_id, so we want it cleaned up.
250  return true;
251  }
252 };
253 } // namespace callbacks
254 } // namespace intrp

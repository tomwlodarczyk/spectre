120 namespace Actions {
121 /// \ingroup ActionsGroup
122 /// \brief Sends points on a trial apparent horizon to an `Interpolator`.
123 ///
124 /// This differs from `KerrHorizon` in the following ways:
125 /// - It supplies points on a prolonged Strahlkorper, at a higher resolution
126 /// than the Strahlkorper in the DataBox, as needed for horizon finding.
127 /// - It uses a `FastFlow` in the DataBox.
128 /// - It has different options (including those for `FastFlow`).
129 ///
130 /// Uses:
131 /// - DataBox:
132 /// - `::Tags::Domain<3, Frame>`
133 /// - `::ah::Tags::FastFlow`
134 /// - `StrahlkorperTags::CartesianCoords<Frame>`
135 /// - `::Tags::Variables<typename
136 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
137 ///
138 /// DataBox changes:
139 /// - Adds: nothing
140 /// - Removes: nothing
141 /// - Modifies:
142 /// - `Tags::IndicesOfFilledInterpPoints`
143 /// - `Tags::IndicesOfInvalidInterpPoints`
144 /// - `::Tags::Variables<typename
145 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
146 ///
147 /// This Action also has an initialize function that adds to the DataBox:
148 /// - `StrahlkorperTags::items_tags<Frame>`
149 /// - `StrahlkorperTags::compute_items_tags<Frame>`
150 /// - `::ah::Tags::FastFlow`
151 /// - `::Tags::Verbosity`
152 ///
153 /// For requirements on InterpolationTargetTag, see InterpolationTarget
154 template <typename InterpolationTargetTag, typename Frame>
155 struct ApparentHorizon {
156  using const_global_cache_tags =
157  tmpl::list<Tags::ApparentHorizon<InterpolationTargetTag, Frame>>;
158  using initialization_tags =
159  tmpl::append<StrahlkorperTags::items_tags<Frame>,
160  tmpl::list<::ah::Tags::FastFlow, ::Tags::Verbosity>,
161  StrahlkorperTags::compute_items_tags<Frame>>;
162  template <typename DbTags, typename Metavariables>
163  static auto initialize(
164  db::DataBox<DbTags>&& box,
165  const Parallel::ConstGlobalCache<Metavariables>& cache) noexcept {
166  const auto& options =
167  Parallel::get<Tags::ApparentHorizon<InterpolationTargetTag, Frame>>(
168  cache);
169 
170  // Put Strahlkorper and its ComputeItems, FastFlow,
171  // and verbosity into a new DataBox.
172  return db::create_from<
173  db::RemoveTags<>,
174  db::AddSimpleTags<
175  tmpl::push_back<StrahlkorperTags::items_tags<Frame>,
176  ::ah::Tags::FastFlow, ::Tags::Verbosity>>,
177  db::AddComputeTags<StrahlkorperTags::compute_items_tags<Frame>>>(
178  std::move(box), options.initial_guess, options.fast_flow,
179  options.verbosity);
180  }
181  template <typename ParallelComponent, typename DbTags, typename Metavariables,
182  typename ArrayIndex,
183  Requires<tmpl::list_contains_v<
184  DbTags, Tags::TemporalIds<Metavariables>>> = nullptr>
185  static void apply(
186  db::DataBox<DbTags>& box,
187  Parallel::ConstGlobalCache<Metavariables>& cache,
188  const ArrayIndex& /*array_index*/,
189  const typename Metavariables::temporal_id::type& temporal_id) noexcept {
190  const auto& fast_flow = db::get<::ah::Tags::FastFlow>(box);
191  const auto& strahlkorper =
192  db::get<StrahlkorperTags::Strahlkorper<Frame>>(box);
193 
194  const size_t L_mesh = fast_flow.current_l_mesh(strahlkorper);
195  const auto prolonged_strahlkorper =
196  Strahlkorper<Frame>(L_mesh, L_mesh, strahlkorper);
197 
198  const auto prolonged_coords =
199  StrahlkorperTags::CartesianCoords<Frame>::function(
200  prolonged_strahlkorper,
201  StrahlkorperTags::Radius<Frame>::function(prolonged_strahlkorper),
202  StrahlkorperTags::Rhat<Frame>::function(
203  StrahlkorperTags::ThetaPhi<Frame>::function(
204  prolonged_strahlkorper)));
205 
206  // In the future, when we add support for multiple Frames,
207  // the code that transforms coordinates from the Strahlkorper Frame
208  // to Frame::Inertial will go here. That transformation
209  // may depend on `temporal_id`.
210 
211  send_points_to_interpolator<InterpolationTargetTag>(
212  box, cache, prolonged_coords, temporal_id);
213  }
214 };

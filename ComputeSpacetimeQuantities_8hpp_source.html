35 /*!
36  * \ingroup GeneralRelativityGroup
37  * \brief Computes the spacetime metric from the spatial metric, lapse, and
38  * shift.
39  * \details The spacetime metric \f$ \psi_{ab} \f$ is calculated as
40  * \f{align}{
41  * \psi_{tt} &= - N^2 + N^m N^n g_{mn} \\
42  * \psi_{ti} &= g_{mi} N^m \\
43  * \psi_{ij} &= g_{ij}
44  * \f}
45  * where \f$ N, N^i\f$ and \f$ g_{ij}\f$ are the lapse, shift and spatial metric
46  * respectively
47  */
48 template <size_t Dim, typename Frame, typename DataType>
49 void spacetime_metric(
50  gsl::not_null<tnsr::aa<DataType, Dim, Frame>*> spacetime_metric,
51  const Scalar<DataType>& lapse, const tnsr::I<DataType, Dim, Frame>& shift,
52  const tnsr::ii<DataType, Dim, Frame>& spatial_metric) noexcept;
53 
54 template <size_t SpatialDim, typename Frame, typename DataType>
55 tnsr::aa<DataType, SpatialDim, Frame> spacetime_metric(
56  const Scalar<DataType>& lapse,
57  const tnsr::I<DataType, SpatialDim, Frame>& shift,

70 /*!
71  * \ingroup GeneralRelativityGroup
72  * \brief Compute inverse spacetime metric from inverse spatial metric, lapse
73  * and shift
74  *
75  * \details The inverse spacetime metric \f$ \psi^{ab} \f$ is calculated as
76  * \f{align}
77  * \psi^{tt} &= - 1/N^2 \\
78  * \psi^{ti} &= N^i / N^2 \\
79  * \psi^{ij} &= g^{ij} - N^i N^j / N^2
80  * \f}
81  * where \f$ N, N^i\f$ and \f$ g^{ij}\f$ are the lapse, shift and inverse
82  * spatial metric respectively
83  */

88  const tnsr::II<DataType, SpatialDim, Frame>&
89  inverse_spatial_metric) noexcept;

91 /*!
92  * \ingroup GeneralRelativityGroup
93  * \brief Compute shift from spacetime metric and inverse spatial metric.
94  *
95  * \details Computes
96  * \f{align}
97  * N^i &= g^{ij} \psi_{jt}

126 /*!
127  * \ingroup GeneralRelativityGroup
128  * \brief Computes spacetime derivative of spacetime metric from spatial metric,
129  * lapse, shift, and their space and time derivatives.
130  *
131  * \details Computes the derivatives as:
132  * \f{align}
133  * \partial_\mu \psi_{tt} &= - 2 N \partial_\mu N
134  * + 2 g_{mn} N^m \partial_\mu N^n
135  * + N^m N^n \partial_\mu g_{mn} \\
136  * \partial_\mu \psi_{ti} &= g_{mi} \partial_\mu N^m
137  * + N^m \partial_\mu g_{mi} \\
138  * \partial_\mu \psi_{ij} &= \partial_\mu g_{ij}
139  * \f}
140  * where \f$ N, N^i, g \f$ are the lapse, shift, and spatial metric
141  * respectively.
142  */

180 /*!
181  * \ingroup GeneralRelativityGroup
182  * \brief Computes extrinsic curvature from metric and derivatives.
183  * \details Uses the ADM evolution equation for the spatial metric,
184  * \f[ K_{ij} = \frac{1}{2N} \left ( -\partial_0 g_{ij}
185  * + N^k \partial_k g_{ij} + g_{ki} \partial_j N^k
186  * + g_{kj} \partial_i N^k \right ) \f]
187  * where \f$K_{ij}\f$ is the extrinsic curvature, \f$N\f$ is the lapse,
188  * \f$N^i\f$ is the shift, and \f$g_{ij}\f$ is the spatial metric. In terms
189  * of the Lie derivative of the spatial metric with respect to a unit timelike
190  * vector \f$t^a\f$ normal to the spatial slice, this corresponds to the sign
191  * convention
192  * \f[ K_{ab} = - \frac{1}{2} \mathcal{L}_{\mathbf{t}} g_{ab} \f]
193  */
194 template <size_t SpatialDim, typename Frame, typename DataType>
195 tnsr::ii<DataType, SpatialDim, Frame> extrinsic_curvature(
196  const Scalar<DataType>& lapse,
197  const tnsr::I<DataType, SpatialDim, Frame>& shift,
198  const tnsr::iJ<DataType, SpatialDim, Frame>& deriv_shift,
199  const tnsr::ii<DataType, SpatialDim, Frame>& spatial_metric,
200  const tnsr::ii<DataType, SpatialDim, Frame>& dt_spatial_metric,
201  const tnsr::ijj<DataType, SpatialDim, Frame>&
202  deriv_spatial_metric) noexcept;

345 /*!
346  * \brief Compute item to get spatial derivative of spacetime metric from
347  * spatial metric, lapse, shift, and their space and time derivatives.
348  *
349  * \details Extracts spatial derivatives from spacetime derivatives computed
350  * with `derivatives_of_spacetime_metric()`. Can be retrieved using
351  * `gr::Tags::SpacetimeMetric` wrapped in `Tags::deriv`.
352  */
353 template <size_t SpatialDim, typename Frame>
354 struct DerivSpacetimeMetricCompute
355  : gr::Tags::DerivSpacetimeMetric<SpatialDim, Frame, DataVector>,
356  db::ComputeTag {
357  using argument_tags = tmpl::list<
358  gr::Tags::DerivativesOfSpacetimeMetric<SpatialDim, Frame, DataVector>>;
359  static constexpr auto function(
360  const tnsr::abb<DataVector, SpatialDim, Frame>&
361  spacetime_deriv_of_spacetime_metric) noexcept {
362  auto deriv_spacetime_metric =
363  make_with_value<tnsr::iaa<DataVector, SpatialDim, Frame>>(
364  spacetime_deriv_of_spacetime_metric, 0.);
365  for (size_t i = 0; i < SpatialDim; ++i) {
366  for (size_t a = 0; a < SpatialDim + 1; ++a) {
367  for (size_t b = a; b < SpatialDim + 1; ++b) {
368  deriv_spacetime_metric.get(i, a, b) =
369  spacetime_deriv_of_spacetime_metric.get(i + 1, a, b);
370  }
371  }
372  }
373  return deriv_spacetime_metric;
374  }
375  using base = gr::Tags::DerivSpacetimeMetric<SpatialDim, Frame, DataVector>;
376 };

149 /*!
150  * \ingroup CoordinateMapsGroup
151  * \brief A coordinate map or composition of coordinate maps
152  *
153  * Maps coordinates from the `SourceFrame` to the `TargetFrame` using the
154  * coordinate maps `Maps...`. The individual maps are applied left to right
155  * from the source to the target Frame. The inverse map, as well as Jacobian
156  * and inverse Jacobian are also provided. The `CoordinateMap` class must
157  * be used even if just wrapping a single coordinate map. It is designed to
158  * be an extremely minimal interface to the underlying coordinate maps. For
159  * a list of all coordinate maps see the CoordinateMaps group or namespace.
160  *
161  * Each coordinate map must contain a `static constexpr size_t dim` variable
162  * that is equal to the dimensionality of the map. The Coordinatemap class
163  * contains a member `static constexpr size_t dim`, a type alias `source_frame`,
164  * a type alias `target_frame` and `typelist of the `Maps...`.
165  */
166 template <typename SourceFrame, typename TargetFrame, typename... Maps>
167 class CoordinateMap
168  : public CoordinateMapBase<SourceFrame, TargetFrame,
169  CoordinateMaps::map_dim<Maps...>> {
170  static_assert(sizeof...(Maps) > 0, "Must have at least one map");
171  static_assert(
172  tmpl::all<tmpl::integral_list<size_t, Maps::dim...>,
173  std::is_same<tmpl::integral_constant<
174  size_t, CoordinateMaps::map_dim<Maps...>>,
175  tmpl::_1>>::value,
176  "All Maps passed to CoordinateMap must be of the same dimensionality.");
177 
178  public:
179  static constexpr size_t dim = CoordinateMaps::map_dim<Maps...>;
180  using source_frame = SourceFrame;

460 template <typename SourceFrame, typename TargetFrame, typename... Maps>
461 template <typename T>
462 constexpr SPECTRE_ALWAYS_INLINE auto
463 CoordinateMap<SourceFrame, TargetFrame, Maps...>::inv_jacobian_impl(
464  tnsr::I<T, dim, SourceFrame>&& source_point, const double time,
465  const std::unordered_map<std::string, FunctionOfTime&>& functions_of_time)
466  const noexcept -> InverseJacobian<T, dim, SourceFrame, TargetFrame> {
467  std::array<T, dim> mapped_point = make_array<T, dim>(std::move(source_point));
468 
469  InverseJacobian<T, dim, SourceFrame, TargetFrame> inv_jac{};
470 
471  tuple_transform(
472  maps_,
473  [&inv_jac, &mapped_point, time, &functions_of_time](
474  const auto& map, auto index, const std::tuple<Maps...>& maps) {
475  constexpr const size_t count = decltype(index)::value;
476 
477  tnsr::Ij<T, dim, Frame::NoFrame> temp_inv_jac{};
478 
479  // chooses the correct call based on time-dependence of jacobian
480  auto inv_jac_overload = make_overloader(
481  [](const gsl::not_null<tnsr::Ij<T, dim, Frame::NoFrame>*> t_inv_jac,
482  const auto& the_map, const std::array<T, dim>& point,
483  const double /*t*/,
484  const std::unordered_map<std::string, FunctionOfTime&>&
485  /*funcs_of_time*/,
486  const std::false_type /*is_time_independent*/) {
487  if (LIKELY(not the_map.is_identity())) {
488  *t_inv_jac = the_map.inv_jacobian(point);
489  } else {
490  *t_inv_jac = identity<dim>(point[0]);
491  }
492  return nullptr;
493  },
494  [](const gsl::not_null<tnsr::Ij<T, dim, Frame::NoFrame>*> t_inv_jac,
495  const auto& the_map, const std::array<T, dim>& point,
496  const double t,
497  const std::unordered_map<std::string, FunctionOfTime&>&
498  funcs_of_time,
499  const std::true_type /*is_time_dependent*/) {
500  *t_inv_jac = the_map.inv_jacobian(point, t, funcs_of_time);
501  return nullptr;
502  });
503 
504  if (LIKELY(count != 0)) {
505  const auto& map_in_loop =
506  std::get<(count != 0 ? count - 1 : 0)>(maps);
507  if (LIKELY(not map_in_loop.is_identity())) {
508  CoordinateMap_detail::apply_map(
509  make_not_null(&mapped_point), map_in_loop, time,
510  functions_of_time,
511  CoordinateMap_detail::is_map_time_dependent_t<decltype(
512  map_in_loop)>{});
513  inv_jac_overload(&temp_inv_jac, map, mapped_point, time,
514  functions_of_time,
515  CoordinateMap_detail::is_jacobian_time_dependent_t<
516  decltype(map), T>{});
517  std::array<T, dim> temp{};
518  for (size_t source = 0; source < dim; ++source) {
519  for (size_t target = 0; target < dim; ++target) {
520  gsl::at(temp, target) =
521  inv_jac.get(source, 0) * temp_inv_jac.get(0, target);
522  for (size_t dummy = 1; dummy < dim; ++dummy) {
523  gsl::at(temp, target) += inv_jac.get(source, dummy) *
524  temp_inv_jac.get(dummy, target);
525  }
526  }
527  for (size_t target = 0; target < dim; ++target) {
528  inv_jac.get(source, target) = std::move(gsl::at(temp, target));
529  }
530  }
531  }
532  } else {
533  inv_jac_overload(
534  &temp_inv_jac, map, mapped_point, time, functions_of_time,
535  CoordinateMap_detail::is_jacobian_time_dependent_t<decltype(map),
536  T>{});
537  for (size_t source = 0; source < dim; ++source) {
538  for (size_t target = 0; target < dim; ++target) {
539  inv_jac.get(source, target) =
540  std::move(temp_inv_jac.get(source, target));

549 template <typename SourceFrame, typename TargetFrame, typename... Maps>
550 template <typename T>
551 constexpr SPECTRE_ALWAYS_INLINE auto
552 CoordinateMap<SourceFrame, TargetFrame, Maps...>::jacobian_impl(
553  tnsr::I<T, dim, SourceFrame>&& source_point, const double time,
554  const std::unordered_map<std::string, FunctionOfTime&>& functions_of_time)
555  const noexcept -> Jacobian<T, dim, SourceFrame, TargetFrame> {
556  std::array<T, dim> mapped_point = make_array<T, dim>(std::move(source_point));
557  Jacobian<T, dim, SourceFrame, TargetFrame> jac{};
558 
559  tuple_transform(
560  maps_,
561  [&jac, &mapped_point, time, &functions_of_time](
562  const auto& map, auto index, const std::tuple<Maps...>& maps) {
563  constexpr const size_t count = decltype(index)::value;
564 
565  tnsr::Ij<T, dim, Frame::NoFrame> noframe_jac{};
566 
567  // chooses the correct call based on time-dependence of jacobian
568  auto jac_overload = make_overloader(
569  [](const gsl::not_null<tnsr::Ij<T, dim, Frame::NoFrame>*>
570  no_frame_jac,
571  const auto& the_map, const std::array<T, dim>& point,
572  const double /*t*/,
573  const std::unordered_map<std::string, FunctionOfTime&>&
574  /*funcs_of_time*/,
575  const std::false_type /*is_time_independent*/) {
576  if (LIKELY(not the_map.is_identity())) {
577  *no_frame_jac = the_map.jacobian(point);
578  } else {
579  *no_frame_jac = identity<dim>(point[0]);
580  }
581  return nullptr;
582  },
583  [](const gsl::not_null<tnsr::Ij<T, dim, Frame::NoFrame>*>
584  no_frame_jac,
585  const auto& the_map, const std::array<T, dim>& point,
586  const double t,
587  const std::unordered_map<std::string, FunctionOfTime&>&
588  funcs_of_time,
589  const std::true_type /*is_time_dependent*/) {
590  *no_frame_jac = the_map.jacobian(point, t, funcs_of_time);
591  return nullptr;
592  });
593 
594  if (LIKELY(count != 0)) {
595  const auto& map_in_loop =
596  std::get<(count != 0 ? count - 1 : 0)>(maps);
597  if (LIKELY(not map_in_loop.is_identity())) {
598  CoordinateMap_detail::apply_map(
599  make_not_null(&mapped_point), map_in_loop, time,
600  functions_of_time,
601  CoordinateMap_detail::is_map_time_dependent_t<decltype(
602  map_in_loop)>{});
603  jac_overload(&noframe_jac, map, mapped_point, time,
604  functions_of_time,
605  CoordinateMap_detail::is_jacobian_time_dependent_t<
606  decltype(map), T>{});
607  std::array<T, dim> temp{};
608  for (size_t source = 0; source < dim; ++source) {
609  for (size_t target = 0; target < dim; ++target) {
610  gsl::at(temp, target) =
611  noframe_jac.get(target, 0) * jac.get(0, source);
612  for (size_t dummy = 1; dummy < dim; ++dummy) {
613  gsl::at(temp, target) +=
614  noframe_jac.get(target, dummy) * jac.get(dummy, source);
615  }
616  }
617  for (size_t target = 0; target < dim; ++target) {
618  jac.get(target, source) = std::move(gsl::at(temp, target));
619  }
620  }
621  }
622  } else {
623  jac_overload(
624  &noframe_jac, map, mapped_point, time, functions_of_time,
625  CoordinateMap_detail::is_jacobian_time_dependent_t<decltype(map),
626  T>{});
627  for (size_t target = 0; target < dim; ++target) {
628  for (size_t source = 0; source < dim; ++source) {
629  jac.get(target, source) =
630  std::move(noframe_jac.get(target, source));
631  }
632  }
633  }
634  },
635  maps_);
636  return jac;
637 }

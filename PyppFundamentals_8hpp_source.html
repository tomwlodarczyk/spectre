62 /// Create a python tuple from Args
63 ///
64 /// \tparam Args the types of the arguments to be put in the tuple (deducible)
65 /// \param t the arguments to put into the tuple
66 /// \return PyObject* containing a Python tuple
67 template <typename... Args>
68 PyObject* make_py_tuple(const Args&... t) {
69  PyObject* py_tuple = PyTuple_New(sizeof...(Args));
70  int entry = 0;
71  const auto add_entry = [&entry, &py_tuple](const auto& arg) {
72  PyObject* value = to_py_object(arg);
73  PyTuple_SetItem(py_tuple, entry, value);
74  entry++;
75  return '0';
76  };
77  (void)add_entry; // GCC warns that add_entry is unused
78  (void)std::initializer_list<char>{add_entry(t)...};
79  return py_tuple;
80 }

485 template <typename T>
486 T tensor_conversion_impl(PyObject* p) {
487  if (p == nullptr) {
488  throw std::runtime_error{"Received null PyObject."};
489  }
490  // clang-tidy: c-style casts. (Expanded from macro)
491  if (not PyArray_CheckExact(p)) { // NOLINT
492  throw std::runtime_error{"Cannot convert non-array type to Tensor."};
493  }
494  // clang-tidy: reinterpret_cast
495  const auto npy_array = reinterpret_cast<PyArrayObject*>(p); // NOLINT
496  if (PyArray_TYPE(npy_array) != NPY_DOUBLE) {
497  throw std::runtime_error{
498  "Cannot convert array of non-double type to Tensor."};
499  } else if (PyArray_NDIM(npy_array) != static_cast<long>(T::rank())) {
500  throw std::runtime_error{
501  "Mismatch between ndim of numpy ndarray and rank of Tensor."};
502  }
503 
504  const auto npy_array_dims = PyArray_DIMS(npy_array);
505  constexpr auto t_array_dims = T::index_dims();
506  for (size_t i = 0; i < T::rank(); ++i) {
507  if (npy_array_dims[i] != static_cast<long>(gsl::at(t_array_dims, i))) {
508  throw std::runtime_error{
509  "Mismatch between number of components of ndarray and Tensor in " +
510  std::to_string(i) + "\'th dim"};
511  }
512  }
513  auto t = make_with_value<T>(
514  *get_ptr_to_elem<typename T::type>(npy_array, make_array<T::rank()>(0ul)),
515  0.);
516  for (IndexIterator<T::rank()> index_it((Index<T::rank()>(T::index_dims())));
517  index_it; ++index_it) {
518  const auto tensor_idx = (*index_it).indices();
519  t.get(tensor_idx) =
520  *get_ptr_to_elem<typename T::type>(npy_array, tensor_idx);
521  }
522  return t;
523 }

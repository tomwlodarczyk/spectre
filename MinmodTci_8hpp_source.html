41 // Implements the troubled-cell indicator corresponding to the Minmod limiter.
42 //
43 // The troubled-cell indicator (TCI) determines whether or not limiting is
44 // needed. See Limiters::Minmod for a full description of the Minmod
45 // limiter. Note that as an optimization, this TCI returns (by reference) some
46 // additional data that are used by the Minmod limiter in the case where the
47 // TCI returns true (i.e., the case where limiting is needed).
48 template <size_t VolumeDim>
49 bool troubled_cell_indicator(
50  gsl::not_null<double*> u_mean,
51  gsl::not_null<std::array<double, VolumeDim>*> u_limited_slopes,
52  gsl::not_null<DataVector*> u_lin_buffer,
53  gsl::not_null<std::array<DataVector, VolumeDim>*> boundary_buffer,
54  Limiters::MinmodType minmod_type, double tvbm_constant, const DataVector& u,
55  const Element<VolumeDim>& element, const Mesh<VolumeDim>& mesh,
56  const std::array<double, VolumeDim>& element_size,
57  const DirectionMap<VolumeDim, double>& effective_neighbor_means,
58  const DirectionMap<VolumeDim, double>& effective_neighbor_sizes,
59  const std::array<std::pair<gsl::span<std::pair<size_t, size_t>>,
60  gsl::span<std::pair<size_t, size_t>>>,
61  VolumeDim>& volume_and_slice_indices) noexcept;
62 
63 // Implements the minmod troubled-cell indicator on several tensors.
64 //
65 // Internally loops over all tensors and tensor components, calling the above
66 // function for each case. Returns true if any of the components needs limiting.
67 // Optimization: stops as soon as one component needs limiting, because no
68 // intermediate results are passed "up" out of this function.
69 //
70 // Expects type `PackagedData` to contain, as in Limiters::Minmod:
71 // - a variable `means` that is a `TaggedTuple<Tags::Mean<Tags>...>`
72 // - a variable `element_size` that is a `std::array<double, VolumeDim>`
73 template <size_t VolumeDim, typename PackagedData, typename... Tags>
74 bool troubled_cell_indicator(
75  const db::const_item_type<Tags>&... tensors,
76  const std::unordered_map<
77  std::pair<Direction<VolumeDim>, ElementId<VolumeDim>>, PackagedData,
78  boost::hash<std::pair<Direction<VolumeDim>, ElementId<VolumeDim>>>>&
79  neighbor_data,
80  const Limiters::MinmodType minmod_type, const double tvbm_constant,
81  const Element<VolumeDim>& element, const Mesh<VolumeDim>& mesh,
82  const std::array<double, VolumeDim>& element_size) noexcept {
83  // Optimization: allocate temporary buffer to be used in TCI
84  std::unique_ptr<double[], decltype(&free)> contiguous_buffer(nullptr, &free);
85  DataVector u_lin_buffer{};
86  std::array<DataVector, VolumeDim> boundary_buffer{};
87  Minmod_detail::allocate_buffers(make_not_null(&contiguous_buffer),
88  make_not_null(&u_lin_buffer),
89  make_not_null(&boundary_buffer), mesh);
90 
91  // Optimization: precompute the slice indices since this is (surprisingly)
92  // expensive
93  const auto volume_and_slice_buffer_and_indices =
94  volume_and_slice_indices(mesh.extents());
95  const auto& volume_and_slice_indices =
96  volume_and_slice_buffer_and_indices.second;
97 
98  const auto effective_neighbor_sizes =
99  compute_effective_neighbor_sizes(element, neighbor_data);
100 
101  // Ideally, as soon as one component is found that needs limiting, then we
102  // would exit the TCI early with return value `true`. But there is no natural
103  // way to return early from the pack expansion. So we use this bool to keep
104  // track of whether a previous component needed limiting... and if so, then
105  // we simply skip any work.
106  bool some_component_needs_limiting = false;
107  const auto wrap_tci_one_tensor = [&](auto tag, const auto tensor) noexcept {
108  if (some_component_needs_limiting) {
109  // Skip this tensor completely
110  return '0';
111  }
112 
113  for (size_t tensor_storage_index = 0; tensor_storage_index < tensor.size();
114  ++tensor_storage_index) {
115  const auto effective_neighbor_means =
116  compute_effective_neighbor_means<decltype(tag)>(
117  element, tensor_storage_index, neighbor_data);
118 
119  const DataVector& u = tensor[tensor_storage_index];
120  double u_mean;
121  std::array<double, VolumeDim> u_limited_slopes{};
122  const bool reduce_slope = troubled_cell_indicator(
123  make_not_null(&u_mean), make_not_null(&u_limited_slopes),
124  make_not_null(&u_lin_buffer), make_not_null(&boundary_buffer),
125  minmod_type, tvbm_constant, u, element, mesh, element_size,
126  effective_neighbor_means, effective_neighbor_sizes,
127  volume_and_slice_indices);
128 
129  if (reduce_slope) {
130  some_component_needs_limiting = true;
131  // Skip remaining components of this tensor
132  return '0';
133  }
134  }
135  return '0';
136  };
137  expand_pack(wrap_tci_one_tensor(Tags{}, tensors)...);
138  return some_component_needs_limiting;
139 }

20 /*!
21  * \ingroup DiscontinuousGalerkinGroup
22  * \ingroup NumericalFluxesGroup
23  * \brief Compute the HLL numerical flux.
24  *
25  * Let \f$U\f$ and \f$F^i(U)\f$ be the state vector of the system and its
26  * corresponding volume flux, respectively. Let \f$n_i\f$ be the unit normal to
27  * the interface. Defining \f$F_n(U) := n_i F^i(U)\f$, and denoting the
28  * corresponding projections of the numerical fluxes as \f${F_n}^*(U)\f$, the
29  * HLL flux \ref hll_ref "[1]" for each variable is
30  *
31  * \f{align*}
32  * {F_n}^*(U) = \frac{c_\text{max} F_n(U_\text{int}) -
33  * c_\text{min} F_n(U_\text{ext})}{c_\text{max} - c_\text{min}}
34  * - \frac{c_\text{min}c_\text{max}}{c_\text{max} - c_\text{min}}
35  * \left(U_\text{int} - U_\text{ext}\right)
36  * \f}
37  *
38  * where "int" and "ext" stand for interior and exterior, respectively.
39  * \f$c_\text{min}\f$ and \f$c_\text{max}\f$ are estimates on the minimum
40  * and maximum signal velocities bounding the interior-moving and
41  * exterior-moving wavespeeds that arise when solving the Riemann problem.
42  * Here we use the simple estimates \ref estimates_ref "[2]"
43  *
44  * \f{align*}
45  * c_\text{min} &= \text{min}\left( \lambda_1(U_\text{int}; n_\text{int}),
46  * \lambda_1(U_\text{ext}; n_\text{ext}), 0\right),\\
47  * c_\text{max} &= \text{max}\left( \lambda_N(U_\text{int}; n_\text{int}),
48  * \lambda_N(U_\text{ext}; n_\text{ext}), 0\right),
49  * \f}
50  *
51  * where \f$\lambda_1(U; n) \leq \lambda_2(U; n) \leq ... \leq
52  * \lambda_N(U; n)\f$ are the (ordered) characteristic speeds of the system.
53  * Note that the definitions above ensure that \f$c_\text{min} \leq 0\f$ and
54  * \f$c_\text{max} \geq 0\f$. Also, for either \f$c_\text{min} = 0\f$ or
55  * \f$c_\text{max} = 0\f$ (i.e. all characteristics move in the same direction)
56  * the HLL flux reduces to pure upwinding.
57  *
58  * \anchor hll_ref [1] A. Harten, P. D. Lax, B. van Leer, On Upstream
59  * Differencing and Godunov-Type Schemes for Hyperbolic Conservation Laws,
60  * SIAM Rev. [25 (1983) 35](https://doi.org/10.1137/1025002)
61  *
62  * \anchor estimates_ref [2] S. F. Davis, Simplified Second-Order Godunov-Type
63  * Methods, SIAM J. Sci. Stat. Comput.
64  * [9 (1988) 445](https://doi.org/10.1137/0909030)
65  */

102  template <typename... VariablesTags, typename... NormalDotFluxTags>
103  struct package_data_helper<tmpl::list<VariablesTags...>,
104  tmpl::list<NormalDotFluxTags...>> {
105  static void function(
106  const gsl::not_null<Variables<package_tags>*> packaged_data,
107  const db::const_item_type<NormalDotFluxTags>&... n_dot_f_to_package,
108  const db::const_item_type<VariablesTags>&... u_to_package,
109  const db::const_item_type<char_speeds_tag>&
110  characteristic_speeds) noexcept {
111  expand_pack((get<VariablesTags>(*packaged_data) = u_to_package)...);
112  expand_pack(
113  (get<NormalDotFluxTags>(*packaged_data) = n_dot_f_to_package)...);
114 
115  get<MinSignalSpeed>(*packaged_data) = make_with_value<Scalar<DataVector>>(
116  characteristic_speeds[0],
117  std::numeric_limits<double>::signaling_NaN());
118  get<MaxSignalSpeed>(*packaged_data) = make_with_value<Scalar<DataVector>>(
119  characteristic_speeds[0],
120  std::numeric_limits<double>::signaling_NaN());
121 
122  // This finds the min and max characteristic speeds at each grid point,
123  // which are used as estimates of the min and max signal speeds.
124  for (size_t s = 0; s < characteristic_speeds[0].size(); ++s) {
125  // This ensures that local_min_signal_speed <= 0.0
126  const double local_min_signal_speed = (*std::min_element(
127  characteristic_speeds.begin(), characteristic_speeds.end(),
128  [&s](const auto& a, const auto& b) { return a[s] < b[s]; }))[s];
129  get(get<MinSignalSpeed>(*packaged_data))[s] =
130  std::min(local_min_signal_speed, 0.0);
131 
132  // Likewise, local_max_signal_speed >= 0.0
133  const double local_max_signal_speed = (*std::max_element(
134  characteristic_speeds.begin(), characteristic_speeds.end(),
135  [&s](const auto& a, const auto& b) { return a[s] < b[s]; }))[s];
136  get(get<MaxSignalSpeed>(*packaged_data))[s] =
137  std::max(local_max_signal_speed, 0.0);
138  }
139  }
140  };

142  template <typename NormalDotNumericalFluxTagList, typename VariablesTagList,

144  struct call_operator_helper;
145  template <typename... NormalDotNumericalFluxTags, typename... VariablesTags,
146  typename... NormalDotFluxTags>
147  struct call_operator_helper<tmpl::list<NormalDotNumericalFluxTags...>,
148  tmpl::list<VariablesTags...>,
149  tmpl::list<NormalDotFluxTags...>> {
150  static void function(
151  const gsl::not_null<
152  db::item_type<NormalDotNumericalFluxTags>*>... n_dot_numerical_f,
153  const db::const_item_type<NormalDotFluxTags>&... n_dot_f_interior,
154  const db::const_item_type<VariablesTags>&... u_interior,
155  const db::const_item_type<MinSignalSpeed>& min_signal_speed_interior,
156  const db::const_item_type<MaxSignalSpeed>& max_signal_speed_interior,
157  const db::const_item_type<NormalDotFluxTags>&... minus_n_dot_f_exterior,
158  const db::const_item_type<VariablesTags>&... u_exterior,
159  const db::const_item_type<MinSignalSpeed>& min_signal_speed_exterior,
160  const db::const_item_type<MaxSignalSpeed>&
161  max_signal_speed_exterior) noexcept {
162  auto min_signal_speed =
163  make_with_value<db::const_item_type<MinSignalSpeed>>(
164  min_signal_speed_interior,
165  std::numeric_limits<double>::signaling_NaN());
166  auto max_signal_speed =
167  make_with_value<db::const_item_type<MaxSignalSpeed>>(
168  max_signal_speed_interior,
169  std::numeric_limits<double>::signaling_NaN());
170  for (size_t s = 0; s < min_signal_speed.begin()->size(); ++s) {
171  get(min_signal_speed)[s] = std::min(get(min_signal_speed_interior)[s],
172  -get(max_signal_speed_exterior)[s]);
173  get(max_signal_speed)[s] = std::max(get(max_signal_speed_interior)[s],
174  -get(min_signal_speed_exterior)[s]);
175  }
176  const DataVector one_over_cp_minus_cm =
177  1.0 / (get(max_signal_speed) - get(min_signal_speed));
178  const auto assemble_numerical_flux =
179  [&min_signal_speed, &max_signal_speed, &one_over_cp_minus_cm ](
180  const auto n_dot_num_f, const auto& n_dot_f_in, const auto& u_in,
181  const auto& minus_n_dot_f_ex, const auto& u_ex) noexcept {
182  for (size_t i = 0; i < n_dot_num_f->size(); ++i) {
183  (*n_dot_num_f)[i] = one_over_cp_minus_cm *
184  (get(max_signal_speed) * n_dot_f_in[i] +
185  get(min_signal_speed) * minus_n_dot_f_ex[i] -
186  get(max_signal_speed) * get(min_signal_speed) *
187  (u_in[i] - u_ex[i]));
188  }
189  return nullptr;
190  };
191  expand_pack(assemble_numerical_flux(n_dot_numerical_f, n_dot_f_interior,
192  u_interior, minus_n_dot_f_exterior,
193  u_exterior)...);
194  }
195  };

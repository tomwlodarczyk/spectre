36 // Compute the unnormalized nonlinear WENO weights. This is a fairly standard
37 // choice of weights; see e.g., Eq. 3.9 of Zhong2013 or Eq. 3.6 of Zhu2016.
38 inline double unnormalized_nonlinear_weight(
39  const double linear_weight, const double oscillation_indicator) noexcept {
40  return linear_weight / square(1.e-6 + oscillation_indicator);
41 }

43 // Compute the WENO weighted reconstruction of several polynomials, see e.g.,
44 // Eq. 4.3 of Zhu2016. This is fairly standard, though different references can
45 // differ in their choice of oscillation/smoothness indicator.
46 //
47 // The neighbor tensors corresponding to `local_tensor` (i.e., the tensor in
48 // each `neighbor_vars` value that is identified by `Tag`), must have the same
49 // mean as the local tensor. This is checked with an ASSERT.
50 template <typename Tag, size_t VolumeDim, typename TagsList>
51 void reconstruct_from_weighted_sum(
52  const gsl::not_null<db::item_type<Tag>*> local_tensor,
53  const Mesh<VolumeDim>& mesh, const double neighbor_linear_weight,
54  const std::unordered_map<
55  std::pair<Direction<VolumeDim>, ElementId<VolumeDim>>,
56  Variables<TagsList>,
57  boost::hash<std::pair<Direction<VolumeDim>, ElementId<VolumeDim>>>>&
58  neighbor_vars,
59  const DerivativeWeight derivative_weight) noexcept {
60  for (size_t tensor_storage_index = 0;
61  tensor_storage_index < local_tensor->size(); ++tensor_storage_index) {
62  auto& local_polynomial = (*local_tensor)[tensor_storage_index];
63 
64 #ifdef SPECTRE_DEBUG
65  // Check inputs match requirements
66  const double local_mean = mean_value(local_polynomial, mesh);
67  for (const auto& kv : neighbor_vars) {
68  const auto& neighbor_polynomial =
69  get<Tag>(kv.second)[tensor_storage_index];
70  const double neighbor_mean = mean_value(neighbor_polynomial, mesh);
71  ASSERT(equal_within_roundoff(local_mean, neighbor_mean),
72  "Invalid inputs to Weno_detail::reconstruct_from_weighted_sum:\n"
73  "The neighbor polynomials should have the same mean as the local\n"
74  "polynomial.");
75  }
76 #endif // ifdef SPECTRE_DEBUG
77 
78  // Store linear weights in `local_weights` and `neighbor_weights`
79  // These weights will have to be generalized for multiple neighbors per
80  // face for use with h-refinement and AMR.
81  double local_weight = 1.;
82  std::unordered_map<
83  std::pair<Direction<VolumeDim>, ElementId<VolumeDim>>, double,
84  boost::hash<std::pair<Direction<VolumeDim>, ElementId<VolumeDim>>>>
85  neighbor_weights;
86  for (const auto& kv : neighbor_vars) {
87  local_weight -= neighbor_linear_weight;
88  neighbor_weights[kv.first] = neighbor_linear_weight;
89  }
90 
91  // Update `local_weights` and `neighbor_weights` to hold the unnormalized
92  // nonlinear weights.
93  local_weight = unnormalized_nonlinear_weight(
94  local_weight,
95  oscillation_indicator(local_polynomial, mesh, derivative_weight));
96  for (const auto& kv : neighbor_vars) {
97  const auto& key = kv.first;
98  const auto& neighbor_tensor_component =
99  get<Tag>(kv.second)[tensor_storage_index];
100  neighbor_weights[key] = unnormalized_nonlinear_weight(
101  neighbor_weights[key],
102  oscillation_indicator(neighbor_tensor_component, mesh,
103  derivative_weight));
104  }
105 
106  // Update `local_weights` and `neighbor_weights` to hold the normalized
107  // weights; these are the final weights of the WENO reconstruction.
108  double normalization = local_weight;
109  for (const auto& kv : neighbor_weights) {
110  normalization += kv.second;
111  }
112  local_weight /= normalization;
113  for (auto& kv : neighbor_weights) {
114  kv.second /= normalization;
115  }
116 
117  // Perform reconstruction, by superposition of local and neighbor
118  // polynomials.
119  local_polynomial *= local_weight;
120  for (const auto& kv : neighbor_vars) {
121  const auto& key = kv.first;
122  const auto& neighbor_polynomial =
123  get<Tag>(kv.second)[tensor_storage_index];
124  local_polynomial += neighbor_weights.at(key) * neighbor_polynomial;
125  }
126  }
127 }

28 /// \brief Holds a `Variables` interpolated onto a list of points, and
29 /// information about those points, for a local `Interpolator`.
30 ///
31 /// `TagList` is a `tmpl::list` of tags that go into the `Variables`.
32 template <size_t VolumeDim, typename TagList>
33 struct Info {
34  /// `block_coord_holders` holds the list of all points (in block
35  /// logical coordinates) that need to be interpolated onto for a
36  /// given `InterpolationTarget`.
37  ///
38  /// The number of interpolated points for which results are stored
39  /// in this `Info` (in `vars` and `global_offsets` below)
40  /// corresponds to only the subset of the points in
41  /// `block_coord_holders` that are contained in local `Element`s.
42  /// Moreover, the number of interpolated points stored in this
43  /// `Info` will change as more `Elements` send data to this
44  /// `Interpolator`, and will be less than or equal to the size of
45  /// `block_coord_holders` even after all `Element`s have sent their
46  /// data (this is because this `Info` lives only on a single core,
47  /// and this core will have access only to the local `Element`s).
48  std::vector<boost::optional<IdPair<
49  domain::BlockId, tnsr::I<double, VolumeDim, typename ::Frame::Logical>>>>
50  block_coord_holders;
51  /// `vars` holds the interpolated `Variables` on some subset of the
52  /// points in `block_coord_holders`. The grid points inside vars
53  /// are indexed according to `global_offsets` below. The size of
54  /// `vars` changes as more `Element`s send data to this `Interpolator`.
55  std::vector<Variables<TagList>> vars{};
56  /// `global_offsets[j][i]` is the index into `block_coord_holders` that
57  /// corresponds to the index `i` of the `DataVector` held in `vars[j]`.
58  /// The size of `global_offsets` changes as more `Element`s
59  /// send data to this `Interpolator`.
60  std::vector<std::vector<size_t>> global_offsets{};
61  /// Holds the `ElementId`s of `Element`s for which interpolation has
62  /// already been done for this `Info`.
63  std::unordered_set<ElementId<VolumeDim>>
64  interpolation_is_done_for_these_elements{};
65 };
66 
67 template <size_t VolumeDim, typename TagList>
68 void pup(PUP::er& p, Info<VolumeDim, TagList>& t) noexcept { // NOLINT
69  p | t.block_coord_holders;
70  p | t.vars;
71  p | t.global_offsets;
72  p | t.interpolation_is_done_for_these_elements;
73 }

75 template <size_t VolumeDim, typename TagList>
76 void operator|(PUP::er& p, Info<VolumeDim, TagList>& t) noexcept { // NOLINT
77  pup(p, t);
78 }

80 /// Holds `Info`s at all `temporal_id`s for a given
81 /// `InterpolationTargetTag`. Also holds `temporal_id`s when data has
82 /// been interpolated; this is used for cleanup purposes. All
83 /// `Holder`s for all `InterpolationTargetTags` are held in a single
84 /// `TaggedTuple` that is in the `Interpolator`'s `DataBox` with the
85 /// tag `Tags::InterpolatedVarsHolders`.
86 template <typename Metavariables,
87  typename InterpolationTargetTag, typename TagList>
88 struct Holder {
89  std::unordered_map<typename Metavariables::temporal_id::type,
90  Info<Metavariables::volume_dim, TagList>>
91  infos;
92  std::unordered_set<typename Metavariables::temporal_id::type>
93  temporal_ids_when_data_has_been_interpolated;
94 };
95 
96 template <typename Metavariables, typename InterpolationTargetTag,
97  typename TagList>
98 void pup(PUP::er& p, // NOLINT
99  Holder<Metavariables, InterpolationTargetTag, TagList>& // NOLINT
100  t) noexcept { // NOLINT
101  p | t.infos;
102  p | t.temporal_ids_when_data_has_been_interpolated;
103 }
104 
105 template <typename Metavariables, typename InterpolationTargetTag,
106  typename TagList>
107 void operator|(PUP::er& p, // NOLINT
108  Holder<Metavariables, InterpolationTargetTag, TagList>&
109  t) noexcept { // NOLINT
110  pup(p, t);
111 }

113 /// Indexes a particular `Holder` in the `TaggedTuple` that is
114 /// accessed from the `Interpolator`'s `DataBox` with tag

117 struct HolderTag {

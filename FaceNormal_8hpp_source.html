87 /// \ingroup DataBoxTagsGroup
88 /// \ingroup ComputationalDomainGroup
89 /// Specialisation of UnnormalizedFaceNormal for the external boundaries which
90 /// inverts the normals. Since ExternalBoundariesDirections are meant to
91 /// represent ghost elements, the normals should correspond to the normals in
92 /// said element, which are inverted with respect to the current element.
93 template <size_t VolumeDim, typename Frame>
94 struct InterfaceCompute<Tags::BoundaryDirectionsExterior<VolumeDim>,
95  UnnormalizedFaceNormalCompute<VolumeDim, Frame>>
96  : db::PrefixTag,
97  db::ComputeTag,
98  Tags::Interface<Tags::BoundaryDirectionsExterior<VolumeDim>,
99  Tags::UnnormalizedFaceNormal<VolumeDim, Frame>> {
100  using dirs = BoundaryDirectionsExterior<VolumeDim>;
101 
102  static std::string name() noexcept {
103  return "BoundaryDirectionsExterior<UnnormalizedFaceNormal>";
104  }
105 
106  static auto function(
107  const db::const_item_type<Tags::Interface<dirs, Mesh<VolumeDim - 1>>>&
108  meshes,
109  const db::const_item_type<Tags::ElementMap<VolumeDim, Frame>>&
110  map) noexcept {
111  std::unordered_map<::Direction<VolumeDim>,
112  tnsr::i<DataVector, VolumeDim, Frame>>
113  normals{};
114  for (const auto& direction_and_mesh : meshes) {
115  const auto& direction = direction_and_mesh.first;
116  const auto& mesh = direction_and_mesh.second;
117  auto internal_face_normal =
118  unnormalized_face_normal(mesh, map, direction);
119  std::transform(internal_face_normal.begin(), internal_face_normal.end(),
120  internal_face_normal.begin(), std::negate<>());
121  normals[direction] = std::move(internal_face_normal);
122  }
123  return normals;
124  }
125 
126  using argument_tags = tmpl::list<Tags::Interface<dirs, Mesh<VolumeDim - 1>>,
127  Tags::ElementMap<VolumeDim, Frame>>;
128 };

84 /// \ingroup ActionsGroup
85 /// \brief Initializes an InterpolationTarget
86 ///
87 /// Uses: nothing
88 ///
89 /// DataBox changes:
90 /// - Adds:
91 /// - `Tags::IndicesOfFilledInterpPoints`
92 /// - `Tags::IndicesOfInvalidInterpPoints`
93 /// - `Tags::TemporalIds<Metavariables>`
94 /// - `Tags::CompletedTemporalIds<Metavariables>`
95 /// - `::Tags::Variables<typename
96 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
97 /// - Removes: nothing
98 /// - Modifies: nothing
99 ///
100 /// For requirements on InterpolationTargetTag, see InterpolationTarget
101 template <typename Metavariables, typename InterpolationTargetTag>
102 struct InitializeInterpolationTarget {
103  using return_tag_list_initial = tmpl::list<
104  Tags::IndicesOfFilledInterpPoints, Tags::IndicesOfInvalidInterpPoints,
105  Tags::TemporalIds<Metavariables>,
106  Tags::CompletedTemporalIds<Metavariables>,
107  ::Tags::Variables<
108  typename InterpolationTargetTag::vars_to_interpolate_to_target>>;
109  using return_tag_list =
110  tmpl::append<return_tag_list_initial,
111  typename initialize_interpolation_target_detail::
112  initialization_tags<InterpolationTargetTag>::type>;
113 
114  template <typename DbTagsList, typename... InboxTags, typename ArrayIndex,
115  typename ActionList, typename ParallelComponent,
116  Requires<not tmpl::list_contains_v<
117  DbTagsList, Tags::IndicesOfFilledInterpPoints>> = nullptr>
118  static auto apply(db::DataBox<DbTagsList>& box,
119  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
120  const Parallel::ConstGlobalCache<Metavariables>& cache,
121  const ArrayIndex& /*array_index*/,
122  const ActionList /*meta*/,
123  const ParallelComponent* const /*meta*/) noexcept {
124  auto init_box = initialize_interpolation_target_detail::make_initial_box<
125  InterpolationTargetTag>(
126  db::create_from<db::RemoveTags<>,
127  db::get_items<return_tag_list_initial>>(
128  std::move(box), db::item_type<Tags::IndicesOfFilledInterpPoints>{},
129  db::item_type<Tags::IndicesOfInvalidInterpPoints>{},
130  db::item_type<Tags::TemporalIds<Metavariables>>{},
131  db::item_type<Tags::CompletedTemporalIds<Metavariables>>{},
132  db::item_type<
133  ::Tags::Variables<typename InterpolationTargetTag::
134  vars_to_interpolate_to_target>>{}),
135  cache);
136  // compute_items_on_target will depend on compute items added in
137  // make_initial_box, so compute_items_on_target must be added
138  // in a separate step.
139  return std::make_tuple(
140  db::create_from<
141  db::RemoveTags<>, db::AddSimpleTags<>,
142  db::AddComputeTags<
143  typename InterpolationTargetTag::compute_items_on_target>>(
144  std::move(init_box)));
145  }
146 
147  template <typename DbTagsList, typename... InboxTags, typename ArrayIndex,
148  typename ActionList, typename ParallelComponent,
149  Requires<tmpl::list_contains_v<
150  DbTagsList, Tags::IndicesOfFilledInterpPoints>> = nullptr>
151  static std::tuple<db::DataBox<DbTagsList>&&> apply(
152  db::DataBox<DbTagsList>& box,
153  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
154  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
155  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
156  const ParallelComponent* const /*meta*/) noexcept {
157  return {std::move(box)};
158  }
159 };

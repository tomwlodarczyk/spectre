25 /*!
26  * \brief TOV solver based on Lindblom's method
27  *
28  * Uses Lindblom's method of integrating the TOV equations from
29  * \cite Lindblom1998dp
30  *
31  * Instead of integrating \f$m(r)\f$ and \f$p(r)\f$
32  * (\f$r\f$=radius, \f$m\f$=mass, \f$p\f$=pressure)
33  * Lindblom introduces the variables \f$u\f$ and \f$v\f$, with \f$u=r^{2}\f$ and
34  * \f$v=m/r\f$.
35  * The integration is then done with the log of the specific enthalpy
36  * (\f$\mathrm{log}(h)\f$) as the independent variable.
37  *
38  * Lindblom's paper simply labels the independent variable as \f$h\f$.
39  * The \f$h\f$ in Lindblom's paper is NOT the specific enthalpy.
40  * Rather, Lindblom's \f$h\f$ is in fact \f$\mathrm{log}(h)\f$.
41  */
42 class TovSolution {
43  public:
44  TovSolution(
45  const EquationsOfState::EquationOfState<true, 1>& equation_of_state,
46  double central_mass_density, double log_enthalpy_at_outer_radius = 0.0,
47  double absolute_tolerance = 1.0e-14, double relative_tolerance = 1.0e-14);
48 
49  TovSolution() = default;
50  TovSolution(const TovSolution& /*rhs*/) = delete;
51  TovSolution& operator=(const TovSolution& /*rhs*/) = delete;
52  TovSolution(TovSolution&& /*rhs*/) noexcept = default;
53  TovSolution& operator=(TovSolution&& /*rhs*/) noexcept = default;
54  ~TovSolution() = default;
55 
56  /// \brief The outer radius of the solution.
57  ///
58  /// \note This is the radius at which `log_specific_enthalpy` is equal
59  /// to the value of `log_enthalpy_at_outer_radius` that was given when
60  /// constructing this TovSolution
61  double outer_radius() const noexcept;
62 
63  /// \brief The mass inside the given radius \f$m(r)\f$
64  ///
65  /// \note `r` should be non-negative and not greater than outer_radius
66  /// \f$m(R)\f$.
67  double mass(double r) const noexcept;
68 
69  /// \brief The log of the specific enthalpy at the given radius
70  ///
71  /// \note `r` should be non-negative and not greater than outer_radius
72  double log_specific_enthalpy(double r) const noexcept;
73 
74  /// \brief The radial variables from which the hydrodynamic quantities and
75  /// spacetime metric can be computed.
76  ///
77  /// For radii greater than the outer_radius, this returns the appropriate
78  /// vacuum spacetime.
79  ///
80  /// \note This solution of the TOV equations is a function of areal radius.
81  template <typename DataType>

83  radial_variables(
84  const EquationsOfState::EquationOfState<true, 1>& equation_of_state,
85  const tnsr::I<DataType, 3>& x) const noexcept;

88  void pup(PUP::er& p) noexcept;
89 
90  private:
91  double outer_radius_{std::numeric_limits<double>::signaling_NaN()};
92  double total_mass_{std::numeric_limits<double>::signaling_NaN()};
93  double log_lapse_at_outer_radius_{
94  std::numeric_limits<double>::signaling_NaN()};
95  intrp::BarycentricRational mass_interpolant_;
96  intrp::BarycentricRational log_enthalpy_interpolant_;
97 };

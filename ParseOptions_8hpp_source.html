69 template <typename T, typename Metavariables>
70 T Option::parse_as() const {
71  try {
72  // yaml-cpp's `as` method won't parse empty nodes, so we need to
73  // inline a bit of its logic.
74  Options_detail::wrap_create_types<T, Metavariables> result{};
75  if (YAML::convert<decltype(result)>::decode(node(), result)) {
76  return Options_detail::unwrap_create_types(std::move(result));
77  }
78  // clang-tidy: thrown exception is not nothrow copy constructible
79  throw YAML::BadConversion(node().Mark()); // NOLINT
80  } catch (const YAML::BadConversion& e) {
81  // This happens when trying to parse an empty value as a container
82  // with no entries.
83  if ((tt::is_a_v<std::vector, T> or tt::is_std_array_of_size_v<0, T> or
84  tt::is_maplike_v<T>) and node().IsNull()) {
85  return T{};
86  }
87  OptionContext error_context = context();
88  error_context.line = e.mark.line;
89  error_context.column = e.mark.column;
90  std::ostringstream ss;
91  ss << "Failed to convert value to type "
92  << Options_detail::yaml_type<T>::value() << ":";
93 
94  const std::string value_text = YAML::Dump(node());
95  if (value_text.find('\n') == std::string::npos) {
96  ss << " " << value_text;
97  } else {
98  // Indent each line of the value by two spaces and start on a new line
99  ss << "\n ";
100  for (char c : value_text) {
101  ss << c;
102  if (c == '\n') {
103  ss << " ";
104  }
105  }
106  }
107 
108  if (tt::is_a_v<std::vector, T> or tt::is_std_array_v<T>) {
109  ss << "\n\nNote: For sequences this can happen because the length of the "
110  "sequence specified\nin the input file is not equal to the length "
111  "expected by the code. Sequences in\nfiles can be denoted either "
112  "as a bracket enclosed list ([foo, bar]) or with each\nentry on a "
113  "separate line, indented and preceeded by a dash ( - foo).";
114  }
115  PARSE_ERROR(error_context, ss.str());
116  } catch (const Options_detail::propagate_context& e) {
117  OptionContext error_context = context();
118  // Avoid line numbers in the middle of the trace
119  error_context.line = -1;
120  error_context.column = -1;
121  PARSE_ERROR(error_context, e.message());
122  } catch (std::exception& e) {
123  ERROR("Unexpected exception: " << e.what());
124  }
125 }

384 template <typename OptionList, typename Group>
385 void Options<OptionList, Group>::parse(const YAML::Node& node) {
386  if (not(node.IsMap() or node.IsNull())) {
387  PARSE_ERROR(context_, "'" << node << "' does not look like options.\n"
388  << help());
389  }
390  for (const auto& name_and_value : node) {
391  const auto& name = name_and_value.first.as<std::string>();
392  const auto& value = name_and_value.second;
393  auto context = context_;
394  context.line = name_and_value.first.Mark().line;
395  context.column = name_and_value.first.Mark().column;
396 
397  // Check for invalid key
398  if (1 != valid_names_.count(name)) {
399  PARSE_ERROR(context, "Option '" << name << "' is not a valid option.\n"
400  << parsing_help(node));
401  }
402 
403  // Check for duplicate key
404  if (0 != parsed_options_.count(name)) {
405  PARSE_ERROR(context, "Option '" << name << "' specified twice.\n"
406  << parsing_help(node));
407  }
408  parsed_options_.emplace(name, value);
409  }
410 }

412 namespace Options_detail {
413 template <typename Tag, typename Metavariables, typename Subgroup>
414 struct get_impl {
415  template <typename OptionList, typename Group>
416  static typename Tag::type apply(const Options<OptionList, Group>& opts) {
417  static_assert(
418  tmpl::list_contains_v<OptionList, Tag>,
419  "Could not find requested option in the list of options provided. Did "
420  "you forget to add the option tag to the OptionList?");
421  const std::string subgroup_label = option_name<Subgroup>();
422  if (0 == opts.parsed_options_.count(subgroup_label)) {
423  PARSE_ERROR(opts.context_, "You did not specify the group '"
424  << subgroup_label << "'.\n"
425  << opts.help());
426  }
427  Options<options_in_group<OptionList, Subgroup>, Subgroup> subgroup_options(
428  Subgroup::help);
429  subgroup_options.context_ = opts.context_;
430  subgroup_options.context_.append("In group " + subgroup_label);
431  subgroup_options.parse(opts.parsed_options_.find(subgroup_label)->second);
432  return subgroup_options.template get<Tag, Metavariables>();
433  }
434 };

436 template <typename Tag, typename Metavariables>
437 struct get_impl<Tag, Metavariables, Tag> {
438  template <typename OptionList, typename Group>
439  static typename Tag::type apply(const Options<OptionList, Group>& opts) {
440  static_assert(
441  tmpl::list_contains_v<OptionList, Tag>,
442  "Could not find requested option in the list of options provided. Did "
443  "you forget to add the option tag to the OptionList?");
444  const std::string label = option_name<Tag>();
445 
446  opts.template validate_default<Tag>();
447  if (0 == opts.parsed_options_.count(label)) {
448  return opts.template get_default<Tag>();
449  }
450 
451  Option option(opts.parsed_options_.find(label)->second, opts.context_);
452  option.append_context("While parsing option " + label);
453 
454  auto t = option.parse_as<typename Tag::type, Metavariables>();
455 
456  opts.template check_lower_bound_on_size<Tag>(t, option.context());
457  opts.template check_upper_bound_on_size<Tag>(t, option.context());
458  opts.template check_lower_bound<Tag>(t, option.context());
459  opts.template check_upper_bound<Tag>(t, option.context());
460  return t;
461  }
462 };

586 template <typename OptionList, typename Group>
587 [[noreturn]] void Options<OptionList, Group>::parser_error(
588  const YAML::Exception& e) const noexcept {
589  auto context = context_;
590  context.line = e.mark.line;
591  context.column = e.mark.column;
592  // Inline the top_level branch of PARSE_ERROR to avoid warning that
593  // the other branch would call terminate. (Parser errors can only
594  // be generated at top level.)
595  ERROR(
596  "\n"
597  << context
598  << "Unable to correctly parse the input file because of a syntax error.\n"
599  "This is often due to placing a suboption on the same line as an "

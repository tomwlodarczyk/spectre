17 /*!
18  * \ingroup ObserversGroup
19  * \brief The group parallel component that is responsible for reducing data
20  * to be observed.
21  *
22  * Once the data from all elements on the processing element (usually a core)
23  * has been collected, it is copied (not sent over the network) to the local
24  * nodegroup parallel component, `ObserverWriter`, for writing to disk.
25  */
26 template <class Metavariables>
27 struct Observer {
28  using chare_type = Parallel::Algorithms::Group;
29  using metavariables = Metavariables;
30  using phase_dependent_action_list = tmpl::list<Parallel::PhaseActions<
31  typename metavariables::Phase, metavariables::Phase::Initialization,
32  tmpl::list<Actions::Initialize<Metavariables>>>>;
33  using initialization_tags = Parallel::get_initialization_tags<
34  Parallel::get_initialization_actions_list<phase_dependent_action_list>>;
35 
36  static void execute_next_phase(
37  const typename Metavariables::Phase /*next_phase*/,
38  Parallel::CProxy_ConstGlobalCache<
39  Metavariables>& /*global_cache*/) noexcept {}
40 };

56  using initialization_tags = Parallel::get_initialization_tags<

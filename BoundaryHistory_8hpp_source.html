25 /// \ingroup TimeSteppersGroup
26 /// History data used by a TimeStepper for boundary integration.
27 /// \tparam LocalVars local variables passed to the boundary coupling
28 /// \tparam RemoteVars remote variables passed to the boundary coupling
29 /// \tparam CouplingResult result of the coupling function
30 template <typename LocalVars, typename RemoteVars, typename CouplingResult>
31 class BoundaryHistory {
32  template <typename Vars>
33  using IteratorType = boost::transform_iterator<
34  const Time& (*)(const std::tuple<Time, Vars>&),
35  typename std::deque<std::tuple<Time, Vars>>::const_iterator>;
36  public:
37  using local_iterator = IteratorType<LocalVars>;
38  using remote_iterator = IteratorType<RemoteVars>;
39 
40  // No copying because of the pointers in the cache. Moving is fine
41  // because we also move the container being pointed into and maps
42  // guarantee that this doesn't invalidate pointers.
43  BoundaryHistory() = default;
44  BoundaryHistory(const BoundaryHistory&) = delete;
45  BoundaryHistory(BoundaryHistory&&) = default;
46  BoundaryHistory& operator=(const BoundaryHistory&) = delete;
47  BoundaryHistory& operator=(BoundaryHistory&&) = default;
48  ~BoundaryHistory() = default;
49 
50  /// Add a new value to the end of the history of the indicated side.
51  //@{
52  void local_insert(const TimeStepId& time_id, LocalVars vars) noexcept {
53  local_data_.emplace_back(time_id.substep_time(), std::move(vars));
54  }
55  void remote_insert(const TimeStepId& time_id, RemoteVars vars) noexcept {
56  remote_data_.emplace_back(time_id.substep_time(), std::move(vars));
57  }
58  //@}
59 
60  /// Add a new value to the front of the history of the indicated
61  /// side. This is often convenient for setting initial data.
62  //@{
63  void local_insert_initial(const TimeStepId& time_id,
64  LocalVars vars) noexcept {
65  local_data_.emplace_front(time_id.substep_time(), std::move(vars));
66  }
67  void remote_insert_initial(const TimeStepId& time_id,
68  RemoteVars vars) noexcept {
69  remote_data_.emplace_front(time_id.substep_time(), std::move(vars));
70  }
71  //@}
72 
73  /// Mark all data before the passed point in history on the
74  /// indicated side as unneeded so it can be removed. Calling this
75  /// directly should not often be necessary, as it is handled
76  /// internally by the time steppers.
77  //@{
78  void local_mark_unneeded(const local_iterator& first_needed) noexcept {
79  mark_unneeded<0>(make_not_null(&local_data_), first_needed);
80  }
81  void remote_mark_unneeded(const remote_iterator& first_needed) noexcept {
82  mark_unneeded<1>(make_not_null(&remote_data_), first_needed);
83  }
84  //@}
85 
86  /// Access to the sequence of times on the indicated side.
87  //@{
88  local_iterator local_begin() const noexcept {
89  return local_iterator(local_data_.begin(), std::get<0>);
90  }
91  local_iterator local_end() const noexcept {
92  return local_iterator(local_data_.end(), std::get<0>);
93  }
94 
95  remote_iterator remote_begin() const noexcept {
96  return remote_iterator(remote_data_.begin(), std::get<0>);
97  }
98  remote_iterator remote_end() const noexcept {
99  return remote_iterator(remote_data_.end(), std::get<0>);
100  }
101 
102  size_t local_size() const noexcept { return local_data_.size(); }
103  size_t remote_size() const noexcept { return remote_data_.size(); }
104  //@}
105 
106  /// Evaluate the coupling function at the given local and remote
107  /// history entries. The coupling function will be passed the local
108  /// and remote FluxVars and should return a CouplingResult. Values are
109  /// cached internally, so callers should ensure that the coupling
110  /// functions provided on separate calls produce the same result.
111  template <typename Coupling>
112  const CouplingResult& coupling(Coupling&& c, const local_iterator& local,
113  const remote_iterator& remote) const noexcept;
114 
115  // clang-tidy: google-runtime-references
116  void pup(PUP::er& p) noexcept; // NOLINT
117 
118  private:
119  template <size_t Side, typename DataType, typename Iterator>
120  void mark_unneeded(gsl::not_null<DataType*> data,
121  const Iterator& first_needed) noexcept;
122 
123  std::deque<std::tuple<Time, LocalVars>> local_data_;
124  std::deque<std::tuple<Time, RemoteVars>> remote_data_;
125  // We use pointers instead of iterators because deque invalidates
126  // iterators when elements are inserted or removed at the ends, but
127  // not pointers.
128  mutable std::map<std::pair<const Time*, const Time*>, CouplingResult>
129  coupling_cache_;
130 };

132 template <typename LocalVars, typename RemoteVars, typename CouplingResult>
133 template <size_t Side, typename DataType, typename Iterator>
134 void BoundaryHistory<LocalVars, RemoteVars, CouplingResult>::mark_unneeded(
135  gsl::not_null<DataType*> data, const Iterator& first_needed) noexcept {
136  for (auto it = data->begin(); it != first_needed.base(); ++it) {
137  // Clean out cache entries referring to the entry we are removing.
138  for (auto cache_entry = coupling_cache_.begin();
139  cache_entry != coupling_cache_.end();) {
140  if (std::get<Side>(cache_entry->first) == &std::get<0>(*it)) {
141  cache_entry = coupling_cache_.erase(cache_entry);
142  } else {
143  ++cache_entry;
144  }
145  }
146  }
147  data->erase(data->begin(), first_needed.base());
148 }

150 template <typename LocalVars, typename RemoteVars, typename CouplingResult>
151 template <typename Coupling>
152 const CouplingResult&
153 BoundaryHistory<LocalVars, RemoteVars, CouplingResult>::coupling(
154  Coupling&& c, const local_iterator& local,
155  const remote_iterator& remote) const noexcept {
156  const auto insert_result = coupling_cache_.insert(
157  std::make_pair(std::make_pair(&*local, &*remote), CouplingResult{}));
158  CouplingResult& inserted_value = insert_result.first->second;
159  const bool is_new_value = insert_result.second;
160  if (is_new_value) {
161  inserted_value =
162  std::forward<Coupling>(c)(cpp17::as_const(std::get<1>(*local.base())),
163  cpp17::as_const(std::get<1>(*remote.base())));
164  }
165  return inserted_value;
166 }

168 template <typename LocalVars, typename RemoteVars, typename CouplingResult>
169 void BoundaryHistory<LocalVars, RemoteVars, CouplingResult>::pup(
170  PUP::er& p) noexcept {
171  p | local_data_;
172  p | remote_data_;
173 
174  const size_t cache_size = PUP_stl_container_size(p, coupling_cache_);
175  if (p.isUnpacking()) {
176  for (size_t entry_num = 0; entry_num < cache_size; ++entry_num) {
177  size_t local_index, remote_index;
178  CouplingResult cache_value;
179  p | local_index;
180  p | remote_index;
181  p | cache_value;
182  const auto cache_key =
183  std::make_pair(&std::get<0>(local_data_[local_index]),
184  &std::get<0>(remote_data_[remote_index]));
185  coupling_cache_.insert(std::make_pair(cache_key, cache_value));
186  }
187  } else {
188  for (auto& cache_entry : coupling_cache_) {
189  // clang-tidy: modernize-use-auto - Ensuring the correct type is
190  // important here to prevent undefined behavior in charm. I
191  // want to be explicit.
192  size_t local_index = static_cast<size_t>( // NOLINT
193  std::find_if(
194  local_data_.begin(), local_data_.end(),
195  [goal = cache_entry.first.first](const auto& entry) noexcept {
196  return &std::get<0>(entry) == goal;
197  }) -
198  local_data_.begin());
199  ASSERT(local_index < local_data_.size(),
200  "Failed to find local history entry for cache entry");
201 
202  // clang-tidy: modernize-use-auto - Ensuring the correct type is
203  // important here to prevent undefined behavior in charm. I
204  // want to be explicit.
205  size_t remote_index = static_cast<size_t>( // NOLINT
206  std::find_if(
207  remote_data_.begin(), remote_data_.end(),
208  [goal = cache_entry.first.second](const auto& entry) noexcept {
209  return &std::get<0>(entry) == goal;
210  }) -
211  remote_data_.begin());
212  ASSERT(remote_index < remote_data_.size(),
213  "Failed to find remote history entry for cache entry");
214 
215  p | local_index;
216  p | remote_index;
217  p | cache_entry.second;
218  }
219  }
220 }

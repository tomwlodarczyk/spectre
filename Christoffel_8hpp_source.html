29 /*!
30  * \ingroup GeneralRelativityGroup
31  * \brief Computes Christoffel symbol of the first kind from derivative of
32  * metric
33  *
34  * \details Computes Christoffel symbol \f$\Gamma_{abc}\f$ as:
35  * \f$ \Gamma_{cab} = \frac{1}{2} ( \partial_a g_{bc} + \partial_b g_{ac}
36  * - \partial_c g_{ab}) \f$
37  * where \f$g_{bc}\f$ is either a spatial or spacetime metric
38  */
39 template <size_t SpatialDim, typename Frame, IndexType Index, typename DataType>
40 void christoffel_first_kind(
41  gsl::not_null<tnsr::abb<DataType, SpatialDim, Frame, Index>*> christoffel,
42  const tnsr::abb<DataType, SpatialDim, Frame, Index>& d_metric) noexcept;
43 
44 template <size_t SpatialDim, typename Frame, IndexType Index, typename DataType>
45 tnsr::abb<DataType, SpatialDim, Frame, Index> christoffel_first_kind(
46  const tnsr::abb<DataType, SpatialDim, Frame, Index>& d_metric) noexcept;
47 // @}

50 /// Compute item for spatial Christoffel symbols of the first kind
51 /// \f$\Gamma_{ijk}\f$ computed from the first derivative of the
52 /// spatial metric.
53 ///
54 /// Can be retrieved using `gr::Tags::SpatialChristoffelFirstKind`
55 template <size_t SpatialDim, typename Frame, typename DataType>
56 struct SpatialChristoffelFirstKindCompute
57  : SpatialChristoffelFirstKind<SpatialDim, Frame, DataType>,
58  db::ComputeTag {
59  using argument_tags = tmpl::list<
60  ::Tags::deriv<gr::Tags::SpatialMetric<SpatialDim, Frame, DataType>,
61  tmpl::size_t<SpatialDim>, Frame>>;
62  static constexpr tnsr::ijj<DataType, SpatialDim, Frame> (*function)(
63  const tnsr::ijj<DataType, SpatialDim, Frame>&) =
64  &christoffel_first_kind<SpatialDim, Frame, IndexType::Spatial, DataType>;
65  using base = SpatialChristoffelFirstKind<SpatialDim, Frame, DataType>;
66 };

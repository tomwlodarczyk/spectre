232 /// \ingroup DataBoxTagsGroup
233 /// \ingroup ComputationalDomainGroup
234 /// \brief Tag which is either a SimpleTag for quantities on an
235 /// interface, base tag to a compute item which acts on tags on an interface, or
236 /// base tag to a compute item which slices a tag from the volume to an
237 /// interface.
238 ///
239 /// The contained object will be a map from ::Direction to the item type of
240 /// `Tag`, with the set of directions being those produced by `DirectionsTag`.
241 ///
242 /// If a SimpleTag is desired on the interface, then this tag can be added to
243 /// the DataBox directly. If a ComputeTag which acts on Tags on the interface is
244 /// desired, then the tag should be added using `InterfaceCompute`. If a
245 /// ComputeTag which slices a TensorTag or a VariablesTag in the volume to an
246 /// Interface is desired, then it should be added using `Slice`. In all cases,
247 /// the tag can then be retrieved using `Tags::Interface<DirectionsTag, Tag>`.
248 ///
249 /// If using the base tag mechanism for an interface tag is desired,
250 /// then `Tag` can have a `base` type alias pointing to its base
251 /// class. (This requirement is due to the lack of a way to determine
252 /// a type's base classes in C++.)
253 ///
254 /// It must be possible to determine the type associated with `Tag`
255 /// without reference to a DataBox, i.e., `db::const_item_type<Tag>` must
256 /// work.
257 ///
258 /// \tparam DirectionsTag the item of directions
259 /// \tparam Tag the tag labeling the item
260 ///
261 /// \see InterfaceCompute, Slice
262 template <typename DirectionsTag, typename Tag>
263 struct Interface;

316  template <typename Subtag>
317  static void create_item(
318  const gsl::not_null<item_type<tag>*> parent_value,
319  const gsl::not_null<item_type<Subtag>*> sub_value) noexcept {
320  sub_value->clear();
321  for (auto& direction_vars : *parent_value) {
322  const auto& direction = direction_vars.first;
323  auto& parent_vars = get<typename Subtag::tag>(direction_vars.second);
324  auto& sub_var = (*sub_value)[direction];
325  for (auto vars_it = parent_vars.begin(), sub_var_it = sub_var.begin();
326  vars_it != parent_vars.end(); ++vars_it, ++sub_var_it) {
327  sub_var_it->set_data_ref(&*vars_it);
328  }
329  }
330  }
331 
332  // The `return_type` can be anything for Subitems because the DataBox figures
333  // out the correct return type, we just use the `return_type` type alias to
334  // signal to the DataBox we want mutating behavior.
335  using return_type = NoSuchType;
336 
337  template <typename Subtag>
338  static void create_compute_item(
339  const gsl::not_null<item_type<Subtag>*> sub_value,
340  const item_type<tag>& parent_value) noexcept {
341  for (const auto& direction_vars : parent_value) {
342  const auto& direction = direction_vars.first;
343  const auto& parent_vars =
344  get<typename Subtag::tag>(direction_vars.second);
345  auto& sub_var = (*sub_value)[direction];
346  auto sub_var_it = sub_var.begin();
347  for (auto vars_it = parent_vars.begin();
348  vars_it != parent_vars.end(); ++vars_it, ++sub_var_it) {
349  // clang-tidy: do not use const_cast
350  // The DataBox will only give out a const reference to the
351  // result of a compute item. Here, that is a reference to a
352  // const map to Tensors of DataVectors. There is no (publicly
353  // visible) indirection there, so having the map const will
354  // allow only allow const access to the contained DataVectors,
355  // so no modification through the pointer cast here is
356  // possible.
357  sub_var_it->set_data_ref(const_cast<DataVector*>(&*vars_it)); // NOLINT
358  }
359  }
360  }
361 };
362 } // namespace detail

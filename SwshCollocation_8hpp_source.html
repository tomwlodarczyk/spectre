18 /// \ingroup SwshGroup
19 /// \brief Convenience function for determining the number of spin-weighted
20 /// spherical harmonic collocation values that are stored for a given `l_max`
21 /// for a libsharp-compatible set of collocation points.
22 constexpr SPECTRE_ALWAYS_INLINE size_t
23 number_of_swsh_collocation_points(const size_t l_max) noexcept {
24  return (l_max + 1) * (2 * l_max + 1);

71 /// \brief A wrapper class for the spherical harmonic library collocation data
72 ///
73 /// \details The currently chosen library for spin-weighted spherical harmonic
74 /// transforms is libsharp. The `CollocationMetadata` class stores the
75 /// libsharp `sharp_geom_info` object, which contains data about
76 /// 1. The angular collocation points used in spin-weighted spherical harmonic
77 /// transforms
78 /// 2. The memory representation of double-type values at those collocation
79 /// points
80 ///
81 /// \tparam Representation the ComplexRepresentation, either
82 /// `ComplexRepresentation::Interleaved` or
83 /// `ComplexRepresentation::RealsThenImags` compatible with the generated
84 /// `CollocationMetadata` - this is necessary because the stored libsharp type
85 /// contains memory stride information.
86 template <ComplexRepresentation Representation>
87 class CollocationMetadata {
88  public:
89  class CollocationConstIterator {
90  public:
91  /// create a new iterator. defaults to the start of the supplied object
92  explicit CollocationConstIterator(
93  const gsl::not_null<const CollocationMetadata<Representation>*>
94  collocation,
95  const size_t start_index = 0) noexcept
96  : index_{start_index}, collocation_{collocation} {}
97 
98  /// recovers the data at the collocation point using a
99  /// `LibsharpCollocationPoint`, which stores the vector `offset` of the
100  /// location of the `theta`, `phi` point in libsharp compatible data
101  LibsharpCollocationPoint operator*() const noexcept {
102  return LibsharpCollocationPoint{index_, collocation_->theta(index_),
103  collocation_->phi(index_)};
104  }
105  /// advance the iterator by one position (prefix)
106  CollocationConstIterator& operator++() noexcept {
107  ++index_;
108  return *this;
109  };
110  /// advance the iterator by one position (postfix)
111  // clang-tidy wants this to return a const iterator
112  CollocationConstIterator operator++(int) noexcept { // NOLINT
113  return CollocationConstIterator(collocation_, index_++);
114  }
115 
116  /// retreat the iterator by one position (prefix)
117  CollocationConstIterator& operator--() noexcept {
118  --index_;
119  return *this;
120  };
121  /// retreat the iterator by one position (prefix)
122  // clang-tidy wants this to return a const iterator
123  CollocationConstIterator operator--(int) noexcept { // NOLINT
124  return CollocationConstIterator(collocation_, index_--);
125  }
126 
127  // @{
128  /// (In)Equivalence checks both the object and index for the iterator
129  bool operator==(const CollocationConstIterator& rhs) const noexcept {
130  return index_ == rhs.index_ and collocation_ == rhs.collocation_;
131  }
132  bool operator!=(const CollocationConstIterator& rhs) const noexcept {
133  return not(*this == rhs);
134  }
135  // @}
136 
137  private:
138  size_t index_;
139  const CollocationMetadata<Representation>* const collocation_;
140  };
141 
142  /// The representation of the block of complex values, which sets the stride
143  /// inside the libsharp type
144  static constexpr ComplexRepresentation complex_representation =
145  Representation;
146 
147  /// \brief Generates the libsharp collocation information and stores it in
148  /// `geom_info_`.
149  ///
150  /// \note If you will potentially use the same `l_max` collocation set more
151  /// than once, it is probably better to use the
152  /// `precomputed_spherical_harmonic_collocation` function
153  explicit CollocationMetadata(size_t l_max) noexcept;
154 
155  /// default constructor required for iterator use
156  ~CollocationMetadata() = default;
157  CollocationMetadata() = default;
158  CollocationMetadata(const CollocationMetadata&) = default;
159  CollocationMetadata(CollocationMetadata&&) = default;
160  CollocationMetadata& operator=(CollocationMetadata&) = default;
161  CollocationMetadata& operator=(CollocationMetadata&&) = default;
162 
163  /// retrieve the `sharp_geom_info*` stored. This should largely be used only
164  /// for passing to other libsharp functions. Otherwise, access elements
165  /// through iterator or access functions.
166  sharp_geom_info* get_sharp_geom_info() const noexcept {
167  return geom_info_.get();
168  }
169 
170  /// Retrieve the \f$\theta\f$ value for a given index in a libsharp-compatible
171  /// array
172  double theta(size_t offset) const noexcept;
173  /// Retrieve the \f$\phi\f$ value for a given index in a libsharp-compatible
174  /// array
175  double phi(size_t offset) const noexcept;
176 
177  constexpr size_t l_max() const noexcept { return l_max_; }
178 
179  /// Compute the number of entries the libsharp-compatible data structure
180  /// should have
181  constexpr size_t size() const noexcept {
182  return (l_max_ + 1) * (2 * l_max_ + 1);
183  }
184 
185  // @{
186  /// Get a bidirectional iterator to the start of the grid. `operator*` for
187  /// that iterator gives a `LibsharpCollocationPoint` with members `offset`,
188  /// `theta`, and `phi`
189  CollocationMetadata<Representation>::CollocationConstIterator begin() const
190  noexcept {
191  return CollocationConstIterator{make_not_null(this), 0};
192  }
193  CollocationMetadata<Representation>::CollocationConstIterator cbegin() const
194  noexcept {
195  return begin();
196  }
197  // @}
198  // @{
199  /// Get a bidirectional iterator to the end of the grid. `operator*` for
200  /// that iterator gives a `LibsharpCollocationPoint` with members `offset`,
201  /// `theta`, and `phi`
202  CollocationMetadata<Representation>::CollocationConstIterator end() const
203  noexcept {
204  return CollocationConstIterator{make_not_null(this), size()};
205  }
206  CollocationMetadata<Representation>::CollocationConstIterator cend() const
207  noexcept {
208  return end();
209  }
210  // @}
211 
212  private:
213  // an extra pointer layer is required for the peculiar way that libsharp
214  // constructs these values.
215  size_t l_max_ = 0;
216  std::unique_ptr<sharp_geom_info, detail::DestroySharpGeometry> geom_info_;
217 };

224 /// so that the libsharp generation need not be re-run. If it has been

227 template <ComplexRepresentation Representation>
228 const CollocationMetadata<Representation>& cached_collocation_metadata(
229  size_t l_max) noexcept;

24 /*!
25  * \ingroup ActionsGroup
26  * \brief Apply the function `Mutator::apply` to the DataBox
27  *
28  * The function `Mutator::apply` is invoked with the `Mutator::argument_tags`.
29  * The result of this computation is stored in the `Mutator::return_tags`.
30  *
31  * Uses:
32  * - DataBox:
33  * - All elements in `Mutator::argument_tags`
34  *
35  * DataBox changes:
36  * - Modifies:
37  * - All elements in `Mutator::return_tags`
38  */
39 template <typename Mutator>
40 struct MutateApply {
41  template <typename DataBox, typename... InboxTags, typename Metavariables,
42  typename ArrayIndex, typename ActionList,
43  typename ParallelComponent>
44  static std::tuple<DataBox&&> apply(
45  DataBox& box, const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
46  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
47  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
48  const ParallelComponent* const /*meta*/) noexcept {
49  db::mutate_apply<Mutator>(make_not_null(&box));
50  return {std::move(box)};
51  }
52 };
53 } // namespace Actions

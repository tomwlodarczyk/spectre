39 /// \ingroup ActionsGroup
40 /// \ingroup DiscontinuousGalerkinGroup
41 /// \brief Packages data on external boundaries for calculating numerical flux.
42 /// Computes contributions on the interior side from the volume, and imposes
43 /// Dirichlet boundary conditions on the exterior side.
44 ///
45 /// With:
46 /// - Boundary<Tag> =
47 /// Tags::Interface<Tags::BoundaryDirections<volume_dim>, Tag>
48 /// - External<Tag> =
49 /// Tags::Interface<Tags::ExternalBoundaryDirections<volume_dim>, Tag>
50 ///
51 /// Uses:
52 /// - ConstGlobalCache:
53 /// - Metavariables::normal_dot_numerical_flux
54 /// - Metavariables::boundary_condition
55 /// - DataBox:
56 /// - Tags::Element<volume_dim>
57 /// - Boundary<Tags listed in
58 /// Metavariables::normal_dot_numerical_flux::type::argument_tags>
59 /// - External<Tags listed in
60 /// Metavariables::normal_dot_numerical_flux::type::argument_tags>
61 /// - Boundary<Tags::Mesh<volume_dim - 1>>
62 /// - External<Tags::Mesh<volume_dim - 1>>
63 /// - Boundary<Tags::Magnitude<Tags::UnnormalizedFaceNormal<volume_dim>>>,
64 /// - External<Tags::Magnitude<Tags::UnnormalizedFaceNormal<volume_dim>>>,
65 /// - Boundary<Tags::BoundaryCoordinates<volume_dim>>,
66 /// - Metavariables::temporal_id
67 ///
68 /// DataBox changes:
69 /// - Adds: nothing
70 /// - Removes: nothing
71 /// - Modifies:
72 /// - Tags::VariablesBoundaryData
73 /// - External<typename system::variables_tag>
74 ///
75 /// \see ReceiveDataForFluxes
76 template <typename Metavariables>
77 struct ImposeDirichletBoundaryConditions {
78  private:
79  // BoundaryConditionMethod and BcSelector are used to select exactly how to
80  // apply the Dirichlet boundary condition depending on properties of the
81  // system. An overloaded `apply_impl` method is used that implements the
82  // boundary condition calculation for the different types.
83  enum class BoundaryConditionMethod {
84  AnalyticBcNoPrimitives,
85  AnalyticBcFluxConservativeWithPrimitives,
86  Unknown
87  };
88  template <BoundaryConditionMethod Method>
89  using BcSelector = std::integral_constant<BoundaryConditionMethod, Method>;
90 
91  public:
92  using const_global_cache_tags =
93  tmpl::list<typename Metavariables::normal_dot_numerical_flux,
94  typename Metavariables::boundary_condition_tag>;
95 
96  template <typename DbTags, typename... InboxTags, typename ArrayIndex,
97  typename ActionList, typename ParallelComponent>
98  static std::tuple<db::DataBox<DbTags>&&> apply(
99  db::DataBox<DbTags>& box, tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
100  Parallel::ConstGlobalCache<Metavariables>& cache,
101  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
102  const ParallelComponent* const /*meta*/) noexcept {
103  using system = typename Metavariables::system;
104  return apply_impl<Metavariables::system::volume_dim>(
105  box, cache,
106  BcSelector<(not system::has_primitive_and_conservative_vars
107  ? BoundaryConditionMethod::AnalyticBcNoPrimitives
108  : (system::has_primitive_and_conservative_vars and
109  system::is_in_flux_conservative_form
110  ? BoundaryConditionMethod::
111  AnalyticBcFluxConservativeWithPrimitives
112  : BoundaryConditionMethod::Unknown))>{});
113  }
114 
115  private:
116  template <typename DbTags>
117  static void contribute_data_to_mortar(
118  const gsl::not_null<db::DataBox<DbTags>*> box,
119  const Parallel::ConstGlobalCache<Metavariables>& cache) noexcept {
120  using system = typename Metavariables::system;
121  constexpr size_t volume_dim = system::volume_dim;
122 
123  const auto& element = db::get<Tags::Element<volume_dim>>(*box);
124  const auto& temporal_id =
125  db::get<typename Metavariables::temporal_id>(*box);
126  const auto& normal_dot_numerical_flux_computer =
127  get<typename Metavariables::normal_dot_numerical_flux>(cache);
128 
129  auto interior_data = DgActions_detail::compute_local_mortar_data(
130  *box, normal_dot_numerical_flux_computer,
131  Tags::BoundaryDirectionsInterior<volume_dim>{}, Metavariables{});
132 
133  auto exterior_data = DgActions_detail::compute_packaged_data(
134  *box, normal_dot_numerical_flux_computer,
135  Tags::BoundaryDirectionsExterior<volume_dim>{}, Metavariables{});
136 
137  for (const auto& direction : element.external_boundaries()) {
138  const auto mortar_id = std::make_pair(
139  direction, ElementId<volume_dim>::external_boundary_id());
140 
141  db::mutate<Tags::VariablesBoundaryData>(
142  box,
143  [
144  &mortar_id, &temporal_id, &direction, &interior_data, &exterior_data
145  ](const gsl::not_null<
146  db::item_type<Tags::VariablesBoundaryData, DbTags>*>
147  mortar_data) noexcept {
148  mortar_data->at(mortar_id).local_insert(
149  temporal_id, std::move(interior_data.at(direction)));
150  mortar_data->at(mortar_id).remote_insert(
151  temporal_id, std::move(exterior_data.at(direction)));
152  });
153  }
154  }
155 
156  template <size_t VolumeDim, typename DbTags>
157  static std::tuple<db::DataBox<DbTags>&&> apply_impl(
158  db::DataBox<DbTags>& box,
159  const Parallel::ConstGlobalCache<Metavariables>& cache,
160  std::integral_constant<
161  BoundaryConditionMethod,
162  BoundaryConditionMethod::AnalyticBcNoPrimitives> /*meta*/) noexcept {
163  using system = typename Metavariables::system;
164 
165  static_assert(
166  system::is_in_flux_conservative_form or
167  cpp17::is_same_v<typename Metavariables::initial_data_tag,
168  typename Metavariables::boundary_condition_tag>,
169  "Only analytic boundary conditions, or dirichlet boundary conditions "
170  "for conservative systems are implemented");
171 
172  // Apply the boundary condition
173  db::mutate_apply<
174  tmpl::list<Tags::Interface<Tags::BoundaryDirectionsExterior<VolumeDim>,
175  typename system::variables_tag>>,
176  tmpl::list<>>(
177  [](const gsl::not_null<db::item_type<
178  Tags::Interface<Tags::BoundaryDirectionsExterior<VolumeDim>,
179  typename system::variables_tag>>*>
180  external_bdry_vars,
181  const double time, const auto& boundary_condition,
182  const auto& boundary_coords) noexcept {
183  for (auto& external_direction_and_vars : *external_bdry_vars) {
184  auto& direction = external_direction_and_vars.first;
185  auto& vars = external_direction_and_vars.second;
186  vars.assign_subset(boundary_condition.variables(
187  boundary_coords.at(direction), time,
188  typename system::variables_tag::type::tags_list{}));
189  }
190  },
191  make_not_null(&box), db::get<Tags::Time>(box),
192  get<typename Metavariables::boundary_condition_tag>(cache),
193  db::get<Tags::Interface<Tags::BoundaryDirectionsExterior<VolumeDim>,
194  Tags::Coordinates<VolumeDim, Frame::Inertial>>>(
195  box));
196 
197  contribute_data_to_mortar(make_not_null(&box), cache);
198  return std::forward_as_tuple(std::move(box));
199  }
200 
201  template <typename TagsList, typename... ReturnTags, typename... ArgumentTags,
202  typename System>
203  static void apply_impl_helper_conservative_from_primitive(
204  const gsl::not_null<Variables<TagsList>*> conservative_vars,
205  const tuples::TaggedTuple<ArgumentTags...>& argument_vars,
206  tmpl::list<ReturnTags...> /*meta*/, tmpl::list<System> /*meta*/
207  ) noexcept {
208  System::conservative_from_primitive::apply(
209  make_not_null(&get<ReturnTags>(*conservative_vars))...,
210  get<ArgumentTags>(argument_vars)...);
211  }

213  template <size_t VolumeDim, typename DbTags>
214  static std::tuple<db::DataBox<DbTags>&&> apply_impl(
215  db::DataBox<DbTags>& box,
216  const Parallel::ConstGlobalCache<Metavariables>& cache,
217  std::integral_constant<
218  BoundaryConditionMethod,
219  BoundaryConditionMethod::
220  AnalyticBcFluxConservativeWithPrimitives> /*meta*/) noexcept {
221  using system = typename Metavariables::system;
222 
223  static_assert(
224  system::is_in_flux_conservative_form and
225  system::has_primitive_and_conservative_vars and
226  cpp17::is_same_v<typename Metavariables::initial_data_tag,
227  typename Metavariables::boundary_condition_tag>,
228  "Only analytic boundary conditions, or dirichlet boundary conditions "
229  "for conservative systems are implemented");
230 
231  // Apply the boundary condition
232  db::mutate_apply<
233  tmpl::list<Tags::Interface<Tags::BoundaryDirectionsExterior<VolumeDim>,
234  typename system::variables_tag>>,
235  tmpl::list<>>(
236  [](const gsl::not_null<db::item_type<
237  Tags::Interface<Tags::BoundaryDirectionsExterior<VolumeDim>,
238  typename system::variables_tag>>*>
239  external_bdry_vars,
240  const double time, const auto& boundary_condition,
241  const auto& boundary_coords) noexcept {
242  for (auto& external_direction_and_vars : *external_bdry_vars) {
243  auto& direction = external_direction_and_vars.first;
244  auto& vars = external_direction_and_vars.second;
245 
246  apply_impl_helper_conservative_from_primitive(
247  make_not_null(&vars),
248  boundary_condition.variables(
249  boundary_coords.at(direction), time,
250  typename system::conservative_from_primitive::
251  argument_tags{}),
252  typename system::conservative_from_primitive::return_tags{},
253  tmpl::list<system>{});
254  }
255  },
256  make_not_null(&box), db::get<Tags::Time>(box),
257  get<typename Metavariables::boundary_condition_tag>(cache),
258  db::get<Tags::Interface<Tags::BoundaryDirectionsExterior<VolumeDim>,
259  Tags::Coordinates<VolumeDim, Frame::Inertial>>>(
260  box));
261 
262  contribute_data_to_mortar(make_not_null(&box), cache);
263  return std::forward_as_tuple(std::move(box));
264  }
265 };

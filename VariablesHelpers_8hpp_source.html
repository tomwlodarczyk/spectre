39 /*!
40  * \ingroup DataStructuresGroup
41  * \brief Slices the data within `volume_tensor` to a codimension 1 slice. The
42  * slice has a constant logical coordinate in direction `sliced_dim`,
43  * slicing the volume at `fixed_index` in that dimension. For
44  * example, to get the lower boundary of `sliced_dim`, pass `0` for
45  * `fixed_index`; to get the upper boundary, pass
46  * `extents[sliced_dim] - 1`.
47  *
48  * \see add_slice_to_data
49  *
50  * returns Tensor class sliced to a hypersurface.
51  */
52 template <std::size_t VolumeDim, typename VectorType, typename... Structure>
53 void data_on_slice(
54  const gsl::not_null<Tensor<VectorType, Structure...>*> interface_tensor,
55  const Tensor<VectorType, Structure...>& volume_tensor,
56  const Index<VolumeDim>& element_extents, const size_t sliced_dim,
57  const size_t fixed_index) noexcept {
58  const size_t interface_grid_points =
59  element_extents.slice_away(sliced_dim).product();
60  if (interface_tensor->begin()->size() != interface_grid_points) {
61  *interface_tensor = Tensor<VectorType, Structure...>(interface_grid_points);
62  }
63 
64  for (SliceIterator si(element_extents, sliced_dim, fixed_index); si; ++si) {
65  for (decltype(auto) interface_and_volume_tensor_components :
66  boost::combine(*interface_tensor, volume_tensor)) {
67  boost::get<0>(interface_and_volume_tensor_components)[si.slice_offset()] =
68  boost::get<1>(
69  interface_and_volume_tensor_components)[si.volume_offset()];
70  }
71  }
72 }

88 /*!
89  * \ingroup DataStructuresGroup
90  * \brief Slices the data within `vars` to a codimension 1 slice. The
91  * slice has a constant logical coordinate in direction `sliced_dim`,
92  * slicing the volume at `fixed_index` in that dimension. For
93  * example, to get the lower boundary of `sliced_dim`, pass `0` for
94  * `fixed_index`; to get the upper boundary, pass
95  * `extents[sliced_dim] - 1`.
96  *
97  * \see add_slice_to_data
98  *
99  * returns Variables class sliced to a hypersurface.
100  */
101 template <std::size_t VolumeDim, typename TagsList>
102 void data_on_slice(const gsl::not_null<Variables<TagsList>*> interface_vars,
103  const Variables<TagsList>& vars,
104  const Index<VolumeDim>& element_extents,
105  const size_t sliced_dim, const size_t fixed_index) noexcept {
106  const size_t interface_grid_points =
107  element_extents.slice_away(sliced_dim).product();
108  const size_t volume_grid_points = vars.number_of_grid_points();
109  constexpr const size_t number_of_independent_components =
110  Variables<TagsList>::number_of_independent_components;
111 
112  if (interface_vars->number_of_grid_points() != interface_grid_points) {
113  *interface_vars = Variables<TagsList>(interface_grid_points);
114  }
115  using value_type = typename Variables<TagsList>::value_type;
116  const value_type* vars_data = vars.data();
117  value_type* interface_vars_data = interface_vars->data();
118  for (SliceIterator si(element_extents, sliced_dim, fixed_index); si; ++si) {
119  for (size_t i = 0; i < number_of_independent_components; ++i) {
120  // clang-tidy: do not use pointer arithmetic
121  interface_vars_data[si.slice_offset() + // NOLINT
122  i * interface_grid_points] = // NOLINT
123  vars_data[si.volume_offset() + i * volume_grid_points]; // NOLINT
124  }
125  }
126 }

142 /*!
143  * \ingroup DataStructuresGroup
144  * \brief Slices volume `Tensor`s into a `Variables`
145  *
146  * The slice has a constant logical coordinate in direction `sliced_dim`,
147  * slicing the volume at `fixed_index` in that dimension. For
148  * example, to get the lower boundary of `sliced_dim`, pass `0` for
149  * `fixed_index`; to get the upper boundary, pass
150  * `extents[sliced_dim] - 1`.
151  */
152 template <typename... TagsToSlice, size_t VolumeDim>
153 void data_on_slice(
154  const gsl::not_null<Variables<tmpl::list<TagsToSlice...>>*> interface_vars,
155  const Index<VolumeDim>& element_extents, const size_t sliced_dim,
156  const size_t fixed_index,
157  const typename TagsToSlice::type&... tensors) noexcept {
158  const size_t interface_grid_points =
159  element_extents.slice_away(sliced_dim).product();
160  if (interface_vars->number_of_grid_points() != interface_grid_points) {
161  *interface_vars =
162  Variables<tmpl::list<TagsToSlice...>>(interface_grid_points);
163  }
164  for (SliceIterator si(element_extents, sliced_dim, fixed_index); si; ++si) {
165  const auto lambda = [&si](auto& interface_tensor,
166  const auto& volume_tensor) noexcept {
167  for (decltype(auto) interface_and_volume_tensor_components :
168  boost::combine(interface_tensor, volume_tensor)) {
169  boost::get<0>(
170  interface_and_volume_tensor_components)[si.slice_offset()] =
171  boost::get<1>(
172  interface_and_volume_tensor_components)[si.volume_offset()];
173  }
174  };
175  expand_pack((lambda(get<TagsToSlice>(*interface_vars), tensors),
176  cpp17::void_type{})...);
177  }
178 }

193 /*!
194  * \ingroup DataStructuresGroup
195  * \brief Adds data on a codimension 1 slice to a volume quantity. The
196  * slice has a constant logical coordinate in direction `sliced_dim`,
197  * slicing the volume at `fixed_index` in that dimension. For
198  * example, to add to the lower boundary of `sliced_dim`, pass `0` for
199  * `fixed_index`; to add to the upper boundary, pass
200  * `extents[sliced_dim] - 1`.
201  *
202  * \see data_on_slice
203  */
204 template <std::size_t VolumeDim, typename TagsList>
205 void add_slice_to_data(const gsl::not_null<Variables<TagsList>*> volume_vars,
206  const Variables<TagsList>& vars_on_slice,
207  const Index<VolumeDim>& extents, const size_t sliced_dim,
208  const size_t fixed_index) noexcept {
209  constexpr const size_t number_of_independent_components =
210  Variables<TagsList>::number_of_independent_components;
211  const size_t volume_grid_points = extents.product();
212  const size_t slice_grid_points = extents.slice_away(sliced_dim).product();
213  ASSERT(volume_vars->number_of_grid_points() == volume_grid_points,
214  "volume_vars has wrong number of grid points. Expected "
215  << volume_grid_points << ", got "
216  << volume_vars->number_of_grid_points());
217  ASSERT(vars_on_slice.number_of_grid_points() == slice_grid_points,
218  "vars_on_slice has wrong number of grid points. Expected "
219  << slice_grid_points << ", got "
220  << vars_on_slice.number_of_grid_points());
221  using value_type = typename Variables<TagsList>::value_type;
222  value_type* const volume_data = volume_vars->data();
223  const value_type* const slice_data = vars_on_slice.data();
224  for (SliceIterator si(extents, sliced_dim, fixed_index); si; ++si) {
225  for (size_t i = 0; i < number_of_independent_components; ++i) {
226  // clang-tidy: do not use pointer arithmetic
227  volume_data[si.volume_offset() + i * volume_grid_points] += // NOLINT
228  slice_data[si.slice_offset() + i * slice_grid_points]; // NOLINT
229  }
230  }
231 }

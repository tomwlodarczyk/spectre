29 /*!
30  * \ingroup ActionsGroup
31  * \ingroup DiscontinuousGalerkinGroup
32  * \brief Compute the time derivative of the system variables
33  *
34  * The `system::compute_time_derivative` operator is invoked with its
35  * `argument_tags`. The result of this computation is stored in
36  * `db::add_tag_prefix<step_prefix, variables_tag>`, where `step_prefix` is
37  * retrieved from the `Metavariables::temporal_id`. For evolution systems this
38  * prefix is generally `Tags::dt`.
39  *
40  * Uses:
41  * - Metavariables:
42  * - `temporal_id::step_prefix`
43  * - System:
44  * - `variables_tag`
45  * - `compute_time_derivative`
46  * - DataBox:
47  * - db::add_tag_prefix<step_prefix, variables_tag>
48  * - All elements in `compute_time_derivative::argument_tags`
49  *
50  * DataBox changes:
51  * - Modifies:
52  * - `db::add_tag_prefix<step_prefix, variables_tag>`
53  */
54 struct ComputeTimeDerivative {
55  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
56  typename ArrayIndex, typename ActionList,
57  typename ParallelComponent,
58  Requires<tmpl::size<DbTagsList>::value != 0> = nullptr>
59  static std::tuple<db::DataBox<DbTagsList>&&> apply(
60  db::DataBox<DbTagsList>& box,
61  tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
62  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
63  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
64  const ParallelComponent* const /*meta*/) noexcept { // NOLINT const
65  using system = typename Metavariables::system;
66  using computer = typename system::compute_time_derivative;
67  // Notes:
68  // - dt_variables is not zeroed and the operator cannot assume this.
69  // - We retrieve the `step_prefix` from the `Metavariables` (as opposed to
70  // hard-coding `Tags::dt`) to retain consistency with other actions that do
71  // the same (for instance `dg::Actions::ApplyFluxes` that is not specific to
72  // evolution systems)
73  db::mutate_apply<db::split_tag<db::add_tag_prefix<
74  Metavariables::temporal_id::template step_prefix,
75  typename system::variables_tag>>,
76  typename computer::argument_tags>(computer{},
77  make_not_null(&box));
78  return std::forward_as_tuple(std::move(box));
79  }
80 };
81 } // namespace Actions

30 /// Sends to all the Interpolators a list of all the points that need
31 /// to be interpolated onto. Also clears information about data that
32 /// has already been interpolated, since calling this function triggers
33 /// a new interpolation.
34 ///
35 /// Called by InterpolationTargetTag::compute_target_points. This is not an
36 /// Action, but rather a helper function that is called by every
37 /// Action (LineSegment, Strahlkorper, etc.) that specifies a
38 /// particular set of points.
39 template <typename InterpolationTargetTag, typename DbTags,
40  typename Metavariables, size_t VolumeDim, typename Frame>
41 void send_points_to_interpolator(
42  db::DataBox<DbTags>& box, Parallel::ConstGlobalCache<Metavariables>& cache,
43  const tnsr::I<DataVector, VolumeDim, Frame>& target_points,
44  const typename Metavariables::temporal_id::type& temporal_id) noexcept {
45  const auto& domain = db::get<::Tags::Domain<VolumeDim, Frame>>(box);
46  auto coords = block_logical_coordinates(domain, target_points);
47 
48  db::mutate<
49  Tags::IndicesOfFilledInterpPoints, Tags::IndicesOfInvalidInterpPoints,
50  ::Tags::Variables<
51  typename InterpolationTargetTag::vars_to_interpolate_to_target>>(
52  make_not_null(&box),
53  [&coords](
54  const gsl::not_null<db::item_type<Tags::IndicesOfFilledInterpPoints>*>
55  indices_of_filled,
56  const gsl::not_null<
57  db::item_type<Tags::IndicesOfInvalidInterpPoints>*>
58  indices_of_invalid_points,
59  const gsl::not_null<db::item_type<::Tags::Variables<
60  typename InterpolationTargetTag::vars_to_interpolate_to_target>>*>
61  vars_dest) noexcept {
62  // Because we are sending new points to the interpolator,
63  // we know that none of these points have been interpolated to,
64  // so clear the list.
65  indices_of_filled->clear();
66 
67  // Set the indices of invalid points.
68  indices_of_invalid_points->clear();
69  for (size_t i = 0; i < coords.size(); ++i) {
70  if (not coords[i]) {
71  indices_of_invalid_points->insert(i);
72  }
73  }
74 
75  // We will be filling vars_dest with interpolated data.
76  // Here we make sure it is allocated to the correct size.
77  if (vars_dest->number_of_grid_points() != coords.size()) {
78  *vars_dest = db::item_type<::Tags::Variables<
79  typename InterpolationTargetTag::vars_to_interpolate_to_target>>(
80  coords.size());
81  }
82  });
83 
84  auto& receiver_proxy =
85  Parallel::get_parallel_component<Interpolator<Metavariables>>(cache);
86  Parallel::simple_action<Actions::ReceivePoints<InterpolationTargetTag>>(
87  receiver_proxy, temporal_id, std::move(coords));
88 }

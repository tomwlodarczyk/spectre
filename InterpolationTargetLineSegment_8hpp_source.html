117 /// \ingroup ActionsGroup
118 /// \brief Sends points on a line segment to an `Interpolator`.
119 ///
120 /// Uses:
121 /// - DataBox:
122 /// - `::Tags::Domain<3, Frame>`
123 /// - `::Tags::Variables<typename
124 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
125 ///
126 /// DataBox changes:
127 /// - Adds: nothing
128 /// - Removes: nothing
129 /// - Modifies:
130 /// - `Tags::IndicesOfFilledInterpPoints`
131 /// - `::Tags::Variables<typename
132 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
133 ///
134 /// For requirements on InterpolationTargetTag, see InterpolationTarget
135 template <typename InterpolationTargetTag, size_t VolumeDim>
136 struct LineSegment {
137  using const_global_cache_tags =
138  tmpl::list<Tags::LineSegment<InterpolationTargetTag, VolumeDim>>;
139  template <typename ParallelComponent, typename DbTags, typename Metavariables,
140  typename ArrayIndex,
141  Requires<tmpl::list_contains_v<
142  DbTags, Tags::TemporalIds<Metavariables>>> = nullptr>
143  static void apply(
144  db::DataBox<DbTags>& box,
145  Parallel::ConstGlobalCache<Metavariables>& cache,
146  const ArrayIndex& /*array_index*/,
147  const typename Metavariables::temporal_id::type& temporal_id) noexcept {
148  const auto& options =
149  Parallel::get<Tags::LineSegment<InterpolationTargetTag, VolumeDim>>(
150  cache);
151 
152  // Fill points on a line segment
153  const double fractional_distance = 1.0 / (options.number_of_points - 1);
154  tnsr::I<DataVector, VolumeDim, Frame::Inertial> target_points(
155  options.number_of_points);
156  for (size_t n = 0; n < options.number_of_points; ++n) {
157  for (size_t d = 0; d < VolumeDim; ++d) {
158  target_points.get(d)[n] =
159  gsl::at(options.begin, d) +
160  n * fractional_distance *
161  (gsl::at(options.end, d) - gsl::at(options.begin, d));
162  }
163  }
164 
165  send_points_to_interpolator<InterpolationTargetTag>(
166  box, cache, target_points, temporal_id);
167  }
168 };

36 /// \ingroup ActionsGroup
37 /// \brief Initializes an Interpolator
38 ///
39 /// Uses: nothing
40 ///
41 /// DataBox changes:
42 /// - Adds:
43 /// - `Tags::NumberOfElements`
44 /// - `Tags::VolumeVarsInfo<Metavariables>`
45 /// - `Tags::InterpolatedVarsHolders<Metavariables>`
46 /// - Removes: nothing
47 /// - Modifies: nothing
48 struct InitializeInterpolator {
49  template <typename Metavariables>
50  using return_tag_list =
51  tmpl::list<Tags::NumberOfElements,
52  Tags::VolumeVarsInfo<Metavariables>,
53  Tags::InterpolatedVarsHolders<Metavariables>>;
54 
55  template <
56  typename DbTagsList, typename... InboxTags, typename Metavariables,
57  typename ArrayIndex, typename ActionList, typename ParallelComponent,
58  Requires<not tmpl::list_contains_v<DbTagsList, Tags::NumberOfElements> and
59  not tmpl::list_contains_v<
60  DbTagsList, Tags::VolumeVarsInfo<Metavariables>> and
61  not tmpl::list_contains_v<
62  DbTagsList, Tags::InterpolatedVarsHolders<Metavariables>>> =
63  nullptr>
64  static auto apply(db::DataBox<DbTagsList>& box,
65  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
66  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
67  const ArrayIndex& /*array_index*/,
68  const ActionList /*meta*/,
69  const ParallelComponent* const /*meta*/) noexcept {
70  return std::make_tuple(
71  db::create_from<db::RemoveTags<>,
72  db::get_items<return_tag_list<Metavariables>>>(
73  std::move(box), 0_st,
74  db::item_type<Tags::VolumeVarsInfo<Metavariables>>{},
75  db::item_type<Tags::InterpolatedVarsHolders<Metavariables>>{}));
76  }
77 
78  template <
79  typename DbTagsList, typename... InboxTags, typename Metavariables,
80  typename ArrayIndex, typename ActionList, typename ParallelComponent,
81  Requires<tmpl::list_contains_v<DbTagsList, Tags::NumberOfElements> and
82  tmpl::list_contains_v<DbTagsList,
83  Tags::VolumeVarsInfo<Metavariables>> and
84  tmpl::list_contains_v<DbTagsList, Tags::InterpolatedVarsHolders<
85  Metavariables>>> = nullptr>
86  static std::tuple<db::DataBox<DbTagsList>&&> apply(
87  db::DataBox<DbTagsList>& box,
88  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
89  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
90  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
91  const ParallelComponent* const /*meta*/) noexcept {
92  return {std::move(box)};
93  }
94 };

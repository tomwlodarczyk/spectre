46 /*!
47  * \brief Damped harmonic gauge source function.
48  *
49  * \details Computes the gauge source function designed for binary black hole
50  * evolutions. These have been taken from \cite Szilagyi2009qz and
51  * \cite Deppe2018uye.
52  *
53  * The covariant form of the source function \f$H_a\f$ is written as:
54  *
55  * \f{align*}
56  * H_a := [1 - R_{H_\mathrm{init}}(t)] H_a^\mathrm{init} +
57  * [\mu_{L1} \mathrm{log}(\sqrt{g}/N) + \mu_{L2} \mathrm{log}(1/N)] t_a
58  * - \mu_S g_{ai} N^i / N
59  * \f}
60  *
61  * where \f$N, N^k\f$ are the lapse and shift respectively, \f$t_a\f$ is the
62  * unit normal one-form to the spatial slice, and \f$g_{ab}\f$ is
63  * the spatial metric (obtained by projecting the spacetime metric onto the
64  * 3-slice, i.e. \f$g_{ab} = \psi_{ab} + t_a t_b\f$). The prefactors are:
65  *
66  * \f{align*}
67  * \mu_{L1} &= A_{L1} R_{L1}(t) W(x^i) \mathrm{log}(\sqrt{g}/N)^{e_{L1}}, \\
68  * \mu_{L2} &= A_{L2} R_{L2}(t) W(x^i) \mathrm{log}(1/N)^{e_{L2}}, \\
69  * \mu_{S} &= A_{S} R_{S}(t) W(x^i) \mathrm{log}(\sqrt{g}/N)^{e_{S}},
70  * \f}
71  *
72  * temporal roll-on functions \f$ R_X(t)\f$ (with label \f$ X\f$) are:
73  *
74  * \f{align*}
75  * \begin{array}{ll}
76  * R_X(t) & = 1 - \exp[-((t - t_{0,X})/ \sigma_t^X)^4], & t\geq t_{0,X} \\
77  * & = 0, & t< t_{0,X} \\
78  * \end{array}
79  * \f}
80  *
81  * and the spatial weight function is:
82  *
83  * \f{align*}
84  * W(x^i) = \exp[-(r/\sigma_r)^2].
85  * \f}
86  * This weight function can be written with multiple constant factors in the
87  * exponent in literature \cite Deppe2018uye, but we absorb them all into
88  * \f$ \sigma_r\f$ here. The coordinate \f$ r\f$ is the Euclidean radius
89  * in Inertial coordinates.
90  *
91  * Note:
92  * - Amplitude factors \f$ A_{X} \f$ are input as amp\_coef\_X
93  * - \f$ R_{X} \f$ is specified by \f$\{ t_{0,X}, \sigma_t^X\}\f$, which are
94  * input as \f$\{\f$t\_start\_X, sigma\_t\_X \f$\}\f$.
95  * - Exponents \f$ e_X\f$ are input as exp\_X.
96  * - The spatial weight function is specified completely by \f$\{\f$sigma\_r
97  * \f$\}\f$.
98  */
99 template <size_t SpatialDim, typename Frame>

164 /*!
165  * \brief Spacetime derivatives of the damped harmonic gauge source function.
166  *
167  * \details Compute spacetime derivatives, i.e. \f$\partial_a H_b\f$, of the
168  * damped harmonic source function H. Using notation from damped_harmonic_h(),
169  * we rewrite the same as: \f{align*}
170  * \partial_a H_b =& \partial_a T_1 + \partial_a T_2 + \partial_a T_3, \\
171  * H_a =& T_1 + T_2 + T_3,
172  * \f}
173  *
174  * where:
175  *
176  * \f{align*}
177  * T_1 =& [1 - R_{H_\mathrm{init}}(t)] H_a^\mathrm{init}, \\
178  * T_2 =& [\mu_{L1} \mathrm{log}(\sqrt{g}/N) + \mu_{L2} \mathrm{log}(1/N)] t_a,
179  * \\ T_3 =& - \mu_S g_{ai} N^i / N. \f}
180  *
181  * Derivation:
182  *
183  * \f$\blacksquare\f$ For \f$ T_1 \f$, the derivatives are:
184  * \f{align*}
185  * \partial_a T_1 = (1 - R_{H_\mathrm{init}}(t))
186  * \partial_a H_b^\mathrm{init}
187  * - H_b^\mathrm{init} \partial_a R_{H_\mathrm{init}}.
188  * \f}
189  *
190  * \f$\blacksquare\f$ Write \f$ T_2 \equiv (\mu_1 + \mu_2) t_b \f$. Then:
191  * \f{align*}
192  * \partial_a T_2 =& (\partial_a \mu_1 + \partial_a \mu_2) t_b \\
193  * +& (\mu_1 + \mu_2) \partial_a t_b,
194  * \f}
195  * where
196  * \f{align*}
197  * \partial_a t_b =& \left(-\partial_a N, 0, 0, 0\right) \\
198  *
199  * \partial_a \mu_1
200  * =& \partial_a [A_{L1} R_{L1}(t) W(x^i) \mathrm{log}(\sqrt{g}/N)^{e_{L1} +
201  * 1}], \\
202  * =& A_{L1} R_{L1}(t) W(x^i) \partial_a [\mathrm{log}(\sqrt{g}/N)^{e_{L1} +
203  * 1}] \\
204  * +& A_{L1} \mathrm{log}(\sqrt{g}/N)^{e_{L1} + 1} \partial_a [R_{L1}(t)
205  * W(x^i)],\\
206  *
207  * \partial_a \mu_2
208  * =& \partial_a [A_{L2} R_{L2}(t) W(x^i) \mathrm{log}(1/N)^{e_{L2} + 1}], \\
209  * =& A_{L2} R_{L2}(t) W(x^i) \partial_a [\mathrm{log}(1/N)^{e_{L2} + 1}] \\
210  * +& A_{L2} \mathrm{log}(1/N)^{e_{L2} + 1} \partial_a [R_{L2}(t) W(x^i)],
211  * \f}
212  * where \f$\partial_a [R W] = \left(\partial_0 R(t), \partial_i
213  * W(x^j)\right)\f$.
214  *
215  * \f$\blacksquare\f$ Finally, the derivatives of \f$ T_3 \f$ are:
216  * \f[
217  * \partial_a T_3 = -\partial_a(\mu_S/N) g_{bi} N^i
218  * -(\mu_S/N) \partial_a(g_{bi}) N^i
219  * -(\mu_S/N) g_{bi}\partial_a N^i,
220  * \f]
221  * where
222  * \f{align*}
223  * \partial_a(\mu_S / N) =& (1/N)\partial_a \mu_S
224  * - \frac{\mu_S}{N^2}\partial_a N, \,\,\mathrm{and}\\
225  * \partial_a \mu_S =& \partial_a [A_S R_S(t) W(x^i)
226  * \mathrm{log}(\sqrt{g}/N)^{e_S}], \\
227  * =& A_S R_S(t) W(x^i) \partial_a
228  * [\mathrm{log}(\sqrt{g}/N)^{e_S}] \\
229  * +& A_S \mathrm{log}(\sqrt{g} / N)^{e_S} \partial_a [R_S(t)
230  * W(x^i)]. \f}
231  */

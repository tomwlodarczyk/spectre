64 /// `Jacobian(i,0)` is \f$\frac{1}{r}\partial x^i/\partial\theta\f$,

174 /// `NormalOneForm(i)` is \f$s_i\f$, the (unnormalized) normal one-form
175 /// to the surface, expressed in Cartesian components.
176 /// This is computed by \f$x_i/r-\partial r_{\rm surf}/\partial x^i\f$,
177 /// where \f$x_i/r\f$ is `Rhat` and
178 /// \f$\partial r_{\rm surf}/\partial x^i\f$ is `DxRadius`.
179 /// See Eq. (8) of \cite Baumgarte1996hh.
180 /// Note on the word "normal": \f$s_i\f$ points in the correct direction
181 /// (it is "normal" to the surface), but it does not have unit length
182 /// (it is not "normalized"; normalization requires a metric).
183 template <typename Frame>
184 struct NormalOneForm : db::ComputeTag {
185  static std::string name() noexcept { return "NormalOneForm"; }
186  static aliases::OneForm<Frame> function(
187  const db::const_item_type<DxRadius<Frame>>& dx_radius,
188  const db::const_item_type<Rhat<Frame>>& r_hat) noexcept;
189  using argument_tags = tmpl::list<DxRadius<Frame>, Rhat<Frame>>;
190 };

192 /// `Tangents(i,j)` is \f$\partial x_{\rm surf}^i/\partial q^j\f$,
193 /// where \f$x_{\rm surf}^i\f$ are the Cartesian coordinates of the
194 /// surface (i.e. `CartesianCoords`) and are considered functions of
195 /// \f$(\theta,\phi)\f$.
196 ///
197 /// \f$\partial/\partial q^0\f$ means
198 /// \f$\partial/\partial\theta\f$; and \f$\partial/\partial q^1\f$
199 /// means \f$\csc\theta\,\,\partial/\partial\phi\f$. Note that the
200 /// vectors `Tangents(i,0)` and `Tangents(i,1)` are orthogonal to the
201 /// `NormalOneForm` \f$s_i\f$, i.e.
202 /// \f$s_i \partial x_{\rm surf}^i/\partial q^j = 0\f$; this statement
203 /// is independent of a metric. Also, `Tangents(i,0)` and
204 /// `Tangents(i,1)` are not necessarily orthogonal to each other,
205 /// since orthogonality between 2 vectors (as opposed to a vector and
206 /// a one-form) is metric-dependent.
207 template <typename Frame>
208 struct Tangents : db::ComputeTag {
209  static std::string name() noexcept { return "Tangents"; }
210  static aliases::Jacobian<Frame> function(
211  const ::Strahlkorper<Frame>& strahlkorper, const DataVector& radius,
212  const db::const_item_type<Rhat<Frame>>& r_hat,
213  const db::const_item_type<Jacobian<Frame>>& jac) noexcept;
214  using argument_tags = tmpl::list<Strahlkorper<Frame>, Radius<Frame>,
215  Rhat<Frame>, Jacobian<Frame>>;
216 };

242 /// Computes the Euclidean-space integral of a vector over a
243 /// Strahlkorper, \f$\oint V^i s_i (s_j s_k \delta^{jk})^{-1/2} d^2 S\f$,
244 /// where \f$s_i\f$ is the Strahlkorper surface unit normal and
245 /// \f$\delta^{ij}\f$ is the Kronecker delta. Note that \f$s_i\f$ is
246 /// not assumed to be normalized; the denominator of the integrand
247 /// effectively normalizes it using the Euclidean metric.
248 template <typename IntegrandTag, typename Frame>
249 struct EuclideanSurfaceIntegralVector : db::ComputeTag {
250  static std::string name() noexcept {
251  return "EuclideanSurfaceIntegralVector(" + IntegrandTag::name() + ")";
252  }
253  static constexpr auto function =
254  ::StrahlkorperGr::euclidean_surface_integral_of_vector<Frame>;
255  using argument_tags = tmpl::list<EuclideanAreaElement<Frame>, IntegrandTag,
256  StrahlkorperTags::NormalOneForm<Frame>,
257  StrahlkorperTags::Strahlkorper<Frame>>;
258 };

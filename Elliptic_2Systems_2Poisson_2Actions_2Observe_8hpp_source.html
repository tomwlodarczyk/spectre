31 /*!
32  * \brief Action to observe volume and reduction data for the Poisson system
33  *
34  * This action observes the following:
35  * - Reduction data:
36  * - `"Iteration"`: The linear solver iteration
37  * - `"NumberOfPoints"`: The total number of grid points across all elements
38  * - `"L2Error"`: The standard L2 vector norm of the pointwise difference
39  * between the numerical and analytic solution \f$\sqrt{\sum_i \left(
40  * u^\mathrm{numerical}_i - u^\mathrm{analytic}_i\right)^2}\f$ over the grid
41  * points across all elements.
42  * - Volume data:
43  * - `Poisson::Tags::Field::name()`: The numerical solution
44  * \f$u^\mathrm{numerical}\f$
45  * - `Poisson::Tags::Field::name() + "Analytic"`: The analytic solution
46  * \f$u^\mathrm{analytic}\f$
47  * - `Poisson::Tags::Field::name() + "Error"`: The pointwise error
48  * \f$u^\mathrm{numerical} - u^\mathrm{analytic}\f$
49  * - `"InertialCoordinates_{x,y,z}"`
50  *
51  * Uses:
52  * - Metavariables:
53  * - `analytic_solution_tag`
54  * - Items required by `observers::Observer<Metavariables>`
55  * - DataBox:
56  * - `LinearSolver::Tags::IterationId`
57  * - `Tags::Mesh<Dim>`
58  * - `Poisson::Tags::Field`
59  * - `Tags::Coordinates<Dim, Frame::Inertial>`
60  *
61  * \note This action can be adjusted before compiling an executable to observe
62  * only the desired quantities.
63  */
64 struct Observe {
65  private:
66  using observed_reduction_data = Parallel::ReductionData<
67  Parallel::ReductionDatum<size_t, funcl::AssertEqual<>>,
68  Parallel::ReductionDatum<size_t, funcl::Plus<>>,
69  Parallel::ReductionDatum<double, funcl::Plus<>,
70  funcl::Sqrt<funcl::Divides<>>,
71  std::index_sequence<1>>>;
72 
73  public:
74  // Compile-time interface for observers
75  struct ElementObservationType {};
76  template <typename ParallelComponent, typename DbTagsList,
77  typename ArrayIndex>
78  static std::pair<observers::TypeOfObservation, observers::ObservationId>
79  register_info(const db::DataBox<DbTagsList>& box,
80  const ArrayIndex& /*array_index*/) noexcept {
81  return {
82  observers::TypeOfObservation::ReductionAndVolume,
83  observers::ObservationId{
84  static_cast<double>(db::get<LinearSolver::Tags::IterationId>(box)),
85  ElementObservationType{}}};
86  }
87  using observed_reduction_data_tags =
88  observers::make_reduction_data_tags<tmpl::list<observed_reduction_data>>;
89 
90  template <typename... DbTags, typename... InboxTags, typename Metavariables,
91  size_t Dim, typename ActionList, typename ParallelComponent>
92  static auto apply(db::DataBox<tmpl::list<DbTags...>>& box,
93  tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
94  Parallel::ConstGlobalCache<Metavariables>& cache,
95  const ElementIndex<Dim>& array_index,
96  const ActionList /*meta*/,
97  const ParallelComponent* const /*meta*/) noexcept {
98  const auto& iteration_id = get<LinearSolver::Tags::IterationId>(box);
99  const auto& mesh = get<::Tags::Mesh<Dim>>(box);
100  const std::string element_name = MakeString{} << ElementId<Dim>(array_index)
101  << '/';
102 
103  // Retrieve the current numeric solution
104  const auto& field = get<Poisson::Tags::Field>(box);
105 
106  // Compute the analytic solution
107  const auto& inertial_coordinates =
108  db::get<::Tags::Coordinates<Dim, Frame::Inertial>>(box);
109  const auto field_analytic = get<Poisson::Tags::Field>(
110  Parallel::get<typename Metavariables::analytic_solution_tag>(cache)
111  .variables(inertial_coordinates,
112  tmpl::list<Poisson::Tags::Field>{}));
113 
114  // Compute error between numeric and analytic solutions
115  const DataVector field_error = get(field) - get(field_analytic);
116 
117  // Compute l2 error squared over local element
118  const double local_l2_error_square = alg::accumulate(
119  field_error, 0.0, funcl::Plus<funcl::Identity, funcl::Square<>>{});
120 
121  // Collect volume data
122  // Remove tensor types, only storing individual components
123  std::vector<TensorComponent> components;
124  components.reserve(3 + Dim);
125  components.emplace_back(element_name + db::tag_name<Poisson::Tags::Field>(),
126  get(field));
127  components.emplace_back(
128  element_name + db::tag_name<Poisson::Tags::Field>() + "Analytic",
129  get(field_analytic));
130  components.emplace_back(
131  element_name + db::tag_name<Poisson::Tags::Field>() + "Error",
132  field_error);
133  components.emplace_back(element_name + "InertialCoordinates_x",
134  get<0>(inertial_coordinates));
135  if (Dim >= 2) {
136  components.emplace_back(element_name + "InertialCoordinates_y",
137  inertial_coordinates.get(1));
138  }
139  if (Dim >= 3) {
140  components.emplace_back(element_name + "InertialCoordinates_z",
141  inertial_coordinates.get(2));
142  }
143 
144  // Send data to volume observer
145  auto& local_observer =
146  *Parallel::get_parallel_component<observers::Observer<Metavariables>>(
147  cache)
148  .ckLocalBranch();
149  Parallel::simple_action<observers::Actions::ContributeVolumeData>(
150  local_observer,
151  observers::ObservationId(iteration_id, ElementObservationType{}),
152  std::string{"/element_data"},
153  observers::ArrayComponentId(
154  std::add_pointer_t<ParallelComponent>{nullptr},
155  Parallel::ArrayIndex<ElementIndex<Dim>>(array_index)),
156  std::move(components), mesh.extents());
157 
158  // Send data to reduction observer
159  Parallel::simple_action<observers::Actions::ContributeReductionData>(
160  local_observer,
161  observers::ObservationId(iteration_id, ElementObservationType{}),
162  std::string{"/element_data"},
163  std::vector<std::string>{"Iteration", "NumberOfPoints", "L2Error"},
164  observed_reduction_data{iteration_id, mesh.number_of_grid_points(),
165  local_l2_error_square});
166 
167  return std::forward_as_tuple(std::move(box));
168  }
169 };
170 } // namespace Actions
171 } // namespace Poisson

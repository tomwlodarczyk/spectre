20 namespace observers {
21 /*!
22  * \ingroup ObserversGroup
23  * \brief A type-erased identifier that combines the identifier's type
24  * and hash used to uniquely identify an observation inside of a
25  * `h5::File::SUB_FILE`.
26  *
27  * The ObservationId is used to uniquely identify an observation inside our H5
28  * subfiles that can be agreed upon across the system. That is, it does not
29  * depend on hardware rounding of floating points because it is an integral. One
30  * example would be a `Time` for output observed every `N` steps in a global
31  * time stepping evolution. Another example could be having a counter class for
32  * dense output observations that increments the counter for each observation
33  * but has a value equal to the physical time.
34  *
35  * The constructor takes a `double` representing the current "time" at which we
36  * are observing. For an evolution this could be the physical time, while for an
37  * elliptic solve this could be a combination of nonlinear and linear iteration.
38  *
39  * A specialization of `std::hash` is provided to allow using `ObservationId`
40  * as a key in associative containers.
41  */
42 class ObservationId {
43  public:
44  ObservationId() = default;
45 
46  /*!
47  * \brief Construct from a value and an `ObservationType`.
48  *
49  * `ObservationType` is used to distinguish different classes that
50  * do observing, so that messages to ObserverWriter do not collide.
51  * Any class that uses the Observer or ObserverWriter infrastructure
52  * should pass itself (or some other type that uniquely identifies
53  * who is doing the observing) as the ObservationType when creating
54  * `ObservationId`s. If a single class is responsible for different
55  * kinds of observation that can be distinguished only at runtime,
56  * these different kinds of observation should be denoted by passing
57  * the optional `observation_subtype` string.
58  */
59  template <typename ObservationType>
60  explicit ObservationId(double t, const ObservationType& meta,
61  const std::string& observation_subtype = "") noexcept;
62 
63  /// Hash used to distinguish between ObservationIds of different
64  /// types and subtypes. This hash does not contain any information about the
65  /// `Id` or its value.
66  size_t observation_type_hash() const noexcept {
67  return observation_type_hash_;
68  }
69 
70  /// Hash distinguishing different ObservationIds, including
71  /// the value of the `Id`.
72  size_t hash() const noexcept { return combined_hash_; }
73 
74  double value() const noexcept { return value_; }
75 
76  // NOLINTNEXTLINE(google-runtime-references)
77  void pup(PUP::er& p) noexcept;
78 
79  private:
80  size_t observation_type_hash_;
81  size_t combined_hash_;
82  double value_;
83 };

85 template <typename ObservationType>
86 ObservationId::ObservationId(const double t, // NOLINT
87  const ObservationType& /*meta*/,
88  const std::string& observation_subtype) noexcept
89  : observation_type_hash_(std::hash<std::string>{}(
90  pretty_type::get_name<ObservationType>() + observation_subtype)),
91  combined_hash_([&t](size_t type_hash) {
92  size_t combined = type_hash;
93  boost::hash_combine(combined, t);
94  return combined;
95  }(observation_type_hash_)),
96  value_(t) {}

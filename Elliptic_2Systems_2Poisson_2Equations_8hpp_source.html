112 /*!
113  * \brief The internal penalty flux for the first oder formulation of the
114  * Poisson equation.
115  *
116  * \details For the sourced equation this is \f$-\frac{\nabla u_\mathrm{int} +
117  * \nabla u_\mathrm{ext}}{2} + \sigma \left(u_\mathrm{int} -
118  * u_\mathrm{ext}\right)\f$ and for the auxiliary equation it is \f$\frac{
119  * u_\mathrm{int} + u_\mathrm{ext}}{2}\f$. The penalty factor \f$\sigma\f$ is
120  * responsible for removing zero eigenmodes and impacts the conditioning of the
121  * linear operator to be solved. It can be chosen as
122  * \f$\sigma=C\frac{N_\mathrm{points}^2}{h}\f$ where \f$N_\mathrm{points}\f$ is
123  * the number of collocation points (i.e. the polynomial degree plus 1),
124  * \f$h\f$ is a measure of the element size in inertial coordinates and \f$C\leq
125  * 1\f$ is a free parameter (see e.g. \cite HesthavenWarburton, section 7.2).
126  */
127 template <size_t Dim>
128 struct FirstOrderInternalPenaltyFlux {
129  public:
130  struct PenaltyParameter {
131  using type = double;
132  // Currently this is used as the full prefactor to the penalty term. When it
133  // becomes possible to compute a measure of the size $h$ of an element and
134  // the number of collocation points $p$ on both sides of the mortar, this
135  // should be changed to be just the parameter multiplying $\frac{p^2}{h}$.
136  static constexpr OptionString help = {
137  "The prefactor to the penalty term of the flux."};
138  };
139  using options = tmpl::list<PenaltyParameter>;
140  static constexpr OptionString help = {
141  "Computes the internal penalty flux for a Poisson system."};
142  static std::string name() noexcept { return "InternalPenalty"; }
143 
144  FirstOrderInternalPenaltyFlux() = default;
145  explicit FirstOrderInternalPenaltyFlux(double penalty_parameter)
146  : penalty_parameter_(penalty_parameter) {}
147 
148  // clang-tidy: non-const reference
149  void pup(PUP::er& p) noexcept { p | penalty_parameter_; } // NOLINT

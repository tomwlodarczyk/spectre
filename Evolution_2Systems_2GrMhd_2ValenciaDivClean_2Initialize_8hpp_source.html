37 struct InitializeGrTags {
38  using initialization_tags = tmpl::list<Initialization::Tags::InitialTime>;
39 
40  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
41  typename ArrayIndex, typename ActionList,
42  typename ParallelComponent>
43  static auto apply(db::DataBox<DbTagsList>& box,
44  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
45  const Parallel::ConstGlobalCache<Metavariables>& cache,
46  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
47  const ParallelComponent* const /*meta*/) noexcept {
48  using system = typename Metavariables::system;
49  static constexpr size_t dim = system::volume_dim;
50  using gr_tag = typename system::spacetime_variables_tag;
51  using simple_tags = db::AddSimpleTags<gr_tag>;
52  using compute_tags = db::AddComputeTags<>;
53 
54  const double initial_time = db::get<Initialization::Tags::InitialTime>(box);
55  using GrVars = typename gr_tag::type;
56 
57  const size_t num_grid_points =
58  db::get<::Tags::Mesh<dim>>(box).number_of_grid_points();
59  const auto& inertial_coords =
60  db::get<::Tags::Coordinates<dim, Frame::Inertial>>(box);
61 
62  // Set initial data from analytic solution
63  GrVars gr_vars{num_grid_points};
64  make_overloader(
65  [ initial_time, &
66  inertial_coords ](std::true_type /*is_analytic_solution*/,
67  const gsl::not_null<GrVars*> local_gr_vars,
68  const auto& local_cache) noexcept {
69  using solution_tag = ::Tags::AnalyticSolutionBase;
70  local_gr_vars->assign_subset(
71  Parallel::get<solution_tag>(local_cache)
72  .variables(inertial_coords, initial_time,
73  typename GrVars::tags_list{}));
74  },
75  [&inertial_coords](std::false_type /*is_analytic_solution*/,
76  const gsl::not_null<GrVars*> local_gr_vars,
77  const auto& local_cache) noexcept {
78  using analytic_data_tag = ::Tags::AnalyticDataBase;
79  local_gr_vars->assign_subset(
80  Parallel::get<analytic_data_tag>(local_cache)
81  .variables(inertial_coords, typename GrVars::tags_list{}));
82  })(
83  evolution::is_analytic_solution<typename Metavariables::initial_data>{},
84  make_not_null(&gr_vars), cache);
85 
86  return std::make_tuple(
87  Initialization::merge_into_databox<InitializeGrTags, simple_tags,
88  compute_tags>(std::move(box),
89  std::move(gr_vars)));
90  }
91 };

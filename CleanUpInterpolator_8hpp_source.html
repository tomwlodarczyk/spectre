39 /// \ingroup ActionsGroup
40 /// \brief Cleans up stored volume data that is no longer needed.
41 ///
42 /// Called by InterpolationTargetReceiveVars.
43 ///
44 /// Uses:
45 /// - Databox:
46 /// - `Tags::InterpolatedVarsHolders<Metavariables>`
47 /// - `Tags::VolumeVarsInfo<Metavariables>`
48 ///
49 /// DataBox changes:
50 /// - Adds: nothing
51 /// - Removes: nothing
52 /// - Modifies:
53 /// - `Tags::InterpolatedVarsHolders<Metavariables>`
54 /// - `Tags::VolumeVarsInfo<Metavariables>`
55 ///
56 /// For requirements on InterpolationTargetTag, see InterpolationTarget
57 template <typename InterpolationTargetTag>
58 struct CleanUpInterpolator {
59  template <
60  typename ParallelComponent, typename DbTags, typename Metavariables,
61  typename ArrayIndex,
62  Requires<tmpl::list_contains_v<DbTags, Tags::NumberOfElements>> = nullptr>
63  static void apply(
64  db::DataBox<DbTags>& box, // HorizonManager's box
65  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
66  const ArrayIndex& /*array_index*/,
67  const typename Metavariables::temporal_id::type& temporal_id) noexcept {
68  // Signal that this InterpolationTarget is done at this time.
69  db::mutate<Tags::InterpolatedVarsHolders<Metavariables>>(
70  make_not_null(&box),
71  [&temporal_id](
72  const gsl::not_null<
73  db::item_type<Tags::InterpolatedVarsHolders<Metavariables>>*>
74  holders) noexcept {
75  get<Vars::HolderTag<InterpolationTargetTag, Metavariables>>(*holders)
76  .temporal_ids_when_data_has_been_interpolated.insert(temporal_id);
77  });
78 
79  // If we don't need any of the volume data anymore for this
80  // temporal_id, we will remove them.
81  bool this_temporal_id_is_done = true;
82  const auto& holders =
83  db::get<Tags::InterpolatedVarsHolders<Metavariables>>(box);
84  tmpl::for_each<typename Metavariables::interpolation_target_tags>(
85  [&](auto tag) noexcept {
86  using Tag = typename decltype(tag)::type;
87  const auto& found = get<Vars::HolderTag<Tag, Metavariables>>(holders)
88  .temporal_ids_when_data_has_been_interpolated;
89  if (found.count(temporal_id) == 0) {
90  this_temporal_id_is_done = false;
91  }
92  });
93 
94  // We don't need any more volume data for this temporal_id,
95  // so remove it.
96  if (this_temporal_id_is_done) {
97  db::mutate<Tags::VolumeVarsInfo<Metavariables>>(
98  make_not_null(&box), [&temporal_id](
99  const gsl::not_null<db::item_type<
100  Tags::VolumeVarsInfo<Metavariables>>*>
101  volume_vars_info) noexcept {
102  volume_vars_info->erase(temporal_id);
103  });
104 
105  // Clean up temporal_ids_when_data_has_been_interpolated
106  db::mutate<Tags::InterpolatedVarsHolders<Metavariables>>(
107  make_not_null(&box),
108  [&temporal_id](
109  const gsl::not_null<
110  db::item_type<Tags::InterpolatedVarsHolders<Metavariables>>*>
111  holders_l) noexcept {
112  tmpl::for_each<typename Metavariables::interpolation_target_tags>(
113  [&](auto tag) noexcept {
114  using Tag = typename decltype(tag)::type;
115  get<Vars::HolderTag<Tag, Metavariables>>(*holders_l)
116  .temporal_ids_when_data_has_been_interpolated.erase(
117  temporal_id);
118  });
119  });
120  }
121  }
122 };
123 } // namespace Actions
124 } // namespace intrp

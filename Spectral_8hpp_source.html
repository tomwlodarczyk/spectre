152 /*!
153  * \brief Weights to compute definite integrals.
154  *
155  * \details These are the coefficients to contract with the nodal
156  * function values \f$f_k\f$ to approximate the definite integral \f$I[f]=\int
157  * f(x)\mathrm{d}x\f$.
158  *
159  * Note that the term _quadrature_ also often refers to the quantity
160  * \f$Q[f]=\int f(x)w(x)\mathrm{d}x\approx \sum_k f_k w_k\f$. Here, \f$w(x)\f$
161  * denotes the basis-specific weight function w.r.t. to which the basis
162  * functions \f$\Phi_k\f$ are orthogonal, i.e \f$\int\Phi_i(x)\Phi_j(x)w(x)=0\f$
163  * for \f$i\neq j\f$. The weights \f$w_k\f$ approximate this inner product. To
164  * approximate the definite integral \f$I[f]\f$ we must employ the
165  * coefficients \f$\frac{w_k}{w(\xi_k)}\f$ instead, where the \f$\xi_k\f$ are
166  * the collocation points. These are the coefficients this function returns.
167  * Only for a unit weight function \f$w(x)=1\f$, i.e. a Legendre basis, is
168  * \f$I[f]=Q[f]\f$ so this function returns the \f$w_k\f$ identically.
169  *
170  * \param num_points The number of collocation points
171  */
172 template <Basis BasisType, Quadrature QuadratureType>
173 const DataVector& quadrature_weights(size_t num_points) noexcept;

204 /*!
205  * \brief %Matrix used to perform an indefinite integral of a function over the
206  * logical grid. The left boundary condition is such that the integral is 0 at
207  * \f$\xi=-1\f$
208  *
209  * Currently only Legendre and Chebyshev polynomials are implemented, but we
210  * provide a derivation for how to compute the indefinite integration matrix for
211  * general Jacobi polynomials.
212  *
213  * #### Legendre Polynomials
214  * The Legendre polynomials have the identity:
215  *
216  * \f{align*}{
217  * P_n(x) = \frac{1}{2n+1}\frac{d}{dx}\left(P_{n+1}(x) - P_{n-1}(x)\right)
218  * \f}
219  *
220  * The goal is to evaluate the integral of a function \f$u\f$ expanded in terms
221  * of Legendre polynomials as
222  *
223  * \f{align*}{
224  * u(x) = \sum_{i=0}^{N} c_i P_i(x)
225  * \f}
226  *
227  * We similarly expand the indefinite integral of \f$u\f$ as
228  *
229  * \f{align*}{
230  * \left.\int u(y) dy\right\rvert_{x}=&\sum_{i=0}^N \tilde{c}_i P_i(x) \\
231  * =&\left.\int\sum_{i=1}^{N}\frac{c_i}{2i+1}
232  * \left(P_{i+1}(y)-P_{i-1}(y)\right)dy\right\rvert_{x}
233  * + \tilde{c}_0 P_0(x) \\
234  * =&\sum_{i=1}^{N}\left(\frac{c_{i-1}}{2i-1} - \frac{c_{i+1}}{2i+3}\right)
235  * P_i(x) + \tilde{c}_0 P_0(x)
236  * \f}
237  *
238  * Thus we get that for \f$i>0\f$
239  *
240  * \f{align*}{
241  * \tilde{c}_i=\frac{c_{i-1}}{2i-1}-\frac{c_{i+1}}{2i+3}
242  * \f}
243  *
244  * and \f$\tilde{c}_0\f$ is a constant of integration, which we choose such that
245  * the integral is 0 at the left boundary of the domain (\f$x=-1\f$). The
246  * condition for this is:
247  *
248  * \f{align*}{
249  * \tilde{c}_0=\sum_{i=1}^{N}(-1)^{i+1}\tilde{c}_i
250  * \f}
251  *
252  * The matrix returned by this function is the product of the tridiagonal matrix
253  * for the \f$\tilde{c}_i\f$ and the matrix for the boundary condition.
254  *
255  * #### Chebyshev Polynomials
256  *
257  * A similar derivation leads to the relations:
258  *
259  * \f{align*}{
260  * \tilde{c}_i=&\frac{c_{i-1}-c_{i+1}}{2i},&\mathrm{if}\;i>1 \\
261  * \tilde{c}_1=&c_0 - \frac{c_2}{2},&\mathrm{if}\;i=1 \\
262  * \f}
263  *
264  * We again have:
265  *
266  * \f{align*}{
267  * \tilde{c}_0=\sum_{i=1}^N(-1)^{i+1}\tilde{c}_i
268  * \f}
269  *
270  * These are then used to define the indefinite integration matrix.
271  *
272  * #### Jacobi Polynomials
273  *
274  * For general Jacobi polynomials \f$P^{(\alpha,\beta)}_n(x)\f$ given by
275  *
276  * \f{align*}{
277  * (1-x)^\alpha(1+x)^\beta P^{(\alpha,\beta)}_n(x)=\frac{(-1)^n}{2^n n!}
278  * \frac{d^n}{dx^n}\left[(1-x)^{\alpha+n}(1+x)^{\beta+n}\right]
279  * \f}
280  *
281  * we have that
282  *
283  * \f{align*}{
284  * P^{(\alpha,\beta)}_n(x)=\frac{d}{dx}\left(
285  * b^{(\alpha,\beta)}_{n-1,n}P^{(\alpha,\beta)}_{n-1}(x) +
286  * b^{(\alpha,\beta)}_{n,n}P^{(\alpha,\beta)}_n(x) +
287  * b^{(\alpha,\beta)}_{n+1,n}P^{(\alpha,\beta)}_{n+1}(x)
288  * \right)
289  * \f}
290  *
291  * where
292  *
293  * \f{align*}{
294  * b^{(\alpha,\beta)}_{n-1,n}=&-\frac{1}{n+\alpha+\beta}
295  * a^{(\alpha,\beta)}_{n-1,n} \\
296  * b^{(\alpha,\beta)}_{n,n}=&-\frac{2}{\alpha+\beta}
297  * a^{(\alpha,\beta)}_{n,n} \\
298  * b^{(\alpha,\beta)}_{n+1,n}=&\frac{1}{n+1}
299  * a^{(\alpha,\beta)}_{n+1,n} \\
300  * a^{(\alpha,\beta)}_{n-1,n}=&\frac{2(n+\alpha)(n+\beta)}
301  * {(2n+\alpha+\beta+1)(2n+\alpha+\beta)} \\
302  * a^{(\alpha,\beta)}_{n,n}=&-\frac{\alpha^2-\beta^2}
303  * {(2n+\alpha+\beta+2)(2n+\alpha+\beta)} \\
304  * a^{(\alpha,\beta)}_{n-1,n}=&\frac{2(n+1)(n+\alpha+\beta+1)}
305  * {(2n+\alpha+\beta+2)(2n+\alpha+\beta+1)}
306  * \f}
307  *
308  * Following the same derivation we get that
309  *
310  * \f{align*}{
311  * \tilde{c}_i=c_{i+1}b^{(\alpha,\beta)}_{i,i+1}
312  * +c_i b^{(\alpha,\beta)}_{i,i}
313  * +c_{i-1}b^{(\alpha,\beta)}_{i,i-1}
314  * \f}
315  *
316  * and the boundary condition is
317  *
318  * \f{align*}{
319  * \tilde{c}_0=\sum_{i=1}^N(-1)^{i+1}
320  * \frac{\Gamma(i+\alpha+1)}{i!\Gamma(\alpha+1)} \tilde{c}_i
321  * \f}
322  *
323  * where \f$\Gamma(x)\f$ is the Gamma function.
324  */

360 /*!
361  * \brief %Matrix used to transform from the spectral coefficients (modes) of a
362  * function to its nodal coefficients. Also referred to as the _Vandermonde
363  * matrix_.
364  *
365  * \details The Vandermonde matrix is computed as
366  * \f$\mathcal{V}_{ij}=\Phi_j(\xi_i)\f$ where the \f$\Phi_j(x)\f$ are the
367  * spectral basis functions used in the modal expansion
368  * \f$u(x)=\sum_j \widetilde{u}_j\Phi_j(x)\f$, e.g. normalized Legendre
369  * polynomials, and the \f$\xi_i\f$ are the collocation points used to construct
370  * the interpolating Lagrange polynomials in the nodal expansion
371  * \f$u(x)=\sum_j u_j l_j(x)\f$. Then the Vandermonde matrix arises as
372  * \f$u(\xi_i)=u_i=\sum_j \widetilde{u}_j\Phi_j(\xi_i)=\sum_j
373  * \mathcal{V}_{ij}\widetilde{u}_j\f$.
374  *
375  * \param num_points The number of collocation points
376 
377  * \see nodal_to_modal_matrix(size_t)
378  */
379 template <Basis BasisType, Quadrature QuadratureType>
380 const Matrix& modal_to_nodal_matrix(size_t num_points) noexcept;

390 /*!
391  * \brief %Matrix used to transform from the nodal coefficients of a function to
392  * its spectral coefficients (modes). Also referred to as the inverse
393  * _Vandermonde matrix_.
394  *
395  * \details This is the inverse to the Vandermonde matrix \f$\mathcal{V}\f$
396  * computed in modal_to_nodal_matrix(size_t). It can be computed
397  * analytically for Gauss quadrature by evaluating
398  * \f$\sum_j\mathcal{V}^{-1}_{ij}u_j=\widetilde{u}_i=
399  * \frac{(u,\Phi_i)}{\gamma_i}\f$
400  * for a Lagrange basis function \f$u(x)=l_k(x)\f$ to find
401  * \f$\mathcal{V}^{-1}_{ij}=\mathcal{V}_{ji}\frac{w_j}{\gamma_i}\f$ where the
402  * \f$w_j\f$ are the Gauss quadrature weights and \f$\gamma_i\f$ is the norm
403  * square of the spectral basis function \f$\Phi_i\f$.
404  *
405  * \param num_points The number of collocation points
406  *
407  * \see modal_to_nodal_matrix(size_t)
408  */

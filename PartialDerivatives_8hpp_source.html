136 /*!
137  * \ingroup DataBoxTagsGroup
138  * \brief Compute the spatial derivatives of tags in a Variables
139  *
140  * Computes the spatial derivatives of the Tensors in the Variables represented
141  * by `VariablesTag` in the frame mapped to by `InverseJacobianTag`. To only
142  * take the derivatives of a subset of these Tensors you can set the
143  * `DerivTags` template parameter. It takes a `tmpl::list` of the desired
144  * tags and defaults to the full `tags_list` of the Variables.
145  *
146  * This tag may be retrieved via `db::variables_tag_with_tags_list<VariablesTag,
147  * DerivTags>` prefixed with `Tags::deriv`.
148  */
149 template <typename VariablesTag, typename InverseJacobianTag,
150  typename DerivTags =
151  typename db::const_item_type<VariablesTag>::tags_list>
152 struct DerivCompute
153  : db::add_tag_prefix<
154  deriv, db::variables_tag_with_tags_list<VariablesTag, DerivTags>,
155  tmpl::size_t<tmpl::back<typename db::const_item_type<
156  InverseJacobianTag>::index_list>::dim>,
157  typename tmpl::back<typename db::const_item_type<
158  InverseJacobianTag>::index_list>::Frame>,
159  db::ComputeTag {
160  private:
161  using inv_jac_indices =
162  typename db::const_item_type<InverseJacobianTag>::index_list;
163  static constexpr auto Dim = tmpl::back<inv_jac_indices>::dim;
164  using deriv_frame = typename tmpl::back<inv_jac_indices>::Frame;
165 
166  public:
167  static constexpr ::Variables<db::wrap_tags_in<
168  Tags::deriv, DerivTags, tmpl::size_t<Dim>, deriv_frame>> (*function)(
169  const ::Variables<typename db::const_item_type<VariablesTag>::tags_list>&,
170  const ::Mesh<Dim>&,
171  const ::InverseJacobian<DataVector, Dim, Frame::Logical, deriv_frame>&) =
172  partial_derivatives<DerivTags,
173  typename db::const_item_type<VariablesTag>::tags_list,
174  Dim, deriv_frame>;
175  using argument_tags =
176  tmpl::list<VariablesTag, Tags::Mesh<Dim>, InverseJacobianTag>;
177 };

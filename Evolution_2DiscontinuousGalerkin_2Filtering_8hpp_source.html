68 /*!
69  * \ingroup DiscontinuousGalerkinGroup
70  * \brief Applies an exponential filter to the specified tags.
71  *
72  * Applies an exponential filter in each logical direction to each component of
73  * the tensors `TagsToFilter`. The exponential filter rescales the 1d modal
74  * coefficients \f$c_i\f$ as:
75  *
76  * \f{align*}{
77  * c_i\to c_i \exp\left[-\alpha_{\mathrm{ef}}
78  * \left(\frac{i}{N}\right)^{2\beta_{\mathrm{ef}}}\right]
79  * \f}
80  *
81  * where \f$N\f$ is the basis order, \f$\alpha_{\mathrm{ef}}\f$ determines how
82  * much the coefficients are rescaled, and \f$\beta_{\mathrm{ef}}\f$ (given by
83  * the `HalfPower` option) determines how aggressive/broad the filter is (lower
84  * values means filtering more coefficients). Setting
85  * \f$\alpha_{\mathrm{ef}}=36\f$ results in effectively zeroing the highest
86  * coefficient (in practice it gets rescaled by machine epsilon). The same
87  * \f$\alpha_{\mathrm{ef}}\f$ and \f$\beta_{\mathrm{ef}}\f$ are used in each
88  * logical direction. For a discussion of filtering see section 5.3 of
89  * \cite HesthavenWarburton.
90  *
91  * If different `Alpha` or `HalfPower` parameters are desired for different tags
92  * then multiple `ExponentialFilter` actions must be inserted into the action
93  * list with different `FilterIndex` values. In the input file these will be
94  * specified as `ExpFilterFILTER_INDEX`, e.g. `ExpFilter0` and `ExpFilter1`.
95  * Below is an example of an action list with two different exponential filters:
96  *
97  * \snippet DiscontinuousGalerkin/Test_Filtering.cpp action_list_example
98  *
99  * #### Action properties
100  *
101  * Uses:
102  * - ConstGlobalCache:
103  * - `ExponentialFilter`
104  * - DataBox:
105  * - `Tags::Mesh`
106  * - DataBox changes:
107  * - Adds: nothing
108  * - Removes: nothing
109  * - Modifies:
110  * - `TagsToFilter`
111  * - System:
112  * - `volume_dim`
113  * - `variables_tag`
114  *
115  * #### Design decision:
116  *
117  * - The reason for the `size_t` template parameter is to allow for different
118  * `Alpha` and `HalfPower` parameters for different tensors while still being
119  * able to cache the matrices.
120  */
121 template <size_t FilterIndex, typename... TagsToFilter>
122 class ExponentialFilter<FilterIndex, tmpl::list<TagsToFilter...>> {
123  public:
124  using const_global_cache_tags = tmpl::list<ExponentialFilter>;
125 
126  /// \brief The value of `exp(-alpha)` is what the highest modal coefficient is
127  /// rescaled by.
128  struct Alpha {
129  using type = double;
130  static constexpr OptionString help =
131  "exp(-alpha) is rescaling of highest coefficient";
132  static type lower_bound() noexcept { return 0.0; }
133  };
134  /*!
135  * \brief Half of the exponent in the exponential.
136  *
137  * \f{align*}{
138  * c_i\to c_i \exp\left[-\alpha \left(\frac{i}{N}\right)^{2m}\right]
139  * \f}
140  */
141  struct HalfPower {
142  using type = unsigned;
143  static constexpr OptionString help =
144  "Half of the exponent in the generalized Gaussian";
145  static type lower_bound() noexcept { return 1; }
146  };
147  /// \brief Turn the filter off
148  ///
149  /// This option exists to temporarily disable the filter for debugging
150  /// purposes. For problems where filtering is not needed, the preferred
151  /// approach is to not compile the filter into the executable.
152  struct DisableForDebugging {
153  using type = bool;
154  static type default_value() noexcept { return false; }
155  static constexpr OptionString help = {"Disable the filter"};
156  };
157 
158  using type = ExponentialFilter;
159  static std::string name() noexcept {
160  return "ExpFilter" + std::to_string(FilterIndex);
161  }
162  using group = ::OptionTags::FilteringGroup;
163 
164  using options = tmpl::list<Alpha, HalfPower, DisableForDebugging>;
165  static constexpr OptionString help = {"An exponential filter."};
166 
167  ExponentialFilter() = default;

169  ExponentialFilter(double alpha, unsigned half_power,
170  bool disable_for_debugging) noexcept;

172  // Action part of the class
173  template <typename DbTags, typename... InboxTags, typename ArrayIndex,
174  typename ActionList, typename ParallelComponent,
175  typename Metavariables>
176  static std::tuple<db::DataBox<DbTags>&&> apply(
177  db::DataBox<DbTags>& box,
178  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
179  const Parallel::ConstGlobalCache<Metavariables>& cache,
180  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
181  const ParallelComponent* const /*meta*/) noexcept {
182  constexpr size_t volume_dim = Metavariables::system::volume_dim;
183  using evolved_vars_tag = typename Metavariables::system::variables_tag;
184  using evolved_vars_tags_list = typename evolved_vars_tag::tags_list;
185  const ExponentialFilter& filter_helper =
186  Parallel::get<ExponentialFilter>(cache);
187  if (UNLIKELY(filter_helper.disable_for_debugging_)) {
188  return {std::move(box)};
189  }
190  const Mesh<volume_dim> mesh = db::get<::Tags::Mesh<volume_dim>>(box);
191  const Matrix empty{};
192  auto filter = make_array<volume_dim>(std::cref(empty));
193  for (size_t d = 0; d < volume_dim; d++) {
194  gsl::at(filter, d) =
195  std::cref(filter_helper.filter_matrix(mesh.slice_through(d)));
196  }
197 
198  // In the case that the tags we are filtering are all the evolved variables
199  // we filter the entire Variables at once to be more efficient. This case is
200  // the first branch of the `if-else`.
201  if (ExponentialFilter_detail::FilterAllEvolvedVars<
202  sizeof...(TagsToFilter) ==
203  tmpl::size<evolved_vars_tags_list>::value,
204  cpp17::is_same_v<evolved_vars_tags_list,
205  tmpl::list<TagsToFilter...>>>::
206  template f<evolved_vars_tags_list, TagsToFilter...>::value) {
207  db::mutate<typename Metavariables::system::variables_tag>(
208  make_not_null(&box),
209  [&filter](
210  const gsl::not_null<
211  db::item_type<typename Metavariables::system::variables_tag>*>
212  vars,
213  const auto& local_mesh) noexcept {
214  *vars = apply_matrices(filter, *vars, local_mesh.extents());
215  },
216  mesh);
217  } else {
218  db::mutate<TagsToFilter...>(
219  make_not_null(&box),
220  [&filter](const gsl::not_null<
221  db::item_type<TagsToFilter>*>... tensors_to_filter,
222  const auto& local_mesh) noexcept {
223  DataVector temp(local_mesh.number_of_grid_points(), 0.0);
224  const auto helper =
225  [&local_mesh, &filter, &temp ](const auto tensor) noexcept {
226  for (auto& component : *tensor) {
227  temp = 0.0;
228  apply_matrices(make_not_null(&temp), filter, component,
229  local_mesh.extents());
230  component = temp;
231  }
232  };
233  EXPAND_PACK_LEFT_TO_RIGHT(helper(tensors_to_filter));
234  },
235  mesh);
236  }
237  return {std::move(box)};
238  }
